HDIRL=Z:\xh\xharbour
BCDIR=Z:\xh\bcc582
FWDIR=Z:\xh\fwh

.path.OBJ = .\obj
.path.CH = $(FWDIR)\include;$(HDIRL)\include
PRG= \

...

OBJ=$(PRG:.prg=.obj)
OBJS=$(OBJ:.\=.obj)

PROJECT : xxxx.exe

veterinaria.exe : $(OBJS)
echo off
echo $(BCDIR)\lib\c0w32.obj + > b32.bc

echo xxx.res >> b32.bc

$(BCDIR)\bin\ilink32 -Gn -aa -Tpe -s @b32.bc

.PRG.OBJ:
$(HDIRL)\bin\harbour $< /N /W /Oobj\ /I$(FWDIR)\include;$(HDIRL)\include;.\include
$(BCDIR)\bin\bcc32 -c -tWM -I$(HDIRL)\include -oobj\$& obj\$&.c

.C.OBJ:
echo -c -tWM -D__HARBOUR__ -DHB_API_MACROS > tmp
echo -I$(HDIRL)\include;$(FWDIR)\include >> tmp
$(BCDIR)\bin\bcc32 -oobj\$& @tmp $&.c

del tmp

veterinaria.res : IF EXIST xxx.rc
$(BCDIR)\bin\brc32.exe -r xxx.rc
