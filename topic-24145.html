 function lGetPassword(oWnd)
   local oDlg, oGet, oFont
   local cPassword := Space( 10 )
   local nTries    := 0
   local lGo       := .f.
   LOCAL MY_PASSWORD:="HATHAL" // WRITE ANY PASSOWD 
 DO WHILE lGo=.f.
   DEFINE DIALOG oDlg ;
      FROM 5, 5 TO 30, 80 ;
