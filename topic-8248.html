Static Function FnMarca(oBrw, i)
Local n := oBrw:nArrayAt
Local nBmp
If oBrw:aArrayData[n, 3] = "S"   // mi tercera columna es el CheckOn/Check Off
    nBmp := 3      // Parece que los resursos 1 y 2 son los indicadores de sort up/down
Else
    nBmp := 4
Endif
Return nBmp
