José +- algo asi:

Code (fw): Select all Collapse
#Include "FiveWin.ch"

STATIC xDllTef

FUNCTION PIN_PAD()

   LOCAL cResultado, cMensagem, cTCTrilha1, cTrilha1, cTCTrilha2, cTrilha2, ;
         cTimeOut, xResp

   xDllTef := LOADLIBRARY( "NOMBREDLLPINPAD.DLL" )  // OR .LIB

   cResultado := Space(6)
   cMensagem  := "Insira ou Passe"
   cTCTrilha1 := Space(12)
   cTrilha1   := Space(128)
   cTCTrilha2 := Space(12)
   cTrilha2   := Space(64)
   cTimeOut   := "3"

   xResp := LeCartao( cResultado, cMensagem, cTCTrilha1, cTrilha1,;
                      cTCTrilha2, cTrilha2, cTimeout )

   ? xResp

RETURN NIL

DLL32 FUNCTION LeCartao(pResultado AS LPSTR,pMensagem AS LPSTR,TCTrilha1 AS LPSTR, ;
Trilha1 AS LPSTR, TCTrilha2 AS LPSTR, Trilha2 AS LPSTR, Timeout AS LPSTR) AS LONG PASCAL ;
FROM "LeCartaoDiretoSeguroA" LIB xDllTef

// FIN
