A working sample:

Code (fw): Select all Collapse
#include "Fivewin.ch"


FUNCTION MAIN()

    LOCAL oWnd

    DEFINE WINDOW oWnd

    ACTIVATE WINDOW oWnd;
             VALID ( oWnd := NIL, .T. )

    STOPUNTIL( { || oWnd = NIL } )

    ? "The end!"

    RETURN NIL
