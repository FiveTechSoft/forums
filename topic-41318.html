Solo ajusta este achivo en los fuentes de TDolphin

Setenvh.bat
@CALL CLEANENV.BAT
echo on
set path=c:\vc2022\bin32;c:\harbour_msvc\bin
set lib=c:\vc2022\lib\win32
set include=c:\vc2022\include
SET PRG_COMPILER=HARBOUR
SET PRG_COMP_PATH=c:\Harbour_msvc
SET PRG_COMP_BIN_PATH=%PRG_COMP_PATH%\bin
SET PRG_COMP_LIB_PATH=%PRG_COMP_PATH%\lib\win\msvc
SET PRG_COMP_INC_PATH=%PRG_COMP_PATH%\include
SET C_COMPILER=MSVC32
SET C_COMP_PATH=c:\vc2022
SET C_COMP_BIN_PATH=%C_COMP_PATH%\bin32
SET C_COMP_LIB_PATH=%C_COMP_PATH%\lib\win32
SET C_COMP_INC_PATH=%C_COMP_PATH%\include
SET DOLPHIN_INC=.\include
@CALL win-make
Dejame saber si te funciono
