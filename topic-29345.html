This is a working sample:

Code (fw): Select all Collapse
#include "Fivewin.ch"


FUNCTION MAIN()

    LOCAL oPrn

    PRINT oPrn PREVIEW
        PAGE
            SETBKCOLOR( oPrn:hDCOut, CLR_HRED )
            oPrn:Say( 0, 0, "This is a test", , , CLR_WHITE, 2 )
        ENDPAGE
    ENDPRINT

    RETURN NIL
