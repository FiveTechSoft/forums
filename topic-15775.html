Buscando por este mismo foro, encontré la solución dada por ukoenig del foro en ingles, y reprodusco aquí por si a alguien le sirve. La función debe ser llamada en el ON INIT del diálogo

Code (fw): Select all Collapse
// ---------------------  Disable EXIT - BUTTON -----------------

FUNCTION DisableX(oDlg, lDisable) 

LOCAL hMenu  := 0 
LOCAL nCount := 0 

IF lDisable 
    hMenu  = GetSystemMenu(oDlg:hWnd, .F.) 
    nCount = GetMItemCount(hMenu) 
    IF oDlg:ClassName() = "TDIALOG" 
        RemoveMenu(hMenu, 1, nOR( 1024, 2) ) 
    ELSE 
        RemoveMenu(hMenu, nCount - 1, nOR( 1024, 2) ) 
        RemoveMenu(hMenu, nCount - 2, nOR( 1024, 2) ) 
    ENDIF     
    DrawMenuBar( oDlg:hWnd ) 
ELSE 
    GetSystemMenu( oDlg:hWnd, .T. ) 
    DrawMenuBar( oDlg:hWnd ) 
ENDIF 

RETURN nil
