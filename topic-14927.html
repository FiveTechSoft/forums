Hola Francisco,

El ejemplo es un bastante antiguo pero puede ser de referencia, si necesitas más dimelo.

//----------------------------------------------------------------------------//

Function lisficha(rad,para)

 STATIC oReport
 LOCAL oFont1, oFont2

 if valtype(para) == 'C'
    IF VAL(PARA) = 0
       IF !MsgYesNo(&quot;Ha seleccionado el total del Club,  ; Desea Continuar ?&quot;, ;
          &quot;Seleccione, Por Favor&quot;)
          RETURN NIL
       ENDIF
    ENDIF
 ELSE
    IF PARA = 0
       IF !MsgYesNo(&quot;Ha seleccionado el total del Club,  ; Desea Continuar ?&quot;, ;
          &quot;Seleccione, Por Favor&quot;)
          RETURN NIL
       ENDIF
    ENDIF
 ENDIF

 DEFINE FONT oFont1 NAME &quot;ARIAL&quot; SIZE 0,-10 ITALIC
 DEFINE FONT oFont2 NAME &quot;ARIAL&quot; SIZE 0,-20 BOLD ITALIC

 USE FILATEMO ALIAS FILATEMO INDEX FILATEMO NEW
 SELECT FILATEMO
 IF RAD = 1 .AND. VAL(PARA) != 0
    SET RELATION TO FILATEMO -&gt; FILAMAT INTO CLIENTES
    COPY TO TEMPORAL.DBF FOR (CLIENTES -&gt; CODIGO = PARA)
    CLOSE FILATEMO
    USE TEMPORAL ALIAS FILATEMO NEW
 ENDIF
 IF RAD = 2 .AND. PARA != 0
    COPY TO TEMPORAL.DBF FOR (FILATEMO -&gt; FILAMAT = PARA)
    CLOSE FILATEMO
    USE TEMPORAL ALIAS FILATEMO NEW
 ENDIF
 USE DEPEN ALIAS DEPENDE INDEX DEPEN NEW
 SELECT CLIENTES
 SET RELATION TO CLIENTES-&gt;CODIGO INTO DEPENDE
 USE TABLAFIL ALIAS TABLAFIL INDEX TABLAFIL NEW
 SELECT FILATEMO
 SET RELATION TO filatemo-&gt;filacod INTO tablafil, ;
              TO filatemo-&gt;filamat INTO clientes ADDITIVE

 DBGOTOP()

IF CPREVIO = 'S'
REPORT oReport ;
FONT oFont1,oFont2;
HEADER " "," "," ", ;
SPACE(30)+" FICHA DE REGISTRO DE SOCIOS ", ;
" "," "," "," ", ;
"DEPENDENCIA... "+CLIENTES->CODIGO+' - '+DEPENDE->DEPEN ,;
"N. MATRICULA.. "+STR(CLIENTES->NUMATRI), ;
"NOMBRE........ "+CLIENTES->NOMBRE, ;
"APELLIDOS..... "+OemToAnsi(CLIENTES->APELLIDOS), ;
"DOMICILIO..... "+CLIENTES->DOMICILIO, ;
"MUNICIPIO..... "+CLIENTES->MUNICIPIO, ;
"PROVINCIA..... "+CLIENTES->PROVINCIA+ ;
" D.P.: "+STR(CLIENTES->DISPOST), ;
"SITUACION..... "+CLIENTES->ESTADO, ;
"FECHA DE ALTA. "+DTOC(CLIENTES->FECHALTA), ;
" "," "," " ;
FOOTER OemToAnsi('Club Filat‚lico de Tabacalera, S.A.') ;
CENTER ;
CAPTION "Fichas de Socios" ;
PREVIEW
ELSE
REPORT oReport ;
FONT oFont1,oFont2;
HEADER " "," "," ", ;
SPACE(30)+" FICHA DE REGISTRO DE SOCIOS ", ;
" "," "," "," ", ;
"DEPENDENCIA... "+CLIENTES->CODIGO+' - '+DEPENDE->DEPEN ,;
"N. MATRICULA.. "+STR(CLIENTES->NUMATRI), ;
"NOMBRE........ "+CLIENTES->NOMBRE, ;
"APELLIDOS..... "+CLIENTES->APELLIDOS, ;
"DOMICILIO..... "+CLIENTES->DOMICILIO, ;
"MUNICIPIO..... "+CLIENTES->MUNICIPIO, ;
"PROVINCIA..... "+CLIENTES->PROVINCIA+ ;
" D.P.: "+STR(CLIENTES->DISPOST), ;
"SITUACION..... "+CLIENTES->ESTADO, ;
"FECHA DE ALTA. "+DTOC(CLIENTES->FECHALTA), ;
" "," "," " ;
FOOTER OemToAnsi('Club Filat‚lico de Tabacalera, S.A.') ;
CENTER ;
CAPTION "Fichas de Socios"
ENDIF

 COLUMN TITLE &quot;COD.&quot; ;
        DATA  (filatemo-&gt;filacod)

 COLUMN TITLE &quot;PRODUCTOS SOLICITADOS&quot; ;
        DATA  OemToAnsi(tablafil-&gt;denomina)

 COLUMN TITLE &quot;CANTIDAD&quot; ;
        DATA  EDINUM(filatemo-&gt;filacan,&quot;999,999&quot;)

 IF RAD &gt; 0 .AND. RAD &lt; 3
    GROUP ON FILATEMO-&gt;FILAMAT ;
          FOOTER &quot;&quot; ;
          EJECT
 ENDIF
 oReport:nGroupLine := RPT_NOLINE   // no imprime linea separadora

// oReport:oHeader:aFont[3] := { || 2 }
END REPORT

 ACTIVATE REPORT oReport ;
      ON STARTPAGE (LinBox(oReport,1.3,1.3,.6,.5), ;
                    Bitmaps(oReport,.3,.3,1,1))

CLOSE FILATEMO
CLOSE TABLAFIL
CLOSE DEPENDE
oFont1:End()
oFont2:End()

RETURN NIL
//----------------------------------------------------------------------------//

Saludos,
