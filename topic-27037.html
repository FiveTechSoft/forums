// dbcre8 (inspired on dbcre8 from successware)

#include "FiveWin.ch"

static aFirst, aLast, aStreet, aCity, aState

function Main()

   local oDlg, nRecords := 10000, cFileName := "database.dbf"

   SetDlgGradient( { { 1, RGB( 199, 216, 237 ), RGB( 237, 242, 248 ) } } )
   
   DEFINE DIALOG oDlg TITLE "DbCre8" SIZE 400, 200
   
   @ 1, 5 SAY "Amount of records:" 
   
   @ 1.2, 10 GET nRecords PICTURE "999999999" SIZE 40, 12

   @ 2, 8.6 SAY "Filename:" 

   @ 2.3, 10 GET cFileName SIZE 70, 12

   @ 4.4, 8 BUTTON "&Ok" SIZE 40, 13 ACTION BuildDbf( nRecords, cFileName )
   
   @ 4.4, 18 BUTTON "&Cancel" ACTION oDlg:End()

   ACTIVATE DIALOG oDlg CENTERED

return nil

function BuildDbf( nRecords, cFileName )

   local n := 1
   
   InitArrays()
   
   DbCreate( cFileName, { { "FIRST",   "C", 20, 0 },;
                          { "LAST",    "C", 20, 0 },;
                          { "STREET",  "C", 30, 0 },;
                          { "CITY",    "C", 30, 0 },;
                          { "STATE",   "C",  2, 0 },;
                          { "MARRIED", "L",  1, 0 } } )
                          
   USE ( cFileName )
   
   while n++ < nRecords
      APPEND BLANK
      Field->First   := aFirst[ Max( nRandom( Len( aFirst ) ), 1 ) ]
      Field->Last    := aLast[ Max( nRandom( Len( aLast ) ), 1 ) ]
      Field->Street  := aStreet[ Max( nRandom( Len( aStreet ) ), 1 ) ]
      Field->City    := aCity[ Max( nRandom( Len( aCity ) ), 1 ) ]
      Field->State   := aState[ Max( nRandom( Len( aState ) ), 1 ) ]
      Field->Married := nRandom( 10 ) % 2 == 0
   end      
   
   USE
   
   XBrowser( cFileName )
   
return nil
