Code: Select all Collapse
LOCAL oError

TRY
     //operacion que puede fallar
CATCH oError
     //mensaje de error
     Alert("Error: " + oError:description )
FINALLY
     //mensaje indicando que todo esta bien
     Alert("La operacion se realizo corrctamente")
END
