al inicio de tu programa debes poner

#require &quot;hbnetio&quot;

#xtranslate net:&lt;!func!&gt;( [&lt;params,...&gt;] ) =&gt; ;
            netio_FuncExec( #&lt;func&gt; [,&lt;params&gt;] )
#xtranslate net:[&lt;server&gt;]:&lt;!func!&gt;( [&lt;params,...&gt;] ) =&gt; ;
            netio_FuncExec( [ #&lt;server&gt; + ] &quot;:&quot; + #&lt;func&gt; [,&lt;params&gt;] )
#xtranslate net:[&lt;server&gt;]:&lt;port&gt;:&lt;!func!&gt;( [&lt;params,...&gt;] ) =&gt; ;
            netio_FuncExec( [ #&lt;server&gt; + ] &quot;:&quot; + #&lt;port&gt; + &quot;:&quot; + #&lt;func&gt; ;
                            [,&lt;params&gt;] )

#xtranslate net:exists:&lt;!func!&gt; =&gt; ;
            netio_ProcExists( #&lt;func&gt; )
#xtranslate net:exists:[&lt;server&gt;]:&lt;!func!&gt; =&gt; ;
            netio_ProcExists( [ #&lt;server&gt; + ] &quot;:&quot; + #&lt;func&gt; )
#xtranslate net:exists:[&lt;server&gt;]:&lt;port&gt;:&lt;!func!&gt; =&gt; ;
            netio_ProcExists( [ #&lt;server&gt; + ] &quot;:&quot; + #&lt;port&gt; + &quot;:&quot; + #&lt;func&gt; )
