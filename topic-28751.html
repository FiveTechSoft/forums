Añade este código al final de tu PRG principal:

Code (fw): Select all Collapse
#pragma BEGINDUMP

#include <windows.h>
#include <hbapi.h>

HB_FUNC( LOADLIBRARY )
{
   hb_retnl( ( LONG ) LoadLibrary( hb_parc( 1 ) ) );
}

HB_FUNC( FREELIBRARY )
{
   hb_retl( FreeLibrary( ( HMODULE ) hb_parnl( 1 ) ) );
}

#pragma ENDDUMP
