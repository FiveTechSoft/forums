include "fivewin.ch"

function main()

ACTION ( Nextlast(), ;
cNum:= TestNum->Num, oNum:Refresh())

        @ 0, 1 say &quot;Number:&quot;

        @  0, 10 say oNum VAR cNum OF oDlg SIZE 45, 11 COLOR CLR_GREEN

ACTIVATE DIALOG oDlg CENTERED

RETURN NIL

STATIC Function Toplast()

GOTO TOP

return nil

STATIC Function Nextlast()

TestNum-&gt;(dbskip())
if eof()
   msginfo(&quot;No More Record Found&quot;)
   GO BOTTOM

   IF bof()
      msginfo(&quot;No Previous Record Found&quot;)
   GOTO TOP
endif

return nil

GO BOTTOM
