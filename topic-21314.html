it is simply

DEFINE CLIPBOARD oClp OF oApp():oDlg

MENU oMenu POPUP 2007
....

MENUITEM "Copia il codice utente in memoria";
RESOURCE "SM_CLIP1";
ACTION oClp:SetText( (cDBF)->CLUSUARIO )

  MENUITEM &quot;Copia la password in memoria&quot;  ;
   RESOURCE &quot;SM_CLIP2&quot;;
  ACTION oClp:SetText( (cDBF)-&gt;CLCLAVE )

  MENUITEM &quot;Vedere la stringa in memoria&quot;  ;
   RESOURCE &quot;SM_CLIP3&quot;;
  ACTION  GetSomeText(oClp)

  SEPARATOR

ENDMENU
ACTIVATE POPUP oMenu
