With FWMYSQL

oQry := oServer:Query( "SELECT * FROM clientes where cliente >= 10" )

oQry:SetWhere( "cliente >= 20" )


can be written as
Code (fw): Select all Collapse
oQry := oServer:Query( "SELECT FROM clientes WHERE cliente >= ?", { 10 } )

// later change the age

oQry:ReQuery( { 20 } )


Code (fw): Select all Collapse
oQry := oServer:Query( "SELECT * FROM `miTabla` WHERE `miCampoFecha` BETWEEN ? AND ?", { dMin, dMax } )

// change values of dMin and dMax

oQry:ReQuery( { dMin, dMax } )
