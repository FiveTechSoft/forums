FUNCTION PRUEBA_PRINTER()



LOCAL nRow := 1

LOCAL cImpresoraT := "POSPrinter POS-80C"




PRINT oPrn FILE "E:\TEMP\MIPDF.PDF" // TO ALLTRIM(cImpresoraT)




define font oFont name "New Roman" size 0, -04 of oPrn

PAGE

FOR nRow := 1 TO 300

@ nRow, .1 PRINT TO oPrn TEXT STR(nRow,3) ;
SIZE 7.1,.5 CM FONT oFont ALIGN "C" LASTROW nRow

NEXT

ENDPAGE

ENDPRINT




RETURN
