   LOCAL oIni

   INI oIni FILE cIni
      SET SECTION cSection ;
         ENTRY cEntry      ;

   ENDINI

FUNCTION GetIni( cIni, cSection, cEntry, xDefault )

   LOCAL oIni
   LOCAL xVar := xDefault

   DEFAULT cIni := oApp:cIniFile

   INI oIni FILE cIni
      GET xVar            ;
         SECTION cSection ;
         ENTRY cEntry     ;

   ENDINI
