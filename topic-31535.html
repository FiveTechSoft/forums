#xcommand MENU [ <oObjMenu> ] ;
             [ <of: OF, WINDOW, DIALOG, CONTROL> <oCtrl> ] ;
             [ <popup: POPUP> ] ;
             [ <l2007: 2007, _2007> ] ;
             [ <l2010: 2010, _2010> ] ;
             [ COLORMENU [ <nClrMenu>, ] <nClrMenutxt> ] ;
             [ COLORSELECT  <nClrHiLi> ,<nClrHiLf> ,<nClrHitx> ] ;

             [ <lMenuBar: MENUBAR> ] ;
             [ FONT <oFont> ] ;
             [ <lColors: COLORS> ] ;
             [ <lVert: LINESVERT> ] ;
             [ <l2013: 2013, _2013> ] ;
             [ <l2015: 2015, _2015> ] ;
             [ <lNoBorder: NOBORDER> ] ;
             [ RIGHTSPACE <nRightSp> ] ;
             [ FACTOR <nFactor> ] ;
             [ HEIGHT <nHeight> ] ;
             [ <lInherit: NOINHERIT> ] ;
             [ <lInvert: INVERT> ] ;
