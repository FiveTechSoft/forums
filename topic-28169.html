Added a TIME-field to my new Xbrowse-sample :



Code (fw): Select all Collapse
// FIELD ("ABC")->TIMESET = C 8  => "10.12.05"

oBrw:aCols[ 11 ]:cHeader    := "Time"
oBrw:aCols[ 11 ]:oDataFont  := oFont2
oBrw:aCols[ 11 ]:oEditFont  := oFont2
oBrw:aCols[ 11 ]:nHeadStrAlign  := AL_CENTER
oBrw:aCols[ 11 ]:nDataStrAlign  := AL_CENTER
oBrw:aCols[ 11 ]:bStrData   := { || ("ABC")->TIMESET }
oBrw:aCols[ 11 ]:bPopUp := { |o| ColMenu1( o, oBrw ) } 
oBrw:aCols[ 11 ]:nWidth := 80
oBrw:aCols[ 11 ]:bToolTip   := {| oBrw, nRow, nCol, nKeyFlag | ;
     IF( oBrw:MouseRowPos(nRow) == oBrw:nRowSel, ;
          "Change time" + CRLF + ;
          "with left" + CRLF + ;
          "mouseclick", ) }

// ---------

STATIC FUNCTION COLMENU1( oCol, oBrw )
LOCAL oPop

MENU oPop POPUP 2007
     MENUITEM "Add time" ;
          ACTION ( NET_RLOCK( 3, 3 ), ;
          ("ABC")->TIMESET := TIME(), ;
          NET_ULOCK(), ;
         oBrw:RefreshCurrent() )
     MENUITEM "Delete time" ;
         ACTION ( NET_RLOCK( 3, 3 ), ;
         ("ABC")->TIMESET := "--.--.--", ;
         NET_ULOCK(), ;
         oBrw:RefreshCurrent() )
ENDMENU

RETURN oPop

Lucas,

You can EXTRACT a timestring from => DateTime() to use in xBrowse :

dt := DateTime()
cHours := ALLTRIM(STR( Hour( dt ) )) + ":"
cMinutes := ALLTRIM(STR( Minute( dt ) )) + ":"
cSeconds := ALLTRIM(STR( Secs( dt ) ))

cTime := cHours + cMinutes + cSeconds
