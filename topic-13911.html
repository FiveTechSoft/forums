FUNCTION SayNum(oGet,xNum)
LOCAL cText := Alltrim(oGet:cText())
if ValType(xNum) == "C"
   if xNum = "."
      oGet:cText( padR(alltrim(cText)+".",100,' ') )
   elseif xNum = '<'
      oGet:cText( padR(left(alltrim(cText),len(cText)-1),100,' ') )
   end
   if xNum <> '.' .and. xNum <> '<'
      xNum := val(xNum)
      oGet:cText( padR(alltrim(cText)+cValToChar(xNum),100,' ') )
   end
else
   oGet:cText( padR(alltrim(cText)+cValToChar(xNum),100,' ') )
end
oGet:SetFocus()
Return Nil
