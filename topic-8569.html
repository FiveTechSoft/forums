Function DIA_SEMM( )

      Case dow( pAG_DATA ) = 1

      Case dow( pAG_DATA ) = 2
         tDIA_SEM := 'Segunda'
      Case dow( pAG_DATA ) = 3
         tDIA_SEM := 'Terça  '
      Case dow( pAG_DATA ) = 4
         tDIA_SEM := 'Quarta '
      Case dow( pAG_DATA ) = 5
         tDIA_SEM := 'Quinta '
      case dow( pAG_DATA ) = 6
         tDIA_SEM := 'Sexta  '
      Case dow( pAG_DATA ) = 7
         tDIA_SEM := 'Sábado '
   ENDCASE
   oDIA_SEM:Refresh()

Return nil
