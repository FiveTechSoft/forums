Or, back to the clipper days ...
I wrote convert program something like this

STATIC FUNCTION Convert_CP1250(kaj)
//*** Big letters
kaj = IF("¬" $ kaj,STRTRAN(kaj,"¬","Č"),kaj)
kaj = if("¦" $ kaj,STRTRAN(kaj,"¦","Ž"),kaj)
kaj = if("ć" $ kaj,STRTRAN(kaj,"ć","Š"),kaj)
kaj = if("Ź" $ kaj,STRTRAN(kaj,"Ź","Ć"),kaj)
kaj = if("Ń" $ kaj,STRTRAN(kaj,"Ń","Đ"),kaj)

 //*** small letters 
 kaj = if(&quot;ź&quot; $ kaj,STRTRAN(kaj,&quot;ź&quot;,&quot;č&quot;),kaj)
 kaj = if(&quot;§&quot; $ kaj,STRTRAN(kaj,&quot;§&quot;,&quot;ž&quot;),kaj)
 kaj = if(&quot;ç&quot; $ kaj,STRTRAN(kaj,&quot;ç&quot;,&quot;š&quot;),kaj)
 kaj = if(&quot;†&quot; $ kaj,STRTRAN(kaj,&quot;†&quot;,&quot;ć&quot;),kaj)
 kaj = if(&quot;Đ&quot; $ kaj,STRTRAN(kaj,&quot;Đ&quot;,&quot;đ&quot;),kaj)

RETURN kaj
