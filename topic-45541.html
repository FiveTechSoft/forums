Aqui tienes el ejemplo con el boton de "Cerrar"


Code: Select all Collapse
</s><i>
</i>#include "fivewin.ch"

FUNCTION Main()
     LOCAL i
     
     REQUEST DBFCDX
     RddSetDefault( "DBFCDX" )
     set DATE FRENCH
     SET CENTURY ON
     SET EPOCH TO Year( Date() ) - 40

     USE c:\fwh\amples\customer NEW SHARED
     
     XBROWSER "customer" TITLE "Prueba" SETUP ;
               ( oBrw:bInit :=  { | o | ;
                    o:l2013 := .T., ;
                    o:l2010 := .F., ;
                    o:l2015 := .F., ;
                    o:aControls[ 5 ]:cTooltip  := o:aControls[ 9 ]:cToolTip, ;
                    o:aControls[ 5 ]:bAction   := o:aControls[ 9 ]:bAction, ;
                    o:aControls[ 5 ]:LoadBitmaps( FWBitmap( "Excel" ) ), ;
                    o:aControls[ 6 ]:cTooltip  := "Exit", ;
                    o:aControls[ 6 ]:bAction   := { | oBtt | oBtt:End() }, ;
                    o:aControls[ 6 ]:LoadBitmaps( FWBitmap( "exit2" ) ), ;  // close16
                    o:aControls[ 6 ]:lRight    := .T., ;
                    AEval( o:aControls, { | oBtt | oBtt:End() }, 7, 9 ), ; 
                    ASize( o:aControls, 6 ), ; 
                    AEval( o:aControls, { | oBtt | oBtt:Refresh() } ), ;
                    o:SetGradients(), ;
                    o:Refresh() ;
                  } )
RETURN nil
