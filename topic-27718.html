Example

Code (fw): Select all Collapse
#include "fivewin.ch"

function Main()
        LOCAL   nLen
        
        nLen := MyTest( "Example" )    
        
        MsgStop( nLen )    
RETURN NIL


#pragma BEGINDUMP

#include "hbapi.h"
#include "hbapiitm.h"


HB_FUNC( MYTEST )
{
        PHB_ITEM myItem = hb_itemNew(NULL) ;
        ULONG nRet ;
        const char *pStr = hb_parc(1) ;
        
        hb_itemPutC( myItem, pStr ) ;
        
        nRet = hb_itemGetCLen( myItem );
        
        hb_itemRelease( myItem );
        
        hb_retnl( nRet ) ;
}

#pragma ENDDUMP
