@ 100, 100 BTNBMP oBtn1 OF oFld:aDialogs[ 1 ] ;

oBtn1:bClrGrad = { | lMouseOver | If( ! lMouseOver,;

oBtn1:lTransparent := .t.   
oBtn1:cToolTip =  { "Exit" + CRLF + "Foldertest","EXIT", 1, CLR_BLACK, 14089979 }
