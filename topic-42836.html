got it :)



you need to change in *.MAK
Code (fw): Select all Collapse
</s>   echo $(HBDIR)\lib\win\bcc\hbvm.lib + >> b32.bc
   echo $(BCDIR)\lib\cw32.lib + >> b32.bc<e>

into (add mt in LIB Name)
Code (fw): Select all Collapse
</s>   echo $(HBDIR)\lib\win\bcc\hbvmmt.lib + >> b32.bc
   echo $(BCDIR)\lib\cw32mt.lib + >> b32.bc<e>

now you can use "mt LIB" for "multi-threading" support under Fivewin with *.MAK (BCC)



this is my *.MAK for BCC 32 Bit
Code (fw): Select all Collapse
</s>HBDIR=c:\harbour
BCDIR=c:\BCC7
FWDIR=c:\fwh

#change these paths as needed
.path.OBJ = .\obj
.path.PRG = .\

.path.C   = .\

DUALGRID.PRG \
TGRID.PRG    \
SHELLAPP.PRG \
HBEJECT.PRG  \
CODEPAGE.PRG \
IPHONE.PRG   \
PROPERTY.PRG \
EVERYTH.PRG  \
HB_FUNC.PRG
