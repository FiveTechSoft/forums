//------------------------
Func _PFormat( cPh, oPh )

Local xPhone

If cPh = "  "
   Return(.t.)
Endif

cPh := alltrim( cPh )

// already formatted  "(999)-999-9999"
If len( cPh ) = 10
Else
   Return(.t.)
Endif

xPhone := Transform( cPh, "@R (999)-999-9999")
xPhone := substr(xPhone+space(15),1,15)

cPh := xPhone
oPh:ReFresh()
SysReFresh()

Return(.t.)
