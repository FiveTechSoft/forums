Requiero enviar correos en forma automatica, usando una cuenta del GMail y usando el servidor
de correos smtp.gmail.com con puerto 465 y seguridad SSL.

Alguien sabe de alguna forma facil de enviarlos.

Estoy usando este ejemplo que encontre, pero no funciona con Harbour 3.2 , alguna idea o ayuda ?

#include &quot;FiveWin.ch&quot;

function Main()

   local oMsg  := CreateObject( &quot;CDO.Message&quot; )
   local oConf := CreateObject( &quot;CDO.Configuration&quot; )
   local cSchema := &quot;http://schemas.microsoft.com/cdo/configuration/&quot;

   oConf:Fields[ cSchema + &quot;sendusing&quot; ] = 2
   oConf:Fields[ cSchema + &quot;smtpserver&quot; ] = &quot;smtp.gmail.com&quot;
   oConf:Fields[ cSchema + &quot;smtpserverport&quot; ] = 465
   oConf:Fields[ cSchema + &quot;smtpauthenticate&quot; ] = 1
   oConf:Fields[ cSchema + &quot;sendusername&quot; ] = &quot;juan@gmail.com&quot;
   oConf:Fields[ cSchema + &quot;sendpassword&quot; ] = &quot;MiPass123&quot;
   oConf:Fields[ cSchema + &quot;smtpusessl&quot; ] = 1
   oConf:Fields:Update()

   oMsg:To = &quot;pedrito@ventas.com&quot;
   oMsg:From =&quot;juan@gmail.com&gt;&quot;
   oMsg:Subject = &quot;Prueba de correos con una cuenta gmail&quot;
   oMsg:HTMLBody = &quot;aqui va el texto del mensaje&quot;
   oMsg:Sender = &quot;juan&quot;
   oMsg:ReplyTo = &quot;maria@ventas.com&quot;
   oMsg:Configuration = oConf

   oMsg:Send()
   MsgInfo( &quot;Correo enviado OK !!&quot; )

return nil
