</s><i>
</i>   IF( XRESOLUCAO = "R1024_768" )

      // PARA RESOLUCAO: 1024 X 768
      DEFINE WINDOW oWndChild MDICHILD FROM 00, 00 TO 36.7, 127.9               ;
         COLOR CLR_CYAN, CLR_WHITE                                              ;
         TITLE cTitle  MDI BORDER NONE NOMAXIMIZE // NOMINIMIZE // prevent resizing

      // ESTA EM: AMBIENT3.RES
      DEFINE DIALOG oDlg RESOURCE "FOLDER_PROPOSTA_1024_768" OF oWndChild GRADIENT aGrad

      REDEFINE FOLDEREX oFld ID 101 OF oDlg UPDATE                   ;
         PROMPTS "&Proposta", "Se&rviços", "&Cliente"                ;
         BITMAPS "Agenda1", "Agenda2"                                ;

                 "FLD_PROPOSTA2"                                     ;
