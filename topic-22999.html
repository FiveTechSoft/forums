I would not call it a bug, but a missing source code

Code (fw): Select all Collapse
HB_FUNC( GETCURDIR )
{
   DWORD dwLen = GetCurrentDirectory( 0, NULL );
   
   if( dwLen )
   {
      char * pBuffer = ( char * ) hb_xgrab( dwLen );
      
      GetCurrentDirectory( dwLen, pBuffer );
      
      hb_retc( pBuffer );
      hb_xfree( pBuffer );
   }
   else
      hb_retc( "" );
}
