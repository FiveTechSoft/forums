Code (fw): Select all Collapse
MENU oMenu FONT oFnt1 2013 ADJUST ;

         COLORSELECT METRO_STEEL, METRO_STEEL, RGB( 0, 0, 1 ) ;
         COLORLEFT CLR_WHITE ;

         ROUND 1 ;  //5   //      RIGHTSPACE 10 ;
         HEIGHT 1.0 * 2
