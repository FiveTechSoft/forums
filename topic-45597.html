</s>#include "fivewin.ch"

REQUEST FWZEBRA  // required for barcodes

function Main()

   local oPrn, cText := "Texto a QR"

   PRINT oPrn
   PAGE
   @ 2,2 PRINT TO oPrn TEXT cText AS BARCODE TYPE "QR" SIZE 2,2 INCHES
   ENDPAGE
   ENDPRINT

return nil<e>
