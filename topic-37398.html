ESEMPIO FUNZIONANANTE FINO al 30/6/19 *

function mail_dct()
parameter _from,_to,_password,_subject,_textbody,_arrattach,_smtpserver

LOCAL oEmailMsg
LOCAL cSchema := "http://schemas.microsoft.com/cdo/configuration/"
LOCAL _j,_cfile

oEmailMsg := CREATEOBJECT ( "CDO.Message" )
WITH OBJECT oEmailMsg
:From := alltrim(_from)
:To := alltrim(_to)

     :CC       := &quot;&quot;
     :BCC      := &quot;&quot;

     :Subject  := _subject
     :TextBody := _textbody

 for _j:=1 to len(_arrattach)
  _cfile:=_arrattach[_j]
  :AddAttachment(_cfile)   
 next _j

     WITH OBJECT :configuration:Fields
        :Item( cSchema + &quot;smtpserver&quot; ):Value       := alltrim(_smtpserver)

        :Item( cSchema + &quot;smtpserverport&quot; ):Value   := 465

        :Item( cSchema + &quot;sendusing&quot; ):Value        := 2

        :Item( cSchema + &quot;smtpauthenticate&quot; ):Value := .T.

        :Item( cSchema + &quot;smtpusessl&quot; ):Value       := .T.

        :Item( cSchema + &quot;sendusername&quot; ):Value     := alltrim(_from)

        :Item( cSchema + &quot;sendpassword&quot; ):Value     := alltrim(_password)

        :Item( cSchema + &quot;smtpconnectiontimeout&quot;):Value := 30

        :Update()

     END WITH

END WITH
*FINE
