WITH OBJECT oBrw
:lDisplayZeros := .F.
:nMarqueeStyle := MARQSTYLE_HIGHLROWRC
:lMultiSelect := .t.
:nStretchCol := STRETCHCOL_WIDEST
:bClrStd := {|| { RGB(0,0,0),colores( aItems[ oBrw:nat, 4 ] ) } }
:lColDividerComplete := .T.
:lAllowColHiding := .F.
:lRecordSelector := .F.
END WITH
