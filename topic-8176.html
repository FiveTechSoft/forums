"\\" is just for other machines on the LAN.

Please add this code to your PRG:
Code: Select all Collapse
#pragma BEGINDUMP

#include <hbapi.h>
#include <WinSock2.h>

HB_FUNC( WSASTARTUP )
{
   static WSADATA wsa;

   hb_retni( WSAStartup( 0x202, &wsa ) );
}

#pragma ENDDUMP
