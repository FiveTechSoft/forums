    LOCAL cEnc := ""

    LOCAL cChr

    LOCAL i

    FOR i = 1 TO LEN( cXml )
        cChr = SUBSTR( cXml, i, 1 )

        DO CASE
            CASE cChr = "<"
                cEnc += "<"
            CASE cChr = ">"
                cEnc += ">"
            CASE cChr = "&"
                cEnc += "&amp;"
            CASE cChr = "'"
                cEnc += "&apos;"
            CASE cChr = ["]
                cEnc += "&quot;"
            OTHERWISE
                cEnc += cChr
        ENDCASE
    NEXT

    RETURN cEnc
