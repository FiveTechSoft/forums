cuatecatl82:

Aqui tienes una base para hacer tooltip personalizados:

Code (fw): Select all Collapse
               FOR nFor := 1 TO Len( oGrid:aCols )

                   oGrid:aCols[nFor]:bToolTip := {| oGrid, nRow, nCol, nKeyFlags| If( oGrid:MouseRowPos( nRow ) == oGrid:nRowSel,;
                                                 ( 'Información del Registro de Tomas:'+CRLF+CRLF+;
                                                   'Fecha:        ' +;
                                                   TransForm( oGrid:aCols[4]:Value(), "99-99-9999" )+CRLF+;
                                                   'Hora:         ' +;
                                                   TransForm( oGrid:aCols[4]:Value(), "99:99:99" )+CRLF+;
                                                   'Ing. Papilla(ml):' +;
                                                   LTrim( TransForm( oGrid:aCols[4]:Value(), "@Z 999,999.99 (ml)" ))+CRLF+;
                                                   'Peso/Día(g):    ' +;
                                                   LTrim( TransForm( oGrid:aCols[4]:Value(), "@Z 999,999.99 (g)" ))+CRLF+;
                                                   'Edad:           ' +;
                                                   LTrim( TransForm( oGrid:aCols[4]:Value(), "@Z 999,999" ))),)}
                   SysRefresh()

               NEXT
