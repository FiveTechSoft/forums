{
       PangoFontDescription *fontDesc;
    if (hb_parnl(2) == 1) {
       fontDesc = pango_font_description_from_string( &quot;courier italic 10&quot; );
    } else if (hb_parnl(2) == 2) {
       fontDesc = pango_font_description_from_string( &quot;courier bold italic 10&quot; );
    } else { 
       fontDesc = pango_font_description_from_string( &quot;courier bold 10&quot; );
    }; 
    gtk_widget_modify_font((GtkWidget *) hb_parnl( 1 ), fontDesc);
}
