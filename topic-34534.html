#include 'fivewin.ch'

function main
 
   local oDlg, oGet[2], oFnt
   local cUser, cPass

   cUser := space(10)
   cPass := space(10)

   DEFINE FONT oFnt NAME "Tahoma" SIZE 0, -20

   DEFINE DIALOG oDlg FROM 0, 0 TO 400, 400 PIXEL FONT oFnt

   @ 30, 30 GET oGet[1] VAR cUser OF oDlg SIZE 100, 12 PIXEL NOBORDER UPDATE ;
            COLOR CLR_BLUE, CLR_WHITE FONT oFnt  CUEBANNER "User Id"

   // oGet[1]:SetCueBanner(.T.,"User Id")

   @ 60, 30 GET oGet[2] VAR cPass OF oDlg SIZE 100, 12 PIXEL NOBORDER UPDATE ;

   // oGet[2]:SetCueBanner(.T.,"Password")

   ACTIVATE DIALOG oDlg CENTER

return nil

En el fichero RC añadir:

1 24 "WindowsXP.Manifest"

Si se está usando FWH 64 bits:

1 24 "WindowsXP.Manifest64"
