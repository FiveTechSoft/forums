// Testing GETs - \SAMPLES\ABURTO.PRG

#include "FiveWin.ch"

function Main()

   LOCAL bPant, oGet, oGet2, oGet3, oOk, oSalida
   LOCAL cCad := "Testing    "
   LOCAL nNum := 0
   LOCAL dDat := Date()
   LOCAL xWS1 := SPACE(01)
   LOCAL xWS2 := SPACE(01)

   Set century On
   Set Date Ansi
   Set Date format "mm/dd/yyyy"

   SET _3DLOOK ON

   DEFINE DIALOG bPant TITLE "TGet from " + FWDESCRIPTION

   @ 1,    2 SAY "Text..:" OF bPant

   // @ 1,2 get VAR xWS1 picture '!'  of bPant //??
   @ 1,    6 GET oGet VAR xWS1 OF bPant SIZE 30, 10 PICTURE "@K!" CENTER ;
      COLOR "W/G"                                                        ;
      VALID {||Solo(xWS1,{'A','I','P','S','W','U'})}

   oGet:cToolTip := "A=Concesionario iniciado por empleados"

   @ 1.8,  2 SAY "Number:" OF bPant

   //@ 2,2 get xWS2 picture '!' valid {||Solo(xWS2,{'C','K','M','N','U'})} of bPant
   @ 2,    6 GET oGet2 VAR xWS2 OF bPant SIZE 30, 10  PICTURE "@K!" CENTER ;
      VALID {||Solo(xWS2,{'C','K','M','N','U'})}

   oGet2:cToolTip := "C=Se canceló la cita"

   @ 2.6,  2 SAY "Date:" OF bPant

   @ 3,    6 GET oGet3 VAR dDat PICTURE "@E" OF bPant SIZE 60, 10 CENTER

   oGet3:cToolTip := OemToAnsi( "FiveWin ‚s mi Amorecito Lindo" )

   @ 3,    7 BUTTON oOk PROMPT "&Ok" OF bPant SIZE 30, 12         ;
      ACTION bPant:End()

   oOk:cToolTip := "Click OK"

   @ 3,   16 BUTTON oSalida PROMPT "&Cancel" SIZE 30, 12 OF bPant ;
      ACTION bPant:End() CANCEL

   // Moderno ToolTip:
   oSalida:cTooltip := { "Salida - Exit - Cancelar", ;
                         "Salida - Exit - Cancelar", 1, CLR_WHITE, CLR_HBLUE }
   
   ACTIVATE DIALOG bPant CENTERED

return nil

FUNCTION SOLO( xWS1 )

RETURN( .T. )
