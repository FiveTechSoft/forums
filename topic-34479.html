E naturalmente devi aggiungere il wrapper:

Code (fw): Select all Collapse
#pragma BEGINDUMP

#include "windows.h"
#include "math.h"
#include "hbapi.h"


HB_FUNC( SETTEXTCHARACTEREXTRA )
{
    hb_retnl( ( LONG ) SetTextCharacterExtra( ( HDC ) hb_parnl( 1 ), hb_parnl( 2 ) ) );
}

#pragma ENDDUMP
