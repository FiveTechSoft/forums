Please try this sample. It works fine here. Change it to reproduce the problem.


Code: Select all Collapse
</s>#include "Fivewin.ch"


FUNCTION MAIN()

    DBCREATE( "PRUEBA1.DBF", { { "TEST", "+", 4, 0 } } )
    DBCREATE( "PRUEBA2.DBF", { { "TEST", "+", 4, 0 } } )

    USE PRUEBA1

    APPEND BLANK
    APPEND BLANK

    CLOSE

    USE PRUEBA2

    APPEND FROM PRUEBA1

    CLOSE

    XBROWSER( "PRUEBA2.DBF" )
