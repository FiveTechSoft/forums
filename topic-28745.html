Code (fw): Select all Collapse
// Building an application just with a DialogBox using accelarator in dialog.

#include "FiveWin.ch"

static oDlg

function Main()

   DEFINE DIALOG oDlg FROM 6, 6 TO 20, 60 TITLE "Using Accelerator"

   @ 2, 2 BUTTON "&Acce" OF oDlg ;
      ACTION ( _SetMenu() )

   @ 4, 2 BUTTON "&Bye!" OF oDlg ACTION oDlg:End() CANCEL

   ACTIVATE DIALOG oDlg ;
      VALID MsgYesNo( "Salir?" )

return nil

//----------------------------------------------------------------------------//

function _SetMenu()

   local oMenu

   MENU oMenu

      MENUITEM "&One<F5>"  ACTION MsgInfo( "One" ) ;
         ACCELERATOR ACC_NORMAL,VK_F5

      MENUITEM "&Two<F6>"  ACTION MsgInfo( "Two" ) ;
         ACCELERATOR ACC_NORMAL,VK_F6

   ENDMENU

   oDlg:SetMenu( oMenu )

return nil
