yo lo utilizo sin problemas cuando indexo mis archivos,,,

oAnimate:CopyFiles()
_tipoDeAnimacion(oAnimate,4)
oAnimate:Play()

  • aqui mi proceso

oAnimate:stop()

function _tipoDeAnimacion( oAnimate, nAvi )

do case
case nAVI == 1
oAnimate:Search()

  case nAVI == 2
       oAnimate:FindFile()

  case nAVI == 3
       oAnimate:Locate()

  case nAVI == 4
       oAnimate:CopyFiles()

  case nAVI == 5
       oAnimate:CopyFile()

  case nAVI == 6
       oAnimate:DelFile()

  case nAVI == 7
       oAnimate:EmptyCan()

  case nAVI == 8
       oAnimate:DestFile()

endcase

return nil
