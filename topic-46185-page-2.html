function ChatGemini( cPrompt, cCode )

<i>    </i>local cKey := "Tu clave Gemini"
<i>    </i>local cUrl := "https://generativelanguage.googleapis.com/v1beta/models/gemini-3-pro-preview:generateContent"
<i>    </i>local oHttp
<i>    </i>local hJson := {=>}
<i>    </i>local hContent := {=>}
<i>    </i>local hPart := {=>}
<i>    </i>local cJson
<i>    </i>local cText := ""

<i>    </i>if Empty( cKey )
<i>    </i>    return "Error: GEMINI_API_KEY environment variable not set."
<i>    </i>endif

<i>    </i>try
<i>    </i>    oHttp := CreateObject( "MSXML2.ServerXMLHTTP.6.0" )
<i>    </i>catch
<i>    </i>    try
<i>    </i>        oHttp := CreateObject( "MSXML2.ServerXMLHTTP" )
<i>    </i>    catch
<i>    </i>        return "Error: Could not create MSXML2.ServerXMLHTTP object."
<i>    </i>    end
<i>    </i>end

<i>    </i>hPart[ "text" ] := cPrompt + IF( !Empty( cCode ), " Code: " + cCode, "" )
<i>    </i>hContent[ "parts" ] := { hPart }
<i>    </i>hJson[ "contents" ] := { hContent }

<i>    </i>cJson := hb_jsonEncode( hJson )

<i>    </i>try 
<i>    </i>    oHttp:Open( "POST", cUrl, .F. )
<i>    </i>    oHttp:SetRequestHeader( "Content-Type", "application/json" )
<i>    </i>    oHttp:SetRequestHeader( "X-goog-api-key", cKey )
<i>    </i>    oHttp:Send( cJson )
<i>    </i>catch
<i>    </i>    return "Error: Could not connect to Gemini API (Timeout or Network Error)."
<i>    </i>end

<i>    </i>if oHttp:Status == 200
<i>    </i>    hJson := hb_jsonDecode( oHttp:ResponseText )
<i>    </i>    if hb_HHasKey( hJson, "candidates" ) .and. Len( hJson[ "candidates" ] ) > 0
<i>    </i>        cText := hJson[ "candidates" ][ 1 ][ "content" ][ "parts" ][ 1 ][ "text" ]
<i>    </i>        // Simple formatting for chat
<i>    </i>        cText := StrTran( cText, Chr( 10 ), "<br>" )
<i>    </i>        cText := StrTran( cText, "```", "<pre>" ) // Very basic code block handling
<i>    </i>        cText := StrTran( cText, "**", "<b>" )
<i>    </i>        cText := StrTran( cText, "**", "</b>" ) // basic bold attempt, not perfect regex
<i>    </i>    else
<i>    </i>        cText := "Error parsing Gemini response: " + oHttp:ResponseText
<i>    </i>    endif
<i>    </i>else
<i>    </i>    cText := "API Error " + AllTrim( Str( oHttp:Status ) ) + ": " + oHttp:ResponseText
<i>    </i>endif

return cText
