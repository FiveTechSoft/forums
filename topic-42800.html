     Do case

          oPanel := TScrollPanel():New(oBar:nheight,1,oWnd:nBottom,oWnd:nWidth/2 +100,oWnd, .t.)

         oPanel := TScrollPanel():New(oBar:nheight,300,oWnd:nBottom,nRight,oWnd, .t.)
