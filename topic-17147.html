IF DTOC(mFechaIni) != " "
mCondi := mCondi + "cfac->fechafac >= mFechaIni .AND. "

cFiltro := "{||"+mCondi+"}"
gFiltro := '"'+mCondi+'"'

SELECT cfac
SET ORDER TO
DbSetFilter(&cFiltro,gFiltro)
