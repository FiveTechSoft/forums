Code (fw): Select all Collapse
METHOD RunAction( nKey, lControl ) CLASS TRibbonBar

   local nCurrent := 0
   local nTemp

   if nKey > 95 .and. nKey < 106
      nKey -= 48
   endif

   DEFAULT ::cBufferAcc := ""  // new !!!

   If nKey > 0
      ::cBufferAcc += Chr( nKey )
      if Len( ::cBufferAcc ) == 1
         if AScan( ::aAcc, {| uItem | Left( uItem[ 2 ], 1 ) == ::cBufferAcc } ) == 0
            ::cBufferAcc = ""
         endif
      endif
