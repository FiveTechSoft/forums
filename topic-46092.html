FUNCTION WdGetNRows( oDoc, nTable, nRow, nCol )
	LOCAL   Pos, CellHeight, CellHeight1, nRows, cBuf, nSizeRow
        

<i>    </i>    Pos := oDoc:Tables[nTable]:Cell( nRow, nCol ):End - 1
<i>    </i>    CellHeight := oDoc:Range( Pos, Pos ):Information( 8 )      // wdVerticalPositionRelativeToTextBoundary 8
<i>    </i>                    
<i>    </i>    cBuf := LEFT( oDoc:Tables[nTable]:Cell( nRow, nCol ):Text, LEN( oDoc:Tables[1]:Cell( nRow, nCol ):Text ) - 2 )

<i>    </i>    oDoc:Tables[nTable]:Cell( nRow, nCol ):Text := cBuf + CRLF
<i>    </i>                    
<i>    </i>    Pos := oDoc:Tables[nTable]:Cell( nRow, nCol ):End - 1
<i>    </i>    CellHeight1 := oDoc:Range( Pos, Pos ):Information( 8 )      // wdVerticalPositionRelativeToTextBoundary 8
<i>    </i>                    
<i>    </i>    nSizeRow := CellHeight1 - CellHeight

<i>    </i>    oDoc:Tables[nTable]:Cell( nRow, nCol ):Text := cBuf

<i>    </i>    IF CellHeight == 0
<i>    </i>            nRows := IIF( EMPTY( cBuf ), 0, 1 )
<i>    </i>    ELSE
<i>    </i>            nRows := INT( CellHeight / nSizeRow ) + 1
<i>    </i>    ENDIF
RETURN nRows<e>
