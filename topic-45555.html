   hb_jsonDecode(cJson, @hData)

   IF ValType(hData) == "A" .AND. Len(hData) > 0
      hData := hData[1]
   ENDIF

   // Estrai il nome della funzione
   cFuncName := hb_HGetDef(hData, "function", "Nessuna funzione")
