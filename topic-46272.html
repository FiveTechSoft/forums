HIX + FWH DLL (Self contained)
Posted: Sun Jan 18, 2026 05:38 AM

This is a guide to build a FWH DLL (with the Harbour's hvm inside the DLL)



go64.bat


Code: Select all Collapse
</s>@setlocal
call "%ProgramFiles%\Microsoft Visual Studio\2022\Community\VC\Auxiliary\Build\vcvarsall.bat" amd64
c:\harbour\bin\win\msvc64\hbmk2 fwh64.hbp -comp=msvc64
dumpbin /exports fwh64.dll
@endlocal<e>


fwh64.prg


Code: Select all Collapse
</s>#include "FiveWin.ch"  
  

function Main()

   MsgInfo( "dentro DLL" )

return nil 

#pragma BEGINDUMP

#include <windows.h>
#include <hbapi.h>
#include <hbvm.h>

__declspec(dllexport) void TEST( void )
{
   hb_vmInit( HB_TRUE );
   hb_vmQuit();   

}

#pragma ENDDUMP<e>


fwh64.hbp


Code: Select all Collapse
</s>-gui  

-hbdynvm  

-ofwh64  

-Ic:\fwh64\include  

fwh64.prg  

-Lc:\fwh64\lib  

-lFiveH64  

-lFiveHC64  

-lgdiplus  

-lole32  

-lOleDlg  

-lversion  

-luxtheme  

-cflag=-MD  

-ldflag+=/NODEFAULTLIB:LIBCMT  

xhb.hbc  

hbct.hbc  

hbwin.hbc  

hbmzip.hbc  

hbziparc.hbc  

hbfoxpro.hbc<e>


fwh.prg (run this from HIX. Remember to have hbdyn.ch at HIX folder)


Code: Select all Collapse
</s>#include "hbdyn.ch"

function Main()

   local hDLL := hb_libLoad( "fwh64.dll" )  
     

   hb_dynCall( { "TEST", hDLL, hb_bitOr( HB_DYN_CALLCONV_STDCALL, HB_DYN_CTYPE_VOID ) } )
   

   hb_libFree( hDLL )  

return nil <e>
