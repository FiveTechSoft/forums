Code (fw): Select all Collapse
STAT FUNC ConnectTo(hHost,hUser,hPassword,hBase,siMsg)

   FWCONNECT oCn HOST hHost USER hUser PASSWORD hPassword PORT 3306
   IF oCn:nError <> 0
       MiError(oCn,oCn:nError,lInternal,@CONECTA,siMsg)
   ENDIF
