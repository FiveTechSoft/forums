Function Set_items_Active(oItem,lActive,oBTn)

local oCtrl

for each oCtrl in oItem:aControls

oCtrl:bClrGrad = { | lInvert | If( ! lInvert,;

{ { 1, CLR_WHITE, CLR_WHITE } },;

{ { 1, RGB( 229,241,251 ), RGB( 229,241,251 ) } } ) }

oCtrl:refresh()

next

SetGradientBtnbmp(,oBtn)

return nil
