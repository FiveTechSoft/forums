Please notice that this is an important fix that we have recommended to apply to FWH 7.11:
Code: Select all Collapse
METHOD SetMsg( cText ) CLASS TMsgBar

   DEFAULT ::cMsg := ""
   
   if ::l2007
      ::cMsg = cText
      ::Refresh()
      return nil
   endif   
   
   ::GetDC()  // here!!!
