function ReArrangeCols( oBrw, aCols )
   AEval( oBrw:aCols, { |o| o:lHide := .f. } )
   oBrw:ReArrangeCols( aCols )
   AEval( oBrw:aCols, { |o| o:lHide := .t. }, Len( aCols ) + 1 )
   oBrw:Refresh()
return nil

else
  // oCol is the column with cHeader
  // do whatever you want with oCol
