FUNCTION validarRif( sRif )

   LOCAL DividendoEntero

   ENDIF

   sPrimerCaracter := Upper( Left( sRif, 1 ) )

   DO CASE
      CASE sPrimerCaracter == "V"
         iFactor := 1
      CASE sPrimerCaracter == "E"
         iFactor := 2
      CASE sPrimerCaracter == "J"
         iFactor := 3
      CASE sPrimerCaracter == "P"
         iFactor := 4
      CASE sPrimerCaracter == "G"
         iFactor := 5
   ENDCASE

   IF iFactor > 0
      suma := ( Val( SubStr( sRif, 9, 1 ) ) * 2 ) + ;
         ( Val( SubStr( sRif, 8, 1 ) ) * 3 ) + ;
         ( Val( SubStr( sRif, 7, 1 ) ) * 4 ) + ;
         ( Val( SubStr( sRif, 6, 1 ) ) * 5 ) + ;
         ( Val( SubStr( sRif, 5, 1 ) ) * 6 ) + ;
         ( Val( SubStr( sRif, 4, 1 ) ) * 7 ) + ;
