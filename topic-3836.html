function Main()

   local obmp, oBtn
   local fil, col


   if bandera_up = nil     
        fil :=141
        col := 10
        contador:=0
   endif

   DEFINE WINDOW oWnd TITLE "BOTON BITMAP"


   @ fil, col BTNBMP obtn filename curdir() + "\rcod8.bmp" adjust noborder size 83,34
