Function test()

   local  nHt       := nBottom * DLG_CHARPIX_H

   aData    := {}
   for n := 1 to 10
      AAdd( aData, { n, "Item-" + StrZero( n, 2 ), ;

         Space( 5 ), ;
         HB_RandomInt( 1, 9999 ) } )

      AUTOCOLS HEADERS { "SlNo", "Item" } ;
      COLSIZES 40 ;

       oBrw:CreateFromCode()
