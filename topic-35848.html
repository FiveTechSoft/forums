   DEFINE DIALOG oDlg SIZE 300, 300 PIXEL COLORS CLR_BLACK, CLR_WHITE ;

          TRANSPARENT UPDATE

   @ 10,75 GET aGet[7] VAR nAcconto SIZE 40,12 PIXEL OF oDlg ;
