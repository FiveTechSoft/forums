#Include "FiveWin.ch"

Function Main()

    Local oWord , cTemplate:="\YourFolderName\YourTemplateFileName.Doc"
    Local cSaveAsFile := "D:\Nota Nro 1.doc"

    oWord := CreateObject( "Word.Application" )

    oWord:Documents:Open(cTemplate)

   // Make the Necessary changes in the word file and then save this file with a different name

    oWord:ActiveDocument:SaveAs(cSaveAsFile)
    oWord:ActiveDocument:Close()
    oWord:Quit()
