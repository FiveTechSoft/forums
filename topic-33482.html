EndIf

eNDIF

  oCol:oBrw:GoRight()
  PostMessage( oCol, WM_CHAR, VK_RETURN)
EndIf

static function AddRow( oBrw )
   If nAccion == DALTAS
      RETURN NIL
   EndIf

   oBrw:GoBottom()

   PostMessage( oBrw:hWnd, WM_CHAR, VK_RETURN)
return nil

static function DelRow( oBrw, aDialogo )
   If oBrw:nLen > 0

      oBrw:Refresh()
   endif
   oBrw:SetFocus()
