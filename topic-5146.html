Try this sample:

Code: Select all Collapse
#include "Fivewin.ch"


FUNCTION MAIN()

    LOCAL oDlg

    LOCAL oGet, cVar := "This is a test" + CRLF + "The second line"

    DEFINE DIALOG oDlg

    @ 0, 0 GET oGet VAR cVar MEMO

    oDlg:bStart = { || oGet:SetPos( 0 ) }

    ACTIVATE DIALOG oDlg;
             ON INIT oDlg:SetControl( oGet );
             CENTER

    RETURN NIL
