static function CheckCustCdx()

   local n

   if File( "customer.cdx" )
         FErase( "customer.cdx" )
   endif

   if !File( "customer.cdx" )
      USE CUSTOMER EXCLUSIVE VIA "DBFCDX"

         CreateTag( FieldName( n ) )
      next
      USE
   endif
