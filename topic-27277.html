En source/api/regions.c:

Code (fw): Select all Collapse
HB_FUNC( CREATEROUNDRECTRGN )
{
   HRGN hRgn;

   if( hb_pcount() > 3 ) // C5 format
      hRgn = CreateRoundRectRgn( hb_parni( 1 ), hb_parni( 2 ), hb_parni( 3 ),
                                 hb_parnl( 4 ), hb_parni( 5 ), hb_parni( 6 ) );
   else
      hRgn = CreateRoundRectRgn( hb_parvni( 1, 2 ), hb_parvni( 1, 1 ), 
                                 hb_parvni( 1, 4 ) - hb_parvni( 1, 2 ), hb_parvni( 1, 3 ) - hb_parvni( 1, 1 ) + 1,
                                 hb_parni( 2 ), hb_parni( 3 ) );                                        

   RegisterResource( hRgn, "REGION" );
    
   #ifndef _WIN64
      hb_retnl( ( LONG ) hRgn );
   #else   
      hb_retnll( ( LONGLONG ) hRgn );
   #endif
}
