Hola Rafael

Code: Select all Collapse
function analiza( hDC, hBmp )

local hDCMem := CreateCompatibleDC( hDC )
local hOldBmp := SelectObject( hDCMem, hBmp ) 
local x := 1
local y := 1
local nColor

nColor := GetPixel( hDCMem, x, y ) //a tutipleni

SelectObject( hDCMem, hOldBmp )
DeleteDC( hDCMem )

return nil
