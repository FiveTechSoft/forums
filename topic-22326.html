Function VerificaCuil()
                        
Local i
Local nTotal
Local cCuil
Local naDigitos
Local nEnteroDivision
Local nTotalOperacion
Local cCadena         := ""
Local anMultiplicador := { 5, 4, 3, 2, 7, 6, 5, 4, 3, 2 }

DbSelectArea( "padron" )
padron -> ( DbGoTop() )
padron -> ( OrdSetFocus( "nombre" ) )
While( padron -> ( !Eof() ) )
   If( !Empty( padron -> cuil ) )
      cCuil := StrTran( padron -> cuil, "-", "" )
      naDigitos := {}
      For i := 1 To 10
         Aadd( naDigitos, Val( SubStr( cCuil, i, 1 ) ) * anMultiplicador[ i ] )
      Next i
      nTotal := 0
      For i := 1 To 10
         nTotal += naDigitos[ i ]
      Next i
      nEnteroDivision := Int( nTotal / 11 )
      nTotalOperacion := nTotal - ( nEnteroDivision * 11 )
      If( nTotalOperacion == 0 )
         If( SubStr( cCuil, 11, 1 ) == "0" )
         Else
            cCadena += cCuil + Space( 3 ) + padron -> nombre + CRLF
         End
      Else
         If( SubStr( cCuil, 11, 1 ) == Alltrim( Str( 11 - nTotalOperacion ) ) )
         Else
            cCadena += cCuil + Space( 3 ) + padron -> nombre + CRLF
         End
      End
   End
   padron -> ( DbSkip() )
Enddo
If( !Empty( cCadena ) )
   MsgStop( cCadena, "No verifica su C.U.I.L" )
End

Return 0
