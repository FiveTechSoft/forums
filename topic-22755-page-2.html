Function fSendMail( cTo, cBCC, cSubject, cBody, cAttach )

   Local oCfg, oMsg, oError, nEle, cToken, ;
         aAttach  := {}, ;
         lAuth    := IIF(! Empty( cUser ) .and. ! Empty( cPass ), .T., .F.) ,;
         nSendOpt := 2   // send using: 1 = pickup folder  2 = port
   *
   Default cSubject := &quot;Text de correo con CDO&quot;, ;
           cBody    := &quot;Test de prueba de correo con CDO&quot;
   *
   IF ! EMPTY( cTo )

      If &quot;GMAIL.COM&quot; $ Upper( cMailServer ) .and. ( Empty( cUser ) .or. Empty( cPass ) )
         MsgStop( &quot;Con GMail son requeridos nombre de usuario y contraseña&quot;, &quot;ATENCION&quot; )
         Return Nil
      EndIf

      CursorWait()

      nEle := 1

      While ! Empty( cToken := StrToken( cAttach, nEle++, &quot;,&quot; ) )
         AAdd( aAttach, cToken )
      EndDo

      Try
         oCfg := CreateObject( &quot;CDO.Configuration&quot; )

         With Object oCfg:Fields

            :Item( &quot;http://schemas.microsoft.com/cdo/configuration/smtpserver&quot; ):Value :=  allTrim( cMailServer )
            :Item( &quot;http://schemas.microsoft.com/cdo/configuration/smtpserverport&quot; ):Value :=  nPort
            :Item( &quot;http://schemas.microsoft.com/cdo/configuration/sendusing&quot; ):Value := nSendOpt   // Remote SMTP = 2, local = 1
            If lAuth
               :Item( &quot;http://schemas.microsoft.com/cdo/configuration/smtpauthenticate&quot; ):Value :=  .T.
               :Item( &quot;http://schemas.microsoft.com/cdo/configuration/smtpusessl&quot; ):Value :=  .T.
               :Item( &quot;http://schemas.microsoft.com/cdo/configuration/sendusername&quot; ):Value := allTrim( cUser )
               :Item( &quot;http://schemas.microsoft.com/cdo/configuration/sendpassword&quot; ):Value := allTrim( cPass )
               :Item( &quot;http://schemas.microsoft.com/cdo/configuration/smtpconnectiontimeout&quot;):Value := 30
            EndIf

            :Update()

         End With

         oMsg := CreateObject( &quot;CDO.Message&quot; )

         With Object oMsg
            :Configuration := oCfg
            :From          := allTrim( cFrom )
            :To            := allTrim( cTo )
            :Subject       := allTrim( cSubject )
            :TextBody      := allTrim( cBody )

            For nEle := 1 To Len( aAttach )
               :AddAttachment(AllTrim( aAttach[ nEle ] ) )
            Next

            If ! Empty( cBCC )
               :BCC := Trim( cBCC )
            EndIf

            :Send()
         End With

      Catch oError
         CursorArrow()
         MsgStop( &quot;Could not send the message&quot; + CRLF  + &quot;Error: &quot; + cValToChar( oError:GenCode) + CRLF + ;
                  &quot;SubC: &quot; + cValToChar( oError:SubCode ) + CRLF + &quot;OSCode: &quot; + cValToChar( oError:OsCode ) + CRLF + ;
                  &quot;SubSystem: &quot; + cValToChar( oError:SubSystem ) + CRLF + &quot;Message: &quot; + oError:Description )

         oCfg := Nil
         oMsg := Nil
         Return Nil
      End Try

      oCfg := Nil
      oMsg := Nil
      SndPlaySound( GetWinDir() + &quot;\media\Tada.wav&quot;, 0 )
      CursorArrow()

   ELSE
      MsgStop(&quot;No hay Destinatario&quot;+CRLF+;
               &quot;Envio CANCELADO&quot;, &quot;Upsss...!&quot; )
   ENDIF
   *

Return Nil

Static Function fAddAttach( oGet )

   Local cFile, ;
         cAttach := oGet:VarGet()

   cFile := cGetFile( &quot;*.*&quot;, &quot;Selecciona el archivo&quot; )

   If ! Empty( cFile )
      cAttach := Lower( PadR( AllTrim( cAttach ) + If( ! Empty( cAttach ), &quot;,&quot;, &quot;&quot; ) + AllTrim( cFile ), 180 ) )
   EndIf

   oGet:cText( cAttach )

Return Nil

Static FUNCTION WriteIni()
    local oIni
    INI oIni FILENAME ( &quot;./emailCdo.ini&quot; )
        SET SECTION &quot;SETUPMAIL&quot; ENTRY  'MailServer' TO cMailServer OF oIni
        SET SECTION &quot;SETUPMAIL&quot; ENTRY  'remitente'  TO cFrom       OF oIni
        SET SECTION &quot;SETUPMAIL&quot; ENTRY  'puerto'     TO nPort       OF oIni
        SET SECTION &quot;SETUPMAIL&quot; ENTRY  'usuario'    TO cUser       OF oIni
        SET SECTION &quot;SETUPMAIL&quot; ENTRY  'password'   TO cpass       OF oIni
    ENDINI

Return nil
