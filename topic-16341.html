Procedure ChangeSelect( aObj )

   local n
     
   aObj[ 1 ]:lSelected := .t.
   for n = 2 to len( aObj )
      aObj[ n ]:lSelected := .f.
      aObj[ n ]:Refresh()
   next
return

Function ChangeBmp( oBmp )
                  
   static lWork
   
   if lWork == nil
      lWork := .f.
   endif

   if !lWork
      lWork := .t.
      nIndex+=nAvance 
      
      if nIndex > LEN( aBmp )
         nAvance := -1
         nIndex := LEN( aBmp ) + nAvance
      elseif nIndex == 0
         nAvance := 1
         nIndex := 1 + nAvance
      endif
      oBmp:hBitmap = aBmp[ nIndex ] 
      oBmp:Refresh()
      lWork := .f.
   
   endif

   
return nil

function AddResource( nHResource, cType )

   AAdd( aResources, { cType, nHResource, ProcName( 3 ), ProcLine( 3 ) } )

//----------------------------------------------------------------------------//

function DelResource( nHResource )

   local nAt

   if ( nAt := AScan( aResources, { | aRes | aRes[ 2 ] == nHResource } ) ) != 0
      ADel( aResources, nAt )
      ASize( aResources, Len( aResources ) - 1 )
   endif    

return nil

METHOD MouseMove( nRow, nCol, nKeyFlags ) CLASS TRBtn

   local lAction := ! Empty( ::bAction )
   
   if ::nTypeButton == TYPE_SAYBUTTON
      return 0
   endif

   if ! ::lMOver
      ::lMOver = .T.
   endif
   
   do case
      case ::nTypeButton == TYPE_POPUP
         ::lSubOver := nCol >= ::nWidth - 13 .and. lAction
         
      case ::nTypeButton == TYPE_SPLITPOPUP
         ::lSubOver := nRow >= ::nHeight/5*3 .and. lAction
   endcase
   
   /*
   if ::ClassName == "TRBGROUP"
      ::MouseMove( nRow, nCol, nKeyFlags )
   endif
   */
   
   if ::oWnd:ClassName == "TRBGROUP"
      if !::oWnd:lMOver
         ::oWnd:lMOver:=.t.
         ::oWnd:Refresh()
      endif
   endif
   
   
   ::Refresh()

   ::oWnd:SetMsg( ::cMsg )
         
   TrackMouseEvent( ::hWnd, TME_LEAVE )
   
return Super:MouseMove( nRow, nCol, nKeyFlags )

METHOD MouseMove( nRow, nCol, nKeyFlags ) CLASS TRBtn

   local lAction := ! Empty( ::bAction )
   
   if ::nTypeButton == TYPE_SAYBUTTON
      return 0
   endif

   if ! ::lMOver
      ::lMOver = .T.
   endif
   
   do case
      case ::nTypeButton == TYPE_POPUP
         ::lSubOver := nCol >= ::nWidth - 13 .and. lAction
         
      case ::nTypeButton == TYPE_SPLITPOPUP
         ::lSubOver := nRow >= ::nHeight/5*3 .and. lAction
   endcase
   
   if ::oWnd:ClassName == "TRBGROUP"
      if !::oWnd:lMOver
         ::oWnd:lMOver:=.t.
         ::oWnd:MouseMove( nRow, nCol, nKeyFlags )
      endif
   else
      ::Refresh()
   endif

   ::oWnd:SetMsg( ::cMsg )
         
   TrackMouseEvent( ::hWnd, TME_LEAVE )
   
return Super:MouseMove( nRow, nCol, nKeyFlags )
