Here it works ok with this little change:
Code: Select all Collapse
    ACTIVATE DIALOG oDlg; 
             ON INIT ( oDlg:SetBrush( oBrush ),;
                       oFld:aDialogs[ 1 ]:bEraseBkgnd := nil,; 
                       oFld:aDialogs[ 1 ]:SetBrush( oBrush ) ); 
             CENTER
