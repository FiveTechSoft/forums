Consider this code ...

Code (fw): Select all Collapse
DEFINE BUTTONBAR oBar OF oWind SIZE 80,58 2010    
       oBar:SetColor(0)
                             
       oBar:bClrGrad = { || { { 0.10,15724527,7303023 }, ;
                       { 0.10,7303023,15724527 } } }

      DEFINE BUTTON oBtn1 of oBar ACTION ( _VehBrow( oWind )) ;
              RESOURCE "Car",;
                       "Car",;
                       "Car";
              PROMPT "Veh\Eq\Boat" ;
              TOOLTIP "Vehicles and Equipment Information"
       oBtn1:lTransparent = .T.      // <-----  add this line


Try adding oBtn1:lTransparent = .t. after your Button Command ..
