correct because BCC7 does not support __asm ​​{ ... } with local variables



Code: Select all Collapse
</s>#pragma BEGINDUMP

#include <hbapi.h>
#include <windows.h>

HB_FUNC( GETCPUID )
{
   char szID[17];
   unsigned long ebx, edx;

   // Usa __asm con sintassi BCC7
   _asm {
      mov eax, 1
      cpuid
      mov ebx, ebx
      mov edx, edx
   }

   // Usa variabili locali per ebx, edx
   _asm mov ebx, ebx
   _asm mov edx, edx

   // Formatta come stringa
   wsprintfA( szID, "%08lX%08lX", ebx, edx );
   hb_retc( szID );
}

#pragma ENDDUMP<e>

FUNCTION GetCpuIDHarbour()

<i>    </i>LOCAL nHandle
<i>    </i>LOCAL cCommand := "wmic cpu get ProcessorId"
<i>    </i>LOCAL aLines
<i>    </i>LOCAL cID
<i>    </i>local cLine

<i>    </i>local nRet
<i>    </i>local cStdOut := space(40)
<i>    </i>local cStdErr := space(40)

<i>    </i>nRet = hb_processRun( cCommand, , @cStdOut, @cStdErr )
<i>    </i>IF Empty( cStdOut )
<i>    </i>    RETURN ""
<i>    </i>ENDIF

<i>    </i>// Procesar la salida. wmic suele devolver una cabecera y el valor.
<i>    </i>// Ejemplo de salida:
<i>    </i>// ProcessorId
<i>    </i>// BFEBFBFF00040671

<i>    </i>// Dividir el resultado por líneas
<i>    </i>aLines := hb_aTokens( cStdOut, Chr(13) + Chr(10) )

<i>    </i>// La segunda línea (índice 2, si la cabecera es la 1ra y no hay líneas vacías)
<i>    </i>// debería contener el ID. Necesitarás limpiar espacios y saltos de línea.
<i>    </i>IF Len( aLines ) >= 2
<i>    </i>    // El ID suele estar en la segunda línea después de limpiar espacios.
<i>    </i>    cID := AllTrim( aLines[ 2 ] )

<i>    </i>    // Buscar la línea que no es 'ProcessorId' y no está vacía
<i>    </i>    FOR EACH cLine IN aLines
<i>    </i>        cLine := AllTrim( cLine )
<i>    </i>        IF ! Empty( cLine ) .AND. Upper( cLine ) <> "PROCESSORID"
<i>    </i>            RETURN cLine // Este es el ProcessorId
<i>    </i>        ENDIF
<i>    </i>    NEXT
<i>    </i>ENDIF

<i>    </i>RETURN ""<e>
