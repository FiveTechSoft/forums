En este ejemplo:
Code (fw): Select all Collapse
  oAdoConexion := Createobject("adodb.connection")
  oAdoConexion:Mode := adModeRead                      // Conectamos en Solo Lectura
  oAdoConexion:Provider = "Advantage.OLEDB.1"
  oAdoConexion:Properties:Item("User ID", 'ADSSYS' )
  oAdoConexion:Properties:Item("PassWord", 'ddd')
  oAdoConexion:Properties:Item("Data Source", "D:\Empr_00\Empr_00.add")
  oAdoConexion:Properties:Item("Persist Security Info", "False")
  oAdoConexion:Properties:Item("Advantage Server Type", "ADS_REMOTE_SERVER")
  oAdoConexion:Open()

  oAdoComand := Createobject("adodb.command")
  oAdoComand:ActiveConnection( oAdoConexion )
  oAdoComand:CommandTimeout := 0

  oRcs := CreateObject("ADODB.recordset")
  oRcs:Open("Select FchCrea From Pedido", oAdoConexion)
  
  ? oRcs:fields('FchCrea'):Value                // 20/11/10 00:00:00.000
  ? Valtype(oRcs:fields('FchCrea'):Value)   // T
  ? oRcs:fields('FchCrea'):Type                 // 133 -> adDBDate
