#include "Fivewin.ch"


FUNCTION MAIN()

    LOCAL oDlg

    LOCAL oCbx, oGet

    LOCAL cVar1 := ""
    LOCAL cVar2 := SPACE( 30 )

    DEFINE DIALOG oDlg

    @ 1, 1 COMBOBOX oCbx VAR cVar1;
           ITEMS { "First", "Second", "Third" };
           ON CHANGE ( oGet:VarPut( PADR( cVar1, LEN( cVar2 ) ) ),;
                       oGet:Refresh() )

    @ 3, 1 GET oGet VAR cVar2

    ACTIVATE DIALOG oDlg;
             CENTER

    RETURN NIL
