lUno := .F.

IF LEN(aPrompts) == 1
AADD( aPrompts, "" )
lUno := .T.
ENDIF

DEFINE DIALOG oDls .....bla bla bla

oFolder := TFolderEx():Redefine(100,oDls,aDialogs,,aPrompts,40,,5)

 IF lUno
    oFolder:aEnable  := { .T., .F. }
    oFolder:aVisible  := { .T., .F. }
 ENDIF
