Javier,

Revisa este ejemplo:
Code: Select all Collapse
#include "fivewin.ch"
#include "xbrowse.ch"


function main()

   local oWnd, oBrw
   local aData := { ;
      { "NY", "       ", 300, 3000 }, ;
      { "NY", "ONE    ", 100, 1000 }, ;
      { "NY", "TWO    ", 200, 2000 }, ;
      { "TX", "       ",1100,11000 }, ;
      { "TX", "THREE  ", 300, 3000 }, ;
      { "TX", "FOUR   ", 350, 3500 }, ;
      { "TX", "FIVE   ", 450, 4500 }  }
// Should start with data grouped and softed like above
  

   DEFINE WINDOW oWnd

   @ 0,0 XBROWSE oBrw OF oWnd ;
      ARRAY aData AUTOCOLS

   oBrw:CreateFromCode()
   oBrw:SetTree( nil, { "OPEN", "CLOSE", "GREEN" } ) // build tree structure from the array automatically

   oWnd:oClient := oBrw

   ACTIVATE WINDOW oWnd


return nil

CLOSE BITMAP "..\bitmaps\16x16\folder.bmp"
