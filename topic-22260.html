STATIC FUNCTION SKIPPER( oRsx, nSkip )
 LOCAL nRec := oRsx:AbsolutePosition
 oRsx:Move( nSkip )
 IF oRsx:EOF; oRsx:MoveLast(); ENDIF
 IF oRsx:BOF; oRsx:MoveFirst(); ENDIF
RETURN( oRsx:AbsolutePosition - nRec )

FUNCTION FILTRA_MAR(cBuscar,oRS,oLBx)
 IF LEN(ALLTRIM(cBuscar))>0
     oRs:Filter:= "descripcion LIKE" + "  '*" + ALLTRIM(cBuscar) + "*' "
 ELSE
     oRs:Filter:= ""
 ENDIF
 oRs:Refresh()
 oLBx:Refresh()
RETURN NIL

///////////////  VALIDAMOS SI EXISTE EL CODIGO INGRESADO //////////////////
FUNCTION EXISTE_CODIGO(cCodigo,oRS)
 LOCAL nBookMark,oRsExis,lBusca:=.F.
 IF EMPTY(cCodigo)
        RETURN .F.
 ENDIF
 nBookMark:=oRs:BookMark // guardamos el puntero actual del recordset para reestaablecerlo cuando el registro no existe
 oRS:MoveFirst()
 oRS:Find:="codigo LIKE" + " '%" + ALLTRIM(cCodigo) + "%' "
 IF !oRS:EOF()
    MSGINFO("Código ya registrado")
    lBusca:=.T.
 ENDIF
 oRs:BookMark:=nBookMark   // aqui se reestablece el puntero del recordset
RETURN lBusca

FUNCTION ELIM_MAR(oRS,oLbx)
 IF MSGYESNO("esta seguro de eliminar este registro?","Elimina")
        oRS:Delete()
        IF !oRs:EOF()
           oRs:MoveNext()
        ELSE
             oRs:MovePrevious()
        ENDIF
        oRS:Update()
        oLbx:Refresh()
 ENDIF
RETURN NIL
