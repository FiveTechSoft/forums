#pragma BEGINDUMP

#include <windows.h>
#include <hbapi.h>

HB_FUNC( SHELLEXECUTE )
{
   hb_retnl( ( LONG ) ShellExecute( ( HWND ) hb_parnl( 1 ), hb_parc( 2 ),
                                    hb_parc( 3 ), hb_parc( 4 ), hb_parc( 5 ),
                                    hb_parnl( 6 ) ) );
}

#pragma ENDDUMP
