function LoadFreeImage( cResFile )

   DEFAULT cResFile := If( IsWin64(), "freeimage64.dll", "freeimage.dll" )

   if Empty( hLib )
      #ifdef __CLIPPER__
         hLib = LoadLib32( cResFile )
      #else
         hLib = LoadLibrary( cResFile )
      #endif

      if hLib <= 32
         MsgAlert( "Cannot load " + If( IsWin64(), "FreeImage64.dll", "FreeImage
         hLib  = 0
      endif

   endif

return hLib
