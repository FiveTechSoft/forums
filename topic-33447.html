//-----------------------
Static FUNCTION _ExitPgm( lCLEAN , oRsDrv, oWndchild,lOk )

LOCAL lOK1

lOK1  := .F.


IF lCLEAN = .T.
   lOK  := .T.
   lOK1 := lOK

   try
      oRsDrv:CLose()
   catch
   end try

   oWndChild:End()


ENDIF

RETURN( lOK1 )
