We have again tested many bitmaps with all compilers xHarbour,Harbour 32 bits, msvc 32 and 64 bits, borland 64 bits.

This is the sample test:
Code (fw): Select all Collapse
#include "fivewin.ch"

function main()

   local oDlg, aBtn[ 6 ], n, nCol
   local aBmp := { ;
      "..\bitmaps\32x32\attach.bmp", ;
      "..\bitmaps\32x32\browse.bmp", ;
      "..\bitmaps\alphabmp\ichat.bmp", ;
      "..\bitmaps\pngs\image5.png", ;
      "..\icons\hires\avalon 1.ico", ;
      "..\icons\hires\calc.ico"  }

   DEFINE DIALOG oDlg SIZE 600,250 PIXEL TRUEPIXEL ;
      TITLE FWVERSION

   nCol  := 4
   for n := 1 to 6
      @ 150,nCol BTNBMP aBtn[ n ] SIZE 96,96 PIXEL OF oDlg ;
         FILE aBmp[ n ] TOP ;
         PROMPT cFileNoPath( aBmp[ n ] ) 2007
      nCol  += 98
   next

   ACTIVATE DIALOG oDlg CENTERED

return nil
