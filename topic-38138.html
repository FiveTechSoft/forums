function MyRun( cComando, nStyle, lWait, lShowResult )

local oShell, nRetorno

IF valtype( nStyle ) != &quot;N&quot;
   nStyle := 0
ENDIF

IF ValType( lWait ) != &quot;L&quot;
   lWait := .T.
ENDIF

IF ValType( lShowResult ) != &quot;L&quot;
   lShowResult := .F.
ENDIF

oShell := CreateObject( &quot;WScript.Shell&quot; )

IF !GetEnv( &quot;OS&quot; ) == &quot;Windows_NT&quot;
   cComando += &quot;start &quot; + cComando
ENDIF

TRY
  nRetorno := oShell:Run( &quot;%comspec% /c &quot; + cComando, nStyle, lWait )
CATCH
  MsgStop(&quot;ERRO NO ENVIO DO COMANDO!&quot;, &quot;MYRUN&quot;)
END

IF lShowResult .and. nRetorno &gt; 0 .and. nRetorno &lt;= 32
   MsgInfo( &quot;Erro Win_Run(): &quot; + ltrim( Str( nRetorno ) ), &quot;Win_Run - Error&quot; )
ENDIF

oShell := NIL

Return nRetorno = 0
