Function xconecta(Xruta, Xusuario, Xcontra, Xpuerto)
local Oserver
local oBrw, Obrw2
local oDlg2, oBtn
local aBases:={}, aTablas:={}
local i
   oServer:= TMYSQLSERVER():new( alltrim(xruta), alltrim(xusuario), alltrim(xcontra), xpuerto)
   IF oserver:lError
      MSGALERT( "Hubo error al conectarse, verifique si hay conexion al servidor o usuario/clave incorecta we...")
      Return Nil
   Endif

  
   
      
   abases:=Oserver:ListDBs()      // Obtiene la lista de las bases de datos we
   if len(abases)>=1
      aTablas:=Metetablas(Oserver, aBases[1] )   
   Endif
   DEFINE DIALOG oDlg2 FROM 1, 1 TO 35, 120 TITLE "Bases de datos Disponibles "+str(len(abases),4) 
 
   @ 1.5, 1 XBROWSE oBrw OF oDlg2 ARRAY aBases size 130,210 update;
          COLUMNS {1};
          HEADERS {"Base De Datos "};
          COLSIZES 200 LINES CELL;
          ON CHANGE (atablas:=Metetablas(Oserver, aBases[ oBrw:nArrayAt] ),oBrw2:aArrayData := aTablas, obrw2:refresh() )
           
   oBrw:CreateFromCode()
   odlg2:oClient := oBrw           
   
   @ 1.5, 20 XBROWSE oBrw2 OF oDlg2 ARRAY aTablas size 150,210 update;
          COLUMNS {1};
          HEADERS {"Tablas de La Base  Datos"};
          COLSIZES 250 LINES CELL;
          ON LEFT DBLCLICK VisorBase(Oserver, aTablas[ oBrw2:nArrayAt], aBases[ oBrw:nArrayAt]  ) 
           
   oBrw2:CreateFromCode()
   odlg2:oClient := oBrw2           
   

    @ 120, 360 BTNBMP OBTN PROMPT "Tras  Dbf/Mysql"      2007  OF oDlg2 size 60,12 action (Diatras(Oserver, Abases[ oBrw:nArrayAt] ),;
                                  atablas:=Metetablas(Oserver, aBases[ oBrw:nArrayAt] ),oBrw2:aArrayData := aTablas, obrw2:refresh())  
                       
      
    @ 140, 360 BTNBMP OBTN PROMPT "a Excel"      2007  OF oDlg2 size 60,12 ACTION;
                       MsgMeter( { |oMeter, oText, oDlg2, lEnd | ;
                       Expexcel( oBrw2, oMeter, oText, oDlg2, @lEnd ) } )
  
   
    @ 160, 360 BTNBMP OBTN PROMPT "Salir  "      2007  OF oDlg2 size 60,12 ACTION  odlg2:end() 
   
   
   
   
   ACTIVATE DIALOG ODLG2 CENTER 
   
   
   
   
Return Nil

         ENDIF

         IF Xoregi:fieldtype(i)="C"
            puro:=quicara(puro)  //quita los posibles " o ' que hayga de valores en los campo porque si no no se da de alta ok 
         Endif
         
         IF Xoregi:fieldtype(i)="M"    // sepse
            puro:=quicara(puro)  //quita los posibles " o ' que hayga de valores en los campo porque si no no se da de alta ok 
         Endif
         
         
         
         IF Xoregi:fieldtype(i)="L"
            //QUIRE DECIR QUE EL CAMPOS ES LOGICO WE
            PURO:=CAMTIPOL(PURO) // convierte el valor logico en f o t para el campo en sql 
         ENDIF
         
         IF Xoregi:fieldtype(i)="D"
            //QUIRE DECIR QUE EL CAMPOS ES FECHA WE
            PURO:=str(year(puro),4)+"-"+strzero(month(puro),2)+"-"+strzero(day(puro),2)    //  "2008-12-12" // convierte el valor logico en f o t para el campo en sql 
         ENDIF

            //QUIRE DECIR QUE ES CAMPO NUMERICO AY QUE CHECAR LOS ENTEROS Y LOS DECIMALES
            PURO:=ALLTRIM(STR(PURO)) // convierte el valor logico en f o t para el campo en sql 
         ENDIF
         CDATO:=CDATO+"'"+puro+"'"+METE
       
        
      
      Next i   /// pasa los datos del objecto de la clase database
      cvalor:=cvalor+cdato+")"
      cinsert:=cinsert+cvalor
     
    TRY

 Function Creatabla(osql, cdata, cuantos, oregi,obase, kprimary, ksegunda)
LOCAL CALIAS, cEstructura, entro:=.f., serbase
local cauxi, i, ctexto1, oquery, n

   
   cEstructura := "CREATE TABLE IF NOT EXISTS "+alltrim(alias())+"  ( "
   
   CTEXTO1:="ele integer default 0,"        // es un campo clave para agregarselo y poder de dar de alta todos los registros 
   cEstructura:=cestructura + ctexto1
   for i:=1 to cuantos  // son los campos que tiene la base de datos calias   la que se selecciono
      entro:=.f.
      
      
      
      if oregi:fieldtype(i)="C"
         
         CTEXTO1:="varchar("+alltrim(str(obase[i][3]))+") default '.',"
         entro:=.t.
      endif
     
      if oregi:fieldtype(i)="D"
         CTEXTO1:="date default '2008-12-12',"
         entro:=.t.
      endif
     
      if oregi:fieldtype(i)="N"
         entro:=.t.
         IF OBASE[I][4]>0   //QUIERE DECIR QUE TIENE DECIMALES WE
            CTEXTO1:="decimal("+alltrim(str(obase[i][3]))+","+alltrim(str(obase[i][4]))+") default 0, "
         ELSE
            CTEXTO1:="integer("+alltrim(str(obase[i][3]))+") default 0,"
         endif
      endif
      
      if oregi:fieldtype(i)="L"
         CTEXTO1:="varchar(1) default ' ',"
         entro:=.t.
      endif
      
      if oregi:fieldtype(i)="M"   // que longitud le puedo poner a esto ( porque con text marca error)  ay levariamos a nuestro gusto o ay un tipo de datos especial 
         CTEXTO1:="varchar(500) default ' ',"
         entro:=.t.
      endif
     
      
     
      if entro   
         ctexto1:=oregi:fieldname(i)+" "+ctexto1
         cEstructura:=cestructura + ctexto1
      endif
     
   next i   // con esto son todos los campos de la base de datos   
   //cEstructura += " ) "
   cEstructura += "PRIMARY KEY (ele) ) "
  // cEstructura += "KEY prodi ( nombre ) )"       //ay que checar con willian que es prodi   y si no hay key prodi como terminamos el primary key we
   cEstructura += " ENGINE=InnoDB DEFAULT CHARSET=latin1"
 
   // Ejecuta query para creación de la tabla
   osql:Query( cEstructura )
   IF osql:lError
       MSGALERT( "Hubo error al seleccionar la base de datos, verifique...")
      osql:end()
      
      // si no hay error aqui podemos llamar el traspaso de la informacion o se hace depues lo weno que ya quedo generado la estructura al sql
      return .f.
   else

   
   ENDIF

   
Return .t.   // dar de alta efectuado verdadero we
