FUNCTION MeuWinExec( cParametro )

    LOCAL cExecute := GetPvProfString( "" )

    // NT, 2000 e XP
    IF  IsWinNT() .OR. IsWin2000()
        cExecute := GetEnv( "COMSPEC" ) + " /C "
    ENDIF

RETURN WinExec( cExecute + cParametro, 0 )
