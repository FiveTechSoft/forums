KeyChar( nKey, nFlags ) CLASS TGet

LOCAL .........
LOCAL .........

if !Empty( ::cPicture ) .and. '@!' $ ::cPicture
nKey := Asc( UpperCase( Chr( nKey ) ) )
endif

...............
...............
Resto de código del método.
