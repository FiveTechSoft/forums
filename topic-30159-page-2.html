Code (fw): Select all Collapse
Function HCadFunction( nOp )
     Local nSymbols
     Local nFunctions
     Local n
     Local aFunc   := {}
     Local oWnd
     Local oBrw
     Local cCad    := ""
     DEFAULT nOp   := 0
     nSymbols      := __dynsCount()
     nFunctions    := 0

     for n := 1 to nSymbols
        if __dynsIsFun( n )
           AAdd( aFunc, __dynsGetName( n ) + "()" )
           cCad := __dynsGetName( n ) + " "
           nFunctions++
        else

        endif
     next

return IF( Empty( nOp ), cCad, aFunc )
