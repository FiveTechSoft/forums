Function FastFilter( cCad, nOrder )
      Local nIndice := OrdNumber()
      Set order to nOrder
      index on &(upper(IndexKey())) tag temp TO tHarbourt CUSTOM ADDITIVE MEMORY      
      Set order to nOrder
      dbGotop()
      Do While OrdWildSeek( cCad, .T. )
         OrdKeyAdd( "temp" )
      Enddo 
      OrdSetFocus( "temp" )
      Browse()
Return nIndice
