FUNCTION get_list_of_real_codepages()
   LOCAL s_uni := { => }
   LOCAL cdp
   s_uni := { => }
   FOR EACH cdp IN hb_cdpList()
      s_uni[ hb_cdpUniID( cdp ) ] := cdp
   NEXT
RETURN s_uni
