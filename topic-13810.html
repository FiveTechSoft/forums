	oFld:nHeight := oDlg:nHeight-80

  oFld:aDialogs[i]:SetColor(, oDlg:nClrPane)
  Next
  oFld:Refresh()
  
  oGrp:nHeight := oFld:nHeight-35
  oGrp:nWidth  := oFld:nWidth-17
  
  obtnCancel:nTop := oDlg:nHeight-70
	obtnCancel:nLeft := oDlg:nWidth-125

return
