HB_PING.PRG


Code: Select all Collapse
</s><i>
</i>#pragma BEGINDUMP
#include <hbapi.h> 
#include <winsock2.h>
#include <iphlpapi.h>
#include <icmpapi.h>

int hb_Ping( const char * cp )
{
    HANDLE hIcmpFile;
    unsigned long ipaddr;
    DWORD dwRetVal;
    char SendData[32] = "Data Buffer";
    LPVOID ReplyBuffer;
    DWORD ReplySize;

<i>    </i>ipaddr = inet_addr( cp );
<i>    </i>if (ipaddr == INADDR_NONE)
<i>    </i>    return 1;
<i>    </i>
<i>    </i>hIcmpFile = IcmpCreateFile();
<i>    </i>if (hIcmpFile == INVALID_HANDLE_VALUE)
<i>    </i>    return 2;

<i>    </i>ReplySize = sizeof(ICMP_ECHO_REPLY) + sizeof(SendData);
<i>    </i>ReplyBuffer = (VOID*) malloc(ReplySize);
<i>    </i>if (ReplyBuffer == NULL)
<i>    </i>    return 3;
<i>    </i>
<i>    </i>dwRetVal = IcmpSendEcho(hIcmpFile, ipaddr, SendData, sizeof(SendData), 
<i>    </i>    NULL, ReplyBuffer, ReplySize, 1000);

<i>    </i>if (dwRetVal == 0)
<i>    </i>    return 4;
<i>    </i>
<i>    </i>return 0;
}

HB_FUNC( HB_PING )
{
   hb_retni( hb_Ping( hb_parc( 1 ) ) );
}

* New: function HB_PING( cIP ) --> nResult. Ejemplo de uso:



Code: Select all Collapse
</s>FUNCTION Main()
    LOCAL nResultado
    LOCAL cIP := "8.8.8.8" // Servidor DNS de Google

<i>    </i>// Llamada correcta
<i>    </i>nResultado := HB_PING( cIP )

<i>    </i>IF nResultado == 0
<i>    </i>   ? "Ping exitoso a " + cIP
<i>    </i>ELSE
<i>    </i>   ? "Error al hacer ping. Código:", nResultado
<i>    </i>ENDIF
  RETURN NIL<e>


Valor Significado Causa en el código C

0 Exito El ping respondió correctamente (dwRetVal != 0).

1 Error de IP El formato de la IP es inválido o se pasó un nombre de dominio (inet_addr falló).

2 Error de Handle No se pudo abrir el archivo ICMP (IcmpCreateFile falló).

3 Error de Memoria No se pudo asignar memoria para la respuesta (malloc falló).

4 Fallo / Timeout Se envió el ping pero no hubo respuesta o hubo error en el envío (dwRetVal == 0).

    Modificamos la función para que acepte dominios que se convierten a IPs:



    Code: Select all Collapse
    </s>#pragma BEGINDUMP
#include <hbapi.h> 
#include <winsock2.h>
#include <iphlpapi.h>
#include <icmpapi.h>

// Nota: Asegúrate de enlazar con las librerías: ws2_32.lib e iphlpapi.lib

int hb_Ping( const char * cp )
{
    HANDLE hIcmpFile;
    unsigned long ipaddr;
    DWORD dwRetVal;
    char SendData[32] = "Data Buffer";
    LPVOID ReplyBuffer;
    DWORD ReplySize;
    

<i>    </i>// Estructuras para manejo de DNS
<i>    </i>WSADATA wsaData;
<i>    </i>struct hostent *host;

<i>    </i>// 1. Inicializar Winsock (Requerido para gethostbyname)
<i>    </i>if (WSAStartup(MAKEWORD(2, 2), &wsaData) != 0)
<i>    </i>    return 5; // Error: No se pudo iniciar Winsock

<i>    </i>// 2. Intentar convertir cadena IP numérica directa
<i>    </i>ipaddr = inet_addr( cp );
<i>    </i>
<i>    </i>// 3. Si no es numérica, intentar resolver como dominio (DNS)
<i>    </i>if (ipaddr == INADDR_NONE)
<i>    </i>{
<i>    </i>    host = gethostbyname( cp );
<i>    </i>    if ( host == NULL )
<i>    </i>    {
<i>    </i>        WSACleanup();
<i>    </i>        return 1; // Error: Nombre de dominio no válido o no encontrado
<i>    </i>    }
<i>    </i>    ipaddr = *(unsigned long *) host->h_addr_list[0];
<i>    </i>}

<i>    </i>hIcmpFile = IcmpCreateFile();
<i>    </i>if (hIcmpFile == INVALID_HANDLE_VALUE)
<i>    </i>{
<i>    </i>    WSACleanup();
<i>    </i>    return 2;
<i>    </i>}

<i>    </i>ReplySize = sizeof(ICMP_ECHO_REPLY) + sizeof(SendData);
<i>    </i>ReplyBuffer = (VOID*) malloc(ReplySize);
<i>    </i>if (ReplyBuffer == NULL)
<i>    </i>{
<i>    </i>    IcmpCloseHandle(hIcmpFile); // Cerrar handle antes de salir
<i>    </i>    WSACleanup();
<i>    </i>    return 3;
<i>    </i>}

<i>    </i>dwRetVal = IcmpSendEcho(hIcmpFile, ipaddr, SendData, sizeof(SendData), 
<i>    </i>    NULL, ReplyBuffer, ReplySize, 1000); // 1000ms de timeout

<i>    </i>// Limpieza de recursos
<i>    </i>free(ReplyBuffer);
<i>    </i>IcmpCloseHandle(hIcmpFile);
<i>    </i>WSACleanup();

<i>    </i>if (dwRetVal == 0)
<i>    </i>    return 4; // Ping fallido (Timeout o red inalcanzable)

<i>    </i>return 0; // Éxito
}

HB_FUNC( HB_PING )
{
   hb_retni( hb_Ping( hb_parc( 1 ) ) );
}
#pragma ENDDUMP<e>


    Cambios Clave Realizados

    WSAStartup: Para usar funciones de DNS (nombres de dominio), Windows requiere que se inicie el subsistema de sockets primero. Si esto falla, devuelve el nuevo error 5.



    gethostbyname: Si inet_addr dice que no es una IP numérica (ej. recibes "microsoft.com"), el código ahora entra en el if y pregunta al sistema por la IP de ese nombre.



    WSACleanup: Es muy importante liberar los recursos de sockets antes de que la función termine (he agregado estas llamadas antes de cada return).



    Limpieza de Memoria: Agregué free(ReplyBuffer) y IcmpCloseHandle antes de salir para evitar fugas de memoria si usas esta función muchas veces seguidas.



    Code: Select all Collapse
    </s>// 1. Funciona con IPs
nRes := HB_PING( "8.8.8.8" ) 

// 2. Funciona con Dominios
nRes := HB_PING( "google.com" )<e>
