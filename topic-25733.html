Bom Dia Luiz Fernando,

Quando voce informar o nome do arquivo para abrir coloque um espaço e /p

Code (fw): Select all Collapse
seu programa
....
cRun_Pdf := "relato.pdf /p"
executa_pdf()
.....
return nil

Function executa_pdf()
   //? cRun_Pdf
   If file(cRun_Pdf)
      Shellexecute(,"open",cRun_Pdf)
      ShellExecute( NIL,;
                      "open",;
                      cRun_Pdf  ,;
                      NIL   ,;
                      "c:\cialoja" ,;
                      NIL   ,;
                      SW_FORCEMINIMIZE )
   EndIf
Return .T.
