Code (fw): Select all Collapse
</s> oBar:bPainted  := { |hDC| PaintBarLogo( hDC, oBar ) }


//----------------------------------------------------------------------------//

static function PaintBarLogo( hDC, oBar )

   local oRect    := oBar:GetCliRect()
   local hBmp2, hBmp, nZeroZeroClr
   local hBmpOld
   local nWidth   := 128

   if File( "logo.bmp" )
      hBmp  := ReadBitmap( 0, "logo.bmp" )
   else
      hBmp  := LoadBitmap( GetResources(), "LINKMAX" )
   endif
   nWidth   := Min( 300, nBmpWidth( hBmp ) * 72.0 / nBmpHeight( hBmp ) )

   if HasAlpha( hBmp )
      hBmp2    := ResizeBitmap( hBmp, nWidth,72, 3 )
      DeleteObject( hBmp )
      ABPaint( hDC, oRect:nRight - nWidth - 2, oRect:nTop, hBmp2, 255 )
      DeleteObject( hBmp2 )
   else
      hBmpOld = SelectObject( hDC, hBmp )
      nZeroZeroClr = GetPixel( hDC, 0, 0 )
      SelectObject( hDC, hBmpOld )
      SetBkColor( hDC, nRGB( 255, 255, 255 ) )
      TransBmp( hBmp, nBmpWidth( hBmp ), nBmpHeight( hBmp ), nZeroZeroClr, ;
         hDC, oRect:nRight - nWidth - 2 , oRect:nTop, nWidth, oRect:nHeight )
      DeleteObject( hBmp )
   endif


return nil<e>
