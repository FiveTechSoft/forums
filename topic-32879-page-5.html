Function WebcamList()
   Local aDrivers    := {}, ;
         nDriver     := 0 , ;
         cName            , ;

      cName     :=  space(nLen)

      IF !wCamGetDrvDesc(nDriver, @cName, nLen, @cVersion, nLen)

      if chr(0)$cName
