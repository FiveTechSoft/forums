->Declarar una DATA lConservo que substituiría a la oPadre en el New(), ya que oPadre no le veo ningún uso.

->METHOD New(lConservo,cPatronSalida) CLASS TCalc
... / ...
::lConservo := lConservo
... / ...

->METHOD End() CLASS TCalc
Local nresultado := 0
nresultado := ::nResult

::aRolloCalc:={{'C',' '}}
::=0
::=''
::cUltimaop:=''
If !::lConservo
Self := Nil
EndIf
return nresultado
