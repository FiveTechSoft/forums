Local oSelf := Self

oTree:Add( oSelf:aEmpresas[i] ) ///

METHOD MyClick(nRow, nCol, oTree) CLASS tNewrol


local oItem
Local nAt
Local oUser
local n, lWhen
local oItem1

oItem := oTree:HitTest( nRow, nCol )

if oItem != NIL
nAt := Ascan( ::aEmpresas, oItem:cPrompt )

aFill( ::aPerms, .T. )
::SetMenu( ::MakeMenu() )
::Refresh()
::Settext( SISTEMA + 'Proy ' + oItem:cPrompt + ' Estación ' + NetName() )

else
MsgInfo( "No Hallado!", SISTEMA )
endif

RETURN Self
