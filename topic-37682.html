FUNCTION Licencia( lic1, oGet, oDlg )

   IF ( lic1 # "S" .AND. lic1 # "N" )

      MsgStop( "Error, Debe Digitar [S/N]" )

      oGet:VARPUT( Lic1 )
      oGet:Refresh()

      RETURN( .F. )

   ENDIF

   oDlg:Update()
   oDlg:Refresh()

RETURN( .T. )
