Gustavo,

add this code to your PRG:
Code: Select all Collapse
#pragma BEGINDUMP

#include <hbapi.h>
#include <windows.h>

HB_FUNC( TERMINATEPROCESS )
{
   hb_retl( TerminateProcess( ( HANDLE ) hb_parnl( 1 ),
                        IF( PCOUNT() > 1, hb_parni( 2 ), 0 ) ) );
}

#pragma ENDDUMP
regards, saludos
