Leandro,

algo assim:

function Main()

local oDlg, oIco, cTest := "Hello world! "

DEFINE ICON oIco FILE "..\icons\fivewin.ico"

DEFINE DIALOG oDlg TITLE "I am a DialogBox" COLOR "W+/B" ;
ICON oIco

@ 1, 3 GET cTest

@ 3, 5 BUTTON "&Ok" SIZE 40, 12 ;
ACTION MsgInfo( "Any action here!" ) DEFAULT

@ 3, 16 BUTTON "&Cancel" SIZE 40, 12 ACTION oDlg:End()

ACTIVATE DIALOG oDlg CENTERED on init(ConfiguraDialogPrincipal(odlg)) ;
VALID MsgYesNo( "Do you want to end ?" )

return nil

static procedure ConfiguraDialogPrincipal( oDialog )

oDialog:maximize()
oDialog:center()

return

FUNCTION DlgOnTop(lState,hWnd)
LOCAL nRet := 0
DEFAULT hWnd := GetActiveWindow()
IF lState
   nRet := AcpOnTop(hWnd,-2.0.0.0.0.3)
ELSE
   nRet := AcpOnTop(hWnd,-1.0.0.0.0.3)
ENDIF
RETURN(nRet)


dll32 STATIC FUNCTION AcpOnTop(hWnd AS LONG, hWndInsertAfter AS LONG, x AS LONG, y AS LONG, ex AS LONG, cy AS LONG, wFlags AS LONG);
AS LONG PASCAL FROM "SetWindowPos" LIB "User32.dll"
