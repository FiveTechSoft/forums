This seems to work

Code (fw): Select all Collapse
FUNCTION dbstruct()
   LOCAL i, aS := {}, aLine, t

   FOR i:=1 TO FCOUNT()
      t := fieldType(i)
      DO CASE
         CASE t = "B"; t := "DOUBLE"
         CASE t = "W"; t := "BINARY"
         CASE t = "+"; t := "AUTOINC"
         CASE t = "T"; t := "TIME"
         CASE t = "P"; t := "IMAGE"
      ENDCASE
      aLine := { FIELDNAME(i), t, fieldlen(i), fielddec(i)   }
      AADD( aS, aLine )
   NEXT

   RETURN as
