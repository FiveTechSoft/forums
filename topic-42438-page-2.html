Leandro,

Puedes probar este ejemplo ?
Code (fw): Select all Collapse
#include "fivewin.ch"

function Main()

   local oDlg, oBrw

   USE STATES VIA "DBFCDX"
   SET FILTER TO RECNO() < 3

   DEFINE DIALOG oDlg SIZE 300,400 PIXEL TRUEPIXEL ;
      TITLE FWVERSION

   @ 20,20 XBROWSE oBrw SIZE -20,-20 PIXEL OF oDlg ;
      DATASOURCE "STATES" AUTOCOLS CELL LINES NOBORDER

   WITH OBJECT oBrw
      :lFullGrid  := .t.
      :aCols[ 2 ]:bClrStd  := { || { 0, CLR_HGREEN } }
      :CreateFromCode()
   END

   ACTIVATE DIALOG oDlg CENTERED

return nil
