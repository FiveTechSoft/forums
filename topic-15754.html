Hi SIlvio:

Function GetMacAdd()
local aMac, cRet

TRY
aMac := GetNetCardID()
CATCH
aMac := ""
END

IF VALTYPE(aMac) = "U" // no hay tarjera de red instalada
Return("")
EndIf
IF VALTYPE(aMac) = "A" // hay varias tarjetas de red
If LEN(aMac) = 0
cRet := ""
Else
cRet := aMac[1]
EndIf
EndIf
If VALTYPE(aMac) = "C"
cRet := aMac // una sola t de red
EndIf
Return(cRet)
