                PORT StrPort ;
                DATABASE StrDatabase ;
		USER StrUserID ;
                PASSWORD StrUserPWD ;
                OPTIONS 35 ;
		DRIVER StrDriver

    ADO EXECUTE StrSqlCommand

    ADO GOTOP

    Browse( oRs )

    ADO CLOSE

    RETURN NIL

function ADOSkipper( oRs, nSkip ) 
    LOCAL nRec := oRs:AbsolutePosition 
    oRs:Move( nSkip ) 
    IF oRs:EOF; oRs:MoveLast(); ENDIF 
    IF oRs:BOF; oRs:MoveFirst(); ENDIF 
    RETURN oRs:AbsolutePosition - nRec
