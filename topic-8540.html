function Main()

   local oWnd

   ACTIVATE WINDOW oWnd ;
      ON INIT BuildChild( oWnd )

return nil

function BuildChild( oWnd )

   local oChild

   DEFINE WINDOW oChild TITLE "Test" MDICHILD

   WndTop( oChild:hWnd, oWnd:nHeight / 7 )
   WndLeft( oChild:hWnd, oWnd:nWidth / 10 )

return nil
