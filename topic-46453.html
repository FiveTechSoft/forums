</i>#include "FiveWin.ch"
#include "xbrowse.ch"

function Main()
   local oDlg, oBrw
   local hBmp1 := LoadBitmap( GetResources(), "imagen1.bmp" )
   local hBmp2 := LoadBitmap( GetResources(), "imagen2.bmp" )

   // ... (Configuración de oDlg y oBrw)

   WITH OBJECT oBrw:aCols[1]
      :cHeader := "Columna"
      // Bloque de código para personalizar el pintado
      :bPaintHeader := { | oCol, hDC, aRect | 
         // Dibujar la primera imagen a la izquierda
         DrawBitmap( hDC, hBmp1, aRect[1], aRect[2] + 2, 16, 16 ) 
         

<i>    </i>     // Dibujar la segunda imagen a la derecha
<i>    </i>     DrawBitmap( hDC, hBmp2, aRect[1], aRect[4] - 18, 16, 16 )
<i>    </i>     
<i>    </i>     // Dibujar el texto del encabezado centrado (opcional)
<i>    </i>     oCol:DrawHeader( hDC, aRect )
<i>    </i>  }
   END

   // ... (oBrw:CreateFromCode())
   // ... (ACTIVATE DIALOG)
   

   // No olvidar liberar las imágenes
   DeleteObject( hBmp1 )
   DeleteObject( hBmp2 )
return nil
