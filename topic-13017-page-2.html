#include "FiveWin.ch"

function Main()

   local oDlg, oGet, cTest := Space( 20 )
   
   DEFINE DIALOG oDlg
   
   @ 2, 2 GET oGet VAR cTest ACTION MsgInfo( "action" ) VALID ( MsgInfo( "valid" ), .T. )
   
   ACTIVATE DIALOG oDlg CENTERED ;
