hi Silvio,

try this
Code (fw): Select all Collapse
PROCEDURE BinCalc(nValue)
LOCAL aImports:={200,100,50,20,10,5,3,2,1,0.50,0.20,0.10,0.05}
LOCAL nPosi := 0
LOCAL nRest := nValue
LOCAL aRet   := {}

   DO WHILE .T.
?     nPosi := ASCAN(aImports,{|e| e <= nRest})
      IF nPosi > 0
         AADD(aRet,nPosi)
         nRest -= aImports[nPosi]
      ELSE
         EXIT
      ENDIF
   ENDDO
RETURN aRet
