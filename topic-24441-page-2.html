pragma BEGINDUMP

include <windows.h>

include <hbapi.h>

HB_FUNC( NEXTMEM ) // --> nHardwareMemory
{
MEMORYSTATUSEX mst;

memset( &mst, 0, sizeof( MEMORYSTATUSEX ) );
mst.dwLength = sizeof( MEMORYSTATUSEX );

GlobalMemoryStatusEx( &mst );

hb_retnll( mst.ullTotalPhys );
}

HB_FUNC( NAVAILMEM ) // --> nHardwareMemory
{
MEMORYSTATUSEX mst;

mst.dwLength = sizeof( MEMORYSTATUSEX );

}
