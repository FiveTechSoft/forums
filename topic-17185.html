Con la TSBUTTON, hago asi para incorporar a mi proyecto:

vou responder, antes que algum <administrador> me ferre... kkkkkkkkk

REM // <<- ESTES ‚ .C PARA TSBUTTON
COPY C:\PIZZAS\TSB\BPAINT.C /Y
REM // <<- ESTES ‚ .C PARA TSBUTTON
COPY C:\PIZZAS\TSB\ROUNDREC.C /Y

REM -> CLASSES DA TSBUTTON
%hdir%\bin\harbour TSBUTTON /es2/m/n/a/w2 /i%fwhdir%\include;%hdir%\include /w /p TSBUTTON.C >> Erro.log
%hdir%\bin\harbour TSBAR /es2/m/n/a/w2 /i%fwhdir%\include;%hdir%\include /w /p TSBAR.C >> Erro.log
%hdir%\bin\harbour TSLINES /es2/m/n/a/w2 /i%fwhdir%\include;%hdir%\include /w /p TSLINES.C >> Erro.log
REM -> FIM DA TSBUTTON

REM -> CLASSES DA TSBUTTON
%bcdir%\bin\bcc32 -O2 -M -c -D__HARBOUR__ -I%hdir%\include TSBUTTON.C >> Erro.log
%bcdir%\bin\bcc32 -O2 -M -c -D__HARBOUR__ -I%hdir%\include TSBAR.C >> Erro.log
%bcdir%\bin\bcc32 -O2 -M -c -D__HARBOUR__ -I%hdir%\include TSLINES.C >> Erro.log
REM - FIM DAS CLASSES TSBUTTON

REM -> TSBUTTON
%FwhDir%\Tools\CLP2HARB BPAINT.C BPAINT.HC
%FwhDir%\Tools\CLP2HARB ROUNDREC.C ROUNDREC.HC
