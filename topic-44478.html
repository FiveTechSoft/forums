Function ChangeSizeBrw( l01, l02, oBox, oBrw )
   LOCAL nHeight
   
   DO CASE
      CASE !l01 .AND. !l02
           nHeight := 390
           oBox[2]:Hide()
           oBox[3]:Hide()
      CASE l01 .AND. !l02
           nHeight := 340
           oBox[2]:Show()
           oBox[3]:Hide()
      CASE !l01 .AND. l02
           nHeight := 310
           oBox[2]:Hide()
           oBox[3]:Show()
      CASE l01 .AND. l02
           nHeight := 310
           oBox[2]:Show()
           oBox[3]:Show()
   ENDCASE
   
   oBrw:SetSize( oBrw:nWidth , nHeight, .T. )
   oBrw:refresh(.f.)

Return(NIL)
