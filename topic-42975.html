function RemoveANSIEscapeCodes( cString )

   local cCleanString := "", n

   for n := 1 to Len( cString )
      cChar := SubStr( cString, n, 1 )
      if cChar == Chr( 27 )
         while cChar != "m"
            n++
            cChar = SubStr( cString, n, 1 )
         end      
         cChar = ""
      endif   

      cCleanString := cCleanString + cChar
   next

return cCleanString
