This has been working fine for me:

FUNCTION SendMAPIMail2( MailTo, cSndFile, cText, cSubj )
LOCAL oMail

  DEFINE MAIL oMail ;
     SUBJECT cSubj ;
     TEXT cText ;
     FILES cSndFile, cSndFile ;
     FROM USER ;
     TO MailTo

  ACTIVATE MAIL oMail

RETURN( .t. )
