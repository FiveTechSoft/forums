// C:\FWH\SAMPLES\TESTICON.PRG - MODIFICADO EN: 07/10/2024

// Playing with icons...

#include "FiveWin.ch"

STATIC oWnd, nOption := 1

// ----------------------------------------------------------------------------//

FUNCTION Main()

   LOCAL oIco

   DEFINE ICON oIco RESOURCE "ICONE01"  // EN TESTICON.RC

   DEFINE WINDOW oWnd FROM 1, 1 TO 20, 44 ;
      TITLE OemToAnsi( "Showing icons from disk" ) ;
      COLOR "W/B" ICON oIco

   ACTIVATE WINDOW oWnd ;
      ON INIT SomeIcons( oWnd )

RETURN NIL

// ----------------------------------------------------------------------------//

FUNCTION SomeIcons( oWnd )

   LOCAL oCursor, oIco, oSay1, oSay2, oSay3, oSay4, aSays

   DEFINE CURSOR oCursor HAND

   @ 2, 2 ICON oIco FILENAME "..\icons\fred.ico" OF oWnd ;
      ON CLICK MsgInfo( "Hi, I am Pedro Picapiedra" )

   oIco:oCursor = oCursor
   oIco:bMMoved = {|| Hilite( aSays, 1 ) }

   @ 6, 2 ICON oIco FILENAME "..\icons\FiveWin.ico" BORDER OF oWnd ;
      ON CLICK MsgInfo( OemToAnsi( "FiveWin - The xBase revolution" ) )

   oIco:oCursor = oCursor
   oIco:bMMoved = {|| Hilite( aSays, 2 ) }

   @ 10, 2 ICON oIco FILENAME "..\icons\Dialog2.ico" OF oWnd ;
      ON CLICK MsgInfo( OemToAnsi( "What do you think ?" ) )

   oIco:oCursor = oCursor
   oIco:bMMoved = {|| Hilite( aSays, 3 ) }

   @ 14, 2 ICON oIco FILENAME "..\icons\Clipper.ico" OF oWnd ;
      ON CLICK MsgInfo( OemToAnsi( "An easy menu based on icons ?" ) )

   oIco:oCursor = oCursor
   oIco:bMMoved = {|| Hilite( aSays, 4 ) }

   @  2.5, 7 SAY oSay1 PROMPT "make your selection" SIZE 200, 20 OF oWnd

   oSay1:lWantClick = .T.
   oSay1:bMMoved = {|| Hilite( aSays, 1 ) }
   oSay1:oCursor = oCursor

   @    7, 7 SAY oSay2 PROMPT "Structured programming" SIZE 200, 20 OF oWnd

   oSay2:lWantClick = .T.
   oSay2:bMMoved = {|| Hilite( aSays, 2 ) }
   oSay2:oCursor = oCursor

   @   11, 7 SAY oSay3 PROMPT "FiveWin is Clipper, Windows & OOPS" SIZE 250, 20 OF oWnd

   oSay3:lWantClick = .T.
   oSay3:bMMoved = {|| Hilite( aSays, 3 ) }
   oSay3:oCursor = oCursor

   @ 15.3, 7 SAY oSay4 PROMPT FWCOPYRIGHT SIZE 200, 20 OF oWnd

   oSay4:lWantClick = .T.
   oSay4:bMMoved = {|| Hilite( aSays, 4 ) }
   oSay4:oCursor = oCursor

   aSays = { oSay1, oSay2, oSay3, oSay4 }

RETURN NIL

// ----------------------------------------------------------------------------//

FUNCTION Hilite( aSays, n )

   IF n != nOption
      AEval( aSays, {| o | o:SetColor( "W/B" ), o:Refresh() } )
      aSays[ n ]:SetColor( "W+/B" )
      aSays[ n ]:Refresh()
      nOption = n
   ENDIF

RETURN NIL
