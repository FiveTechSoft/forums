Aunque no creo que se algo muy importante, en los formatos de fehca ISO, suele ponerse la diferencia horaria, o en su defecto grabarlo en GMT e indicarlo con una "Z" al final.


Code (fw): Select all Collapse
function IsoDateTime()
   LOCAL nDif,cSigno
   nDif:=GmtSecDif()/3600
   cSigno:=IIf(nDif<0,'-','+')
   nDif:=Abs(nDif)
return( sDate() +"T"+ time()+cSigno+StrZero(nDif,2)+':00' )


Code (fw): Select all Collapse
HB_FUNC( GMTSECDIF )                /*Diferencia en segundos entre GMT y local Time */
{
   struct tm *tptr;
   time_t secs, local_secs, gmt_secs;
   long diff_secs;
   time( &secs );  
   tptr = localtime( &secs );
   local_secs = mktime( tptr );
   tptr = gmtime( &secs );
   gmt_secs = mktime( tptr );
   diff_secs = local_secs - gmt_secs;
   hb_retnl( diff_secs);
}
