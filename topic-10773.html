CLASS TCLIENTE // Clase para DATA DE CLIENTES

DATA      cNombre             INIT Space(40)

DATA      cCedula             INIT space(10)

DATA      cEmail              INIT space(30)

ENDCLASS

//******************************************************************
//
METHOD New(aData) CLASS TCLIENTE
//*******************************************************************

IF aData&lt;&gt;nil
  ::nTpoCliente         := aData[6]
  ::cNombre             := IIf(::nTpoCliente=1,aData[1],"")
  ::cDenom              := IIf(::nTpoCliente&lt;&gt;1,aData[1],"")
  ::cApellido           := aData[2]
  ::cCedLetra           := Substr(adata[5],1,1)

  ::cNit                := aData[4]
  ::cRifLetra           := IIF(::nTpoCliente&lt;&gt;1,Substr(adata[3],1,1)," ")
  ::cRif                := IIF(::nTpoCliente&lt;&gt;1,iif(len(alltrim(aData[3]))&gt;0,Substr(adata[3],3,13),space(10)),"")
  ::cRifNat             := IIf(::nTpoCliente=1,iif(len(alltrim(aData[3]))&gt;0,Substr(adata[3],3,13),space(10)),"")
  ::cRifNatLetra        := IIF(::nTpoCliente=1,Substr(adata[3],1,1)," ")
  ::nTpoEstadoCivil     := aData[7]
  ::nTpoGenero          := aData[8]

  ::nTpoActividad       := aData[10]
  ::nTpoClienteRegistro := aData[11]
  ::nIdDtsBanco         := aData[12]

ENDIF

  ENDIF
ENDIF
RETURN lOk
