FUNCTION fEof( nHandle ) 
   LOCAL nCurPos, nEOF 
   nCurPos := fSeek( nHandle, 0, 1 )
   nEOF    := fSeek( nHandle, 0, 2 )
   fSeek( nHandle, nCurPos, 0 )
RETURN nEOF <= nCurPos
