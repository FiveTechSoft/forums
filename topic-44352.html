Function Hide_Ctrls(oBrw,oGet,oSay,lSearchBar,aBtn,oBar)

local oBtn,n



If lSearchBar

oSay[1]:show()

oSay[2]:show()

oGet:show()

oBrw:show()

oBrw:=32

oBrw:= oBar:ntop-40

else

oSay[1]:hide()

oSay[2]:hide()

oGet:hide()

oBrw:hide()

oBrw:=2

oBrw:= oBar:ntop-40

Endif



for n=1 to Len(aBtn)

aBtn[n]:= oBar:ntop-40

aBtn[n]:refresh()

next



return nil
