function logline( cText, cWert )
  
  local cLog
  local cFile := ""
  local cPRGPATH := ""
  
  cPRGPATH := hb_GetEnv( "PRGPATH" )
  cPRGPATH := STRTRAN( cPRGPATH  ,"/","\")
  
  cFile := cPRGPATH + "\log.log"
  
  cLog := memoread( cFile )
  
  if VALTYPE(cText) = "U"
    cText := ""
  endif
  if VALTYPE(cWert) = "U"
    cWert := ""
  endif
  cLog   +=  ALLTRIM( str(procline(1)) ) + "    "  + PadR( cText, 40 ) + ":  " + cWert + CRLF
  
  MEMOWRIT( cFile, cLog, .f. )
  
return nil
