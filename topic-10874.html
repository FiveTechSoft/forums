Aqui tienes un ejemplo funcionando:
Code: Select all Collapse
#include "FiveWin.ch"

function Main()

   local oDlg, oTabs

   DEFINE DIALOG oDlg RESOURCE "Test"

   REDEFINE TABS oTabs PROMPTS "One", "Two", "Three" ID 100 OF oDlg

   oDlg:oTop = oTabs

   ACTIVATE DIALOG oDlg CENTERED

return nil

test.rc
