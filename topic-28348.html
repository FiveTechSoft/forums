             aDat[2]:= S99->REFERE  // ID Principal de referencias folios o facturas

             DO WHILE S99->(!EOF())
                 IF S99->REFERE <> aDat[2] // ID Referencia
                     IF ( ROUND(aSum[3],2) <> 0.00 )
                         aTot[1] += aSum[1]  // Total Cargos
                         aTot[2] += aSum[2]  // Total Abonos

                     ENDIF

                     AFill(aSum,0)
                 ENDIF

                 IF S99->CUENTAS > cCtaIni
                     EXIT
                 ENDIF

                 aDat[2]:= S99->REFERE  // ID Principal de referencias folios o facturas
