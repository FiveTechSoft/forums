include "ADS.ch"

RddRegister("ADS",1)
RddSetDefault("ADS")
SET SERVER LOCAL
SET FILETYPE TO NTX

USE STOCK INDEX STOCK01,STOCK02,STOCK03,STOCK04,STOCK05,STOCK06,;
STOCK07,STOCK08,STOCK09,STOCK10,STOCK11,STOCK12,STOCK13 NEW
DATABASE oDbf
oDbf:bEoF=nil
oDbf:SetOrder(2)
oDbf:GoTop()

Esto es basicamente lo que probe, por supuesto que el mismo problema ocurre con cualquier dbf que modifique con fecha de hoy exclusivamente.
Este codigo esta compilado y funcionando con FWH hace 3 años en los cuales sufrio cientos de modificaciones y jamas paso esto.
