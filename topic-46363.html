</s>FUNCTION SendSmtpDesatendido( cDe, cPara, cAsunto, cCuerpo, cPassword )
   LOCAL oCfg, oMsg
   LOCAL cSchema := "http://schemas.microsoft.com/cdo/configuration/"

   TRY
      oMsg := CreateObject("CDO.Message")
      oCfg := CreateObject("CDO.Configuration")

<i>    </i>  // Configuración del Servidor SMTP de Office 365
<i>    </i>  oCfg:Item( cSchema + "sendusing"      ):Value := 2 // cdoSendUsingPort
<i>    </i>  oCfg:Item( cSchema + "smtpserver"     ):Value := "smtp.office365.com"
<i>    </i>  oCfg:Item( cSchema + "smtpserverport" ):Value := 587
<i>    </i>  oCfg:Item( cSchema + "smtpusessl"     ):Value := .T.
<i>    </i>  oCfg:Item( cSchema + "smtpauthenticate" ):Value := 1 // cdoBasic
<i>    </i>  oCfg:Item( cSchema + "sendusername"   ):Value := cDe
<i>    </i>  oCfg:Item( cSchema + "sendpassword"   ):Value := cPassword
<i>    </i>  oCfg:Update()

<i>    </i>  oMsg:Configuration := oCfg
<i>    </i>  oMsg:From    := cDe
<i>    </i>  oMsg:To      := cPara
<i>    </i>  oMsg:Subject := cAsunto
<i>    </i>  oMsg:TextBody := cCuerpo

<i>    </i>  oMsg:Send()
<i>    </i>  ? "Correo enviado correctamente por SMTP (Desatendido)"

   CATCH oErr
      ? "Error al enviar: " + oErr:Description
   END
RETURN NIL<e>
