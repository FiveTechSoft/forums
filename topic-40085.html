Code (fw): Select all Collapse
METHOD PEOPEN_()

   LOCAL cFarProc
   LOCAL cFunc := "PEOpenEngine"

   cFarProc := GetProcAddress( ::hDLL, cFunc, .T., LONG )

   ::lError := ( FWCallDll( cFarProc ) == FALSE )

   IF ::lError

      MsgAlert( "Could not open the Print Engine!" + CRLF + ::PeErrorText())

   ENDIF

RETURN( .NOT. ::lError )
