Code (fw): Select all Collapse
#include "dbinfo.ch"
ANNOUNCE RDDSYS


FUNCTION MAIN
LOCAL nInizio  := SECONDS()
LOCAL nContati := 0 
LOCAL aArray   := {}
SET EXCLUSIVE OFF

USE n:\hse\la
SET INDEX TO n:\hse\la
dbInfo( DBI_SHARED, .F. )

SET ORDER TO 1
GO TOP 
DO WHILE !EOF()         

   AADD( aArray , field->modello ) 
   SKIP        
   inkey()
ENDDO
? SECONDS() - nInizio             
? LEN( aArray ) 

INIT PROCEDURE RddInit
REQUEST DBFFPT
REQUEST DBFCDX
rddSetDefault( "DBFCDX" )
RETURN
