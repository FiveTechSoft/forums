Muy bien, Rodolfo, te habia preparado esto, no se me ocurrió ver MoveCol(): (vaya que reinventamos la rueda, ja-ja-ja)

Function InterCamb(oBrw,nCol1,nCol2)
local n, nPosAct:=nCol1
local nNvaPos:=0
For n:= 1 to nCol1
nNvaPos:= nPosAct-1
oBrw:SwapCols(nPosAct,nNvaPos)
oBrw:Refresh()
nPosAct-=1
Next
Return nil
