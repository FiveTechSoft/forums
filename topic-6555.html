Code: Select all Collapse
#include "Fivewin.ch"


FUNCTION MAIN()

    LOCAL oDlg

    LOCAL cVar := SPACE( 30 )

    SETKEY( VK_F1, { || MyHelp() } )

    DEFINE DIALOG oDlg

    @ 1, 1 GET cVar

    @ 3, 1 BUTTON "&Close" ACTION oDlg:End()

    ACTIVATE DIALOG oDlg;
             CENTER

    RETURN NIL


STATIC FUNCTION MYHELP()

    ? "Help!"

    RETURN NIL


FUNCTION HELPINDEX()

    RETURN NIL


FUNCTION HELPTOPIC()

    RETURN NIL


FUNCTION HELPPOPUP()

    RETURN NIL
