//----------------------------------------------------------------------------//

Function AllHSysBitmaps()

   local oBrw
   local aHBmps   := {}
   local aBmps    := {}
   local hBmp
   local x

   For x = 32000 to 35799
       hBmp := HSysBitmap( x )
       if !Empty( hBmp )
          AAdd( aHBmps, x ) 
          AAdd( aBmps, hBmp )
       endif
   Next x     
   XBrowse( aHBmps, "Bitmaps System",, { | o | PonBmps( o, aBmps ) } )
   AEval( aBmps, { | h | DeleteObject( h ) } )
    
Return nil

//----------------------------------------------------------------------------//

Function PonBmps( o, aBmpH )

   o:aCols[ 1 ]:AddBitmap( aBmpH )
   o:aCols[ 1 ]:bBmpData := { || o:nArrayAt }

Return nil

//----------------------------------------------------------------------------//
