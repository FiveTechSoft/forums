Tim,

Have you tried turning on oSocket:lDebug? This logs all the communications. I think you will have to modify TSMTP since it creates the socket object and there seems to be no way to turn on oSocked:lDebug without modifying the TSMTP source. It really should have its own switch.

Also, some SMTP servers require that you be logged into the internet on the same ISP or you cannot use the SMTP server.

I suggest you this sample from asnker:

#Include &quot;FiveWin.ch&quot;

Function Main()
    Local oEmailCfg,oEmailMsg

    TRY
       oEmailCfg := CREATEOBJECT( &quot;CDO.Configuration&quot; )
      WITH OBJECT  oEmailCfg:Fields
         :Item( &quot;http://schemas.microsoft.com/cdo/configuration/smtpserver&quot; ):Value :=   &quot;mail.xxxxxxxx.com&quot;
         :Item( &quot;http://schemas.microsoft.com/cdo/configuration/smtpserverport&quot; ):Value :=  25
         :Item( &quot;http://schemas.microsoft.com/cdo/configuration/sendusing&quot; ):Value := 2   // Remote SMTP = 2, local = 1
         :Item( &quot;http://schemas.microsoft.com/cdo/configuration/smtpauthenticate&quot; ):Value :=   .T.
         :Item( &quot;http://schemas.microsoft.com/cdo/configuration/smtpusessl&quot; ):Value :=  .F.
         :Item( &quot;http://schemas.microsoft.com/cdo/configuration/sendusername&quot; ):Value :=  &quot;xxanser@xxxxxxxx.com&quot;
         :Item( &quot;http://schemas.microsoft.com/cdo/configuration/sendpassword&quot; ):Value :=  &quot;xxxxxx&quot;
         :Item( &quot;http://schemas.microsoft.com/cdo/configuration/smtpconnectiontimeout&quot;):Value := 30
         :Update()
      END WITH
    CATCH oError
      MsgInfo( &quot;Could not create message configuration&quot; + &quot;;&quot;  + ;
             &quot;Error: &quot; + TRANSFORM(oError:GenCode, NIL) + &quot;;&quot; + ;
             &quot;SubC: &quot; + TRANSFORM(oError:SubCode, NIL) + &quot;;&quot; + ;
             &quot;OSCode: &quot; + TRANSFORM(oError:OsCode, NIL) + &quot;;&quot; + ;
             &quot;SubSystem: &quot; + TRANSFORM(oError:SubSystem, NIL) + &quot;;&quot; + ;
             &quot;Message: &quot; + oError:Description )
       Return .F.
    END
    oError:=NIL


    TRY
     oEmailMsg := CREATEOBJECT ( &quot;CDO.Message&quot; )
     WITH OBJECT oEmailMsg
        :Configuration =  oEmailCfg
        :From = chr(34)+&quot; Anser K.K. &quot;+chr(34)+ &quot;&lt;anser@xxxxxxxx.com&gt;&quot; // This will be displayed in the From (The email id does not appear)
        :To = &quot;xxanserkk@xxxxx.com&quot;    // &lt;-----   Place the TO email address
        :Subject =  &quot;This is a Tst message&quot;
        :ReplyTo =  &quot;xxanser@xxxxxxxxx.com&quot;
        :Sender =  &quot;xxanser@xxxxxxxxx.com&quot;  // Read Receipt message is send to this
        :Organization =  &quot;My xxxxxx Company&quot;    // &quot;My Company Name&quot;
        :HTMLBody =  &quot;&lt;HTML&gt; Hello  &lt;/HTML&gt;&quot;
        :Send()
     END WITH
     SysRefresh()
    CATCH oError

       MsgInfo( &quot;Could not send message&quot; + &quot;;&quot;  + CRLF+ ;
         &quot;Error: &quot; + TRANSFORM(oError:GenCode, NIL) + &quot;;&quot; + CRLF+;
         &quot;SubC: &quot; + TRANSFORM(oError:SubCode, NIL) + &quot;;&quot; + CRLF+ ;
         &quot;OSCode: &quot; + TRANSFORM(oError:OsCode, NIL) + &quot;;&quot; + CRLF +;
         &quot;SubSystem: &quot; + TRANSFORM(oError:SubSystem, NIL) + &quot;;&quot; +CRLF+ ;
         &quot;Message: &quot; + oError:Description )
       Return .F.

    END
    MsgInfo(&quot;Email Send&quot;)

Return
