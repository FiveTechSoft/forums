Sergio,

Solucionado

hazlo asi:
Code (fw): Select all Collapse
#include "FWCE.ch" 

function Main() 

   local oWnd, cValue :="One", oCombo

   DEFINE window oWnd TITLE "Combos" 

   @ 1, 1 COMBOBOX oCombo VAR cValue ;
      ITEMS { "One", "Two", "Three" } OF oWnd SIZE 120, 120

   ACTIVATE window oWnd ;
      ON INIT oCombo:SetSize( 120, 120 )

return nil
