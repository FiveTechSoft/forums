function noslashending( cPathMask )  // returns cPathMask without slash ending
    local cTmp := ALLTRIM( cPathMask )
    
    if right(cTmp,1) = "\" .or. right(cTmp,1) = "/"
        cPathMask := left( cPathMask, ( len(cPathMask)- 1 ) )
    endif
return cPathMask
