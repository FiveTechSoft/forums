Silvio,

a sample :

from ORD.CH :

#define TOPSCOPE 0
#define BOTTOMSCOPE 1

set order to tag dates
// set range
dates->(ordScope(0, dtos(Start) )) // set top scope
dates->(ordScope(1, dtos(End) )) // set bottom scope
dates->(dbgotop())
dates->(ordScope(0, NIL )) // set top to NIL
dates->(ordScope(1, NIL )) // set bottom to NIL
