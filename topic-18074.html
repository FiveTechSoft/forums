With FWH 10.1 ( should also work with 9.12 )
Code (fw): Select all Collapse
#include 'fivewin.ch'
#include 'xbrowse.ch'

function Main()

   local oDlg, oBrw, oFont, d, n
   local aData := {}
   local aGrad := { { .5, RGB( 193, 219, 252 ), RGB( 220, 235, 252 ) }, ;
                    { .5, RGB( 220, 235, 252 ), RGB( 193, 219, 252 ) }  }

   d := {^ 2000/01/01 }
   for n := 1 to 12
      AAdd( aData, { cMonth( d ) } )
      d  := AddMonth( d, 1 )
   next


   DEFINE FONT oFont NAME 'TAHOMA' SIZE 0,-14
   DEFINE DIALOG oDlg RESOURCE 'TEST' FONT oFont

   REDEFINE XBROWSE oBrw ID 101 OF oDlg ;
      COLUMNS 1 ;
      ARRAY aData CELL

   WITH OBJECT oBrw
      :bClrSelFocus     := { || { CLR_BLACK, aGrad } }
      :nStretchCol      := 1
      :lHeader          := ;
      :lRecordSelector  := .f.
   END

   ACTIVATE DIALOG oDlg CENTERED
   RELEASE FONT oFont

return nil
