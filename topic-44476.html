FUNCTION CHANGE_INDICES_CASADO( nOpcRad, cAlias, oBrw )

   LOCAL lCasado := .T., lNoCasado := .F.

   IF nOpcRad     == 2 // No Casados

      SET ORDER TO 02  // MARRIED

      ORDSCOPE( 0, lNoCasado )
      ORDSCOPE( 1, lNoCasado )
      GO TOP

      IF EOF()

         ORDSCOPE( 0, Nil )
         ORDSCOPE( 1, Nil )
         GO TOP

         SELECT( cAlias )
         SET ORDER TO 01
         GO TOP

      ENDIF

   ELSEIF nOpcRad == 1 //->  CASADOS

      SET ORDER TO 02  // MARRIED

      ORDSCOPE( 0, lCasado )
      ORDSCOPE( 1, lCasado )
      GO TOP

      IF EOF()

         ORDSCOPE( 0, Nil )
         ORDSCOPE( 1, Nil )
         GO TOP

         SELECT( cAlias )
         SET ORDER TO 01
         GO TOP

      ENDIF

   ENDIF

   GO TOP

   oBrw:GoBottom()
   oBrw:Refresh()
   oBrw:Gotop()

RETURN NIL
