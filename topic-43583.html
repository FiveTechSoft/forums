Estimado Gustavo,



Tienes que modificar el método End() en source\classes\TTVITEM.PRG de esta manera:
Code (fw): Select all Collapse
</s>METHOD End() CLASS TTVItem

   local nAt := 0

   if ! Empty( ::oParent )
      nAt := AScan( ::aItems, { | o | o == Self } )
   endif

   if nAt != 0
      if ::IsKindOf( "TTREEVIEW" )
         ADel( ::aItems, nAt )
         ASize( ::aItems, Len( ::aItems ) - 1 )
      else
         ADel( ::aItems, nAt )
         ASize( ::aItems, Len( ::aItems ) - 1 )
      endif
   endif

   TVDeleteItem( ::hWnd, ::hItem )

return nil
 <e>
