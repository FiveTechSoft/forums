function TESTBROWSE()

   local oDlg
   local oBrw

   define dialog oDlg title "Test xBrowse " + cvaltochar(datetime()) from 0,0 to 24,79

Gracias a la ayuda de Cristóbal, este es el fix:

Code (fw): Select all Collapse
METHOD KeyCount() CLASS TXBrowse

   TRY
      if Empty( ::aFilter )
         ::nLen   := Eval( ::bKeyCount )
      else
         ::nLen   := Len( ::aFilter )
      endif
      ::lClosed   := .f.

   CATCH
      ::nLen      := 0
      ::lClosed   := .t.
   END


   if ::oVScroll != nil
      ::VSetRange( 1, ::nLen )

      if !::lSqlRDD
        ::VUpdatePos()
      endif
   endif

return ::nLen
