function ToolTipIcon()

   local oDlg, oBtn

   SetBalloon( .t. )

   DEFINE DIALOG oDlg SIZE 300,100 PIXEL TRUEPIXEL ;
      TITLE "TOOLTIP ICON"

   @ 20,50 BTNBMP oBtn PROMPT "TEST" SIZE 100,60 PIXEL OF oDlg ;
      2007 CENTER ;
      TOOLTIP { "This is the text", "TOOLTIP", "FACE" }

   ACTIVATE DIALOG oDlg CENTERED

return nil
