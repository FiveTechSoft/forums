La verdad lo que estaba haciendo era poder imprimir en una impresora serial NCR 7167 termica, ocupe los mismo comandos de que uso en clipper y me funciona OK.

     xCOM:='COM1'

     hpos:=Fopen(xCOM,1)

     texto := &quot;texto que incluye, caracteres de control y todo lo que se quiera imprimir&quot;

     While !resp7
            txt = Left(texto, 1024)
            nbytes = Fwrite(hpos, txt); linkey(5)
            texto = Subs(texto, nbytes+1); resp7 = Empty(Alltrim(texto))
     Enddo

     Fclose(hpos)
