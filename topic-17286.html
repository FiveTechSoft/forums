STATIC FUNCTION ImpRep( lModal, oDlg )
LOCAL nTImp, nTPago, nTSaldo, nTxven, nT1a30, nT31a60, nT61a90, ;
nTmas90
LOCAL mCliNum, mCliNom, cCliAte, cDirec, cColon, cCiudad, cEstado, ;
cCodPos, TotPag, SaldoFac
LOCAL cFacNum, nImpFac, dVencFac, dFechaFac

nTImp := nTPago := nTSaldo := nTxven := nT1a30 := 0
nT31a60 := nT61a90 := nTmas90 := 0
mCliNum := cCliNum
mCliNom := cCliNom
cTitulo := &quot;*** Antiguedad de saldos de clientes&quot;
Pagina := 1
SELECT cli
mCliNum := cli-&gt;clinum
mCliNom := TRIM(cli-&gt;clinom)
cCliAte := TRIM(cli-&gt;cliate)
cDirec  := TRIM(cli-&gt;direc)
cColon  := TRIM(cli-&gt;colon)
cCiudad := TRIM(cli-&gt;ciudad)
cEstado := TRIM(cli-&gt;estado)
cCodPos := cli-&gt;codpos

oPrn:= TDosPrn():New()

IF ! oPrn:lCancel
oPrn:StartPage()
Enc_Rep(mCliNum, mCliNom, cCliAte, cDirec, cColon, ;
cCiudad, cEstado, cCodPos)

    SELECT cli                     
   DO WHILE cli-&gt;(!EOF())
   // Inicializa variables de clientes
   mCliNum := cli-&gt;clinum
        mCliNom := TRIM(cli-&gt;clinom)
        cCliAte := TRIM(cli-&gt;cliate)
        cDirec  := TRIM(cli-&gt;direc)
        cColon  := TRIM(cli-&gt;colon)
        cCiudad := TRIM(cli-&gt;ciudad)
        cEstado := TRIM(cli-&gt;estado)
        cCodPos := cli-&gt;codpos

    IF !cfac-&gt;(DBSEEK(cCliNum))
         cli-&gt;(DBSKIP())
      LOOP         
    ENDIF
     DO WHILE cfac-&gt;clinum = cCliNum
         IF cfac-&gt;tip_fac = &quot;M&quot;
             cfac-&gt;(DBSKIP())
              LOOP
        ENDIF
        IF cfac-&gt;estado = &quot;C&quot;
            cfac-&gt;(DBSKIP())
             LOOP
           ENDIF
            IF LEFT(cSel,1) == &quot;S&quot;
                IF cfac-&gt;impfac = cfac-&gt;pagototal
                    cfac-&gt;(DBSKIP())
                    LOOP
                ENDIF
            ENDIF

            // Inicializa variables de facturas
            cFacNum   := cfac-&gt;facnum
            dFechaFac := cfac-&gt;fechafac
            dVencFac  := cfac-&gt;vencfac
            nImpFac   := cfac-&gt;impfac

            // Imprime partidas
            oPrn:Say( nLinea, 004, cFacNum )
            oPrn:Say( nLinea, 012, DTOC(dFechaFac) )
            oPrn:Say( nLinea, 022, DTOC(dVencFac) )
            oPrn:Say( nLinea, 032, TRANSFORM(nImpFac,&quot;999,999,999.99&quot;) )
            SELECT pag
            IF !pag-&gt;(DBSEEK(cFacNum))
            nLinea := nLinea + 1
               IF nLinea &gt; 50
            salto(mCliNum, mCliNom, cCliAte, cDirec, cColon, ;
                            cCiudad, cEstado, cCodPos)
               ENDIF
            ENDIF            
            totpag = 0
            saldofac := 0
            DO WHILE pag-&gt;facnum = cFacNum
            oPrn:Say( nLinea, 048, TRANSFORM(pag-&gt;cantpag,&quot;999,999,999.99&quot;) )
               oPrn:Say( nLinea, 065, DTOC(pag-&gt;fechapag) )
            TotPag := TotPag + pag-&gt;cantpag
               nLinea := nLinea + 1
            IF nLinea &gt; 50
                salto(mCliNum, mCliNom, cCliAte, cDirec, cColon, ;
                            cCiudad, cEstado, cCodPos)
               ENDIF                  
            pag-&gt;(DBSKIP())
            ENDDO
            nTImp    := nTImp + nImpFac
            nTPago   := nTPago + totpag
            nTSaldo  := nTImp - nTPago
            saldofac := nImpFac - totpag
            DO CASE
            CASE dVencFac &gt;= DATE()
                nTxven := nTxven + saldofac
               CASE DATE() - dVencFac &lt;= 30
               nT1a30 := nT1a30 + saldofac
               CASE DATE() - dVencFac &lt;= 60
            nT31a60 := nT31a60 + saldofac                   
            CASE DATE() - dVencFac &lt;= 90
            nT61a90 := nT61a90 + saldofac
            OTHERWISE
            nTmas90 := nTmas90 + saldofac
            ENDCASE                      
          cfac-&gt;(DBSKIP())
     ENDDO

      // Imprime totales
        IF nTImp &lt;&gt; 0
        nLinea = nLinea + 1      
        oPrn:Say( nLinea, 032, REPLICATE(&quot;-&quot;,14) )
        oPrn:Say( nLinea, 048, REPLICATE(&quot;-&quot;,14) )
        nLinea = nLinea + 1
        oPrn:Say( nLinea, 032, TRANSFORM(nTImp,&quot;999,999,999.99&quot;) )
        oPrn:Say( nLinea, 048, TRANSFORM(nTPago,&quot;999,999,999.99&quot;) )
        oPrn:Say( 053, 009, &quot;NO VENCIDO                &quot; + ;
                     TRANSFORM(nTxven,&quot;999,999,999.99&quot;) )
        oPrn:Say( 054, 009, &quot; 1 A 30 DIAS DE RETRASO   &quot; + ;
                     TRANSFORM(nT1a30, &quot;999,999,999.99&quot;) )
        oPrn:Say( 055, 009, &quot;31 A 60 DIAS DE RETRASO   &quot; + ;
                     TRANSFORM(nT31a60,&quot;999,999,999.99&quot;) )
        oPrn:Say( 056, 009, &quot;61 A 90 DIAS DE RETRASO   &quot; + ;
                     TRANSFORM(nT61a90,&quot;999,999,999.99&quot;))
        oPrn:Say( 057, 009, &quot;MAS DE 90 DIAS DE RETRASO &quot; + ;
                     TRANSFORM(nTmas90,&quot;999,999,999.99&quot;) )
        oPrn:Say( 058, 035, REPLICATE(&quot;-&quot;,14) )
        oPrn:Say( 059, 009, &quot;T O T A L : -             &quot; + ;
                     TRANSFORM(nTSaldo,&quot;999,999,999.99&quot;) )
        DO CASE
        CASE nTSaldo &lt; 0
            oPrn:Say( 061, 009, &quot;SU SALDO A FAVOR PUEDE SER &quot; + ;
                         &quot;ACREDITADO A FUTURAS COMPRAS&quot; )
            CASE nTSaldo = 0
            oPrn:Say( 061, 009, &quot;GRACIAS POR SU PRONTO PAGO&quot; )
            CASE nTmas90 &gt; 0
                oPrn:Say( 061, 009, &quot;SU CUENTA ESTA SERIAMENTE ATRASADA; &quot; + ;
                             &quot;FAVOR DE PAGAR&quot; )
                oPrn:Say( 062, 009, &quot;LO MAS RAPIDAMENTE POSIBLE&quot; )
        CASE nT61a90 &gt; 0
            oPrn:Say( 061, 009, &quot;SU CUENTA PRESENTA UN GRAN ATRASO; &quot; + ;
                         &quot;FAVOR DE ENVIARNOS&quot;  )
                oPrn:Say( 062, 009, &quot;SU PAGO LO MAS PRONTO POSIBLE&quot; )
            CASE nT31a60 &gt; 0
            oPrn:Say( 061, 009, &quot;SU CUENTA PRESENTA UN ATRASO; &quot; + ;
                         &quot;FAVOR DE ENVIARNOS SU PAGO&quot; )
            CASE nT1a30 &gt; 0
                oPrn:Say( 061, 009, &quot;FAVOR DE PAGAR SU CUENTA ATRASADA&quot; )
            CASE nTxven &gt; 0
            oPrn:Say( 061, 009, &quot;LE AGRADECEREMOS ENVIE SU PAGO &quot; + ;
                         &quot;AL VENCIMIENTO&quot; )
           ENDCASE
        nTImp := nTPago := nTSaldo := nTxven := 0
        nT1a30 := nT31a60 := nT61a90 := nTmas90 := 0
        ENDIF


       cli-&gt;(DBSKIP())
     IF LEFT(cSel,1) == &quot;S&quot;
         EXIT
       ENDIF
    ENDDO
IF nLinea &gt;= 33
    EJECT
ENDIF

ENDIF
CLOSE DATABASES
oPrn:EndPage() // salta pagina
oPrn:End() 

//--- Passar a Dialog como ultimo parametro.
if oPrn:lPreview
    TxtPreview( oPrn:cDevice, cTitulo, lModal,,, oPrn, oDlg, , ;
  oPrn:lModoGraf )
endif

RETURN NIL

//---------------------------------------------------------//

STATIC FUNCTION OkCli( oGet2, oDlg )
IF EMPTY(cCliNum)
MsgAlert("Introduzca un número de cliente...")
RETURN .F.
ENDIF
cli->(DBSETORDER(1))
cli->(DBSEEK(cCliNum))
cCliNom := cli->clinom
oGet2:varput(TRIM(cCliNom))
oGet2:refresh()
RETURN .T.

//---------------------------------------------------------//

STATIC FUNCTION Salto(mCliNum, mCliNom, cCliAte, cDirec, cColon, ;
cCiudad, cEstado, cCodPos)
Pagina := Pagina + 1
Enc_Rep(mCliNum, mCliNom, cCliAte, cDirec, cColon, cCiudad, ;
cEstado, cCodPos)
RETURN NIL

//---------------------------------------------------------//

// FIN DE EMIEDO.PRG
