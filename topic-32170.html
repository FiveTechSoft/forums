FUNCTION Main(cDatos)
///////////////////////////////////////

local cNew:='',x,lOk:=.T.,curl
local cToken:=&quot;;&quot;,cProd,cProd2,cprod3

   RETU .T.
   QUIT

 oHttp:= CreateObject( 'MSXML2.XMLHTTP')
CATCH
 MsgInfo(&quot;Error al crear el objeto HTTP&quot;)
 retu .t.    
END

x:=1

WHILE .T.

    cNew:=substr( cDatos,1,at('&lt;END&gt;',cDatos)+4 )

    cProd       := strtoken(cNew,1,cToken)
    cProd2      := strtoken(cNew,2,cToken)
    cProd3      := strtoken(cNew,3,cToken)

    Aadd(aProd,cProd)
    Aadd(aProd,cProd2)
    Aadd(aProd,cProd3)

    cDatos:= substr(cDatos,at('&lt;END&gt;',cDatos)+5)

    if( at('&lt;END&gt;',cDatos) = 0)
      exit
    endif
 x+=1 
ENDDO
