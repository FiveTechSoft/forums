Intentas asi:

xFocus( oGet2 )

...
Code: Select all Collapse
//-------------------------------------------------------------------------//
FUNCTION xFocus( oObj )

    xSetFocus( oObj )
    xSetFocus( oObj )

RETURN( .T. )
//-------------------------------------------------------------------------//
FUNCTION xSetFocus( oObj )

    LOCAL _oWnd := oObj:oWnd, _oTempo := ""

    DEFINE Timer _oTempo Interval 10 of _oWnd ;
           Action ( oObj:SetFocus(), _oTempo:End() )
    ACTIVATE Timer _oTempo

RETURN( .T. )
//-------------------------------------------------------------------------//
