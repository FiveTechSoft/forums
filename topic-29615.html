*13:00~02:00( over 17:00 ) get 9 hours .....
*08:00~17:00(no over 17:00 ) get 0 hour

LOCAL difference

LOCAL nTime2 := Val( SubStr( cTime2, 1, 2 ) ) * 60 + ;
                Val( SubStr( cTime2, 3, 2 ) ) * 60

                Val( SubStr( cOutTime, 3, 2 ) ) * 60
