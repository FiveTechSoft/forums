Prueba esto:

Code (fw): Select all Collapse
#include "FiveWin.ch"

STATIC aRect := {0,0,0,0}
STATIC nFin := 0

FUNCTION Prueba()
   LOCAL oDlg,oBot,oTim
   *
   nFin := 0
   DEFINE GDIALOG oDlg  SIZE 400,200 COLOR 0,CLR_WHITE TITLE " Negación"
      DEFINE TIMER oTim INTERVAL 500 ACTION Negar(oDlg,oTim)
      @ 20,20 BUTTONBMP oBot PROMPT "Negación" SIZE 60,12 OF oDlg PIXEL ACTION oTim:Activate()
   ACTIVATE GDIALOG oDlg CENTER ON INIT aRect := GetWndRect(oDlg:hWnd)
   *
   oTim:End()
RETURN NIL
*
FUNCTION Negar(oDlg,oTim)
   oDlg:Hide()
   IF nFin = 6
      oDlg:Move(aRect[1],aRect[2])
      nFin := 0
      oTim:DeActivate()
      ShowWindow( oDlg:hWnd, 1 )
   ELSEIF nFin%2 = 0
      oDlg:Move(aRect[1],aRect[2]-50)
      oDlg:Show()
   ELSE
      oDlg:Move(aRect[1],aRect[2]+50)
      oDlg:Show()
   ENDIF
   ++nFin
RETURN NIL

No es perfecto, pero como comienzo no está mal
