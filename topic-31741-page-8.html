@set path=%oldpath%

Note the .bat file calls the nmake command

Here are some different implementations you may try:

  link @msvc.tmp /nologo /subsystem:windows /machine:x64 /force:multiple > link.log
   
.prg.c:
   $(HBDIR)\bin\harbour $< /n /w /p /d__64__ /i$(FWDIR)\include;$(HBDIR)\include > comp.log

.c.obj:
   cl.exe -c -TC -W3 -O2 -I$(HBDIR)\include -I$(VCDIR)\include $<
