This is a function I used with Clipper long time ago:

Code: Select all Collapse
#include "extend.h"


CLIPPER crc16()
{
    char *s = _parc( 1 );
    int crc = 0;
    int i, j;

    for ( i = 0; i < 128; i++ )
    {
        crc ^= *s++ << 8;

        for ( j = 0; j < 8; j++ )
            if ( crc & 0x8000 )
                crc = ( crc << 1 ) ^ 0x1021;
            else
                crc = crc << 1;
    }

    _retni( crc );
}
