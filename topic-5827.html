te adjunto un ejemplo y el resultado con FWPPC y FWH:

#include "FWCE.ch"

//----------------------------------------------------------------------------//

function Main()

local oWnd
LOCAL cText:= "123456789" + CRLF + "---------"
LOCAL oFont

DEFINE FONT oFont NAME "Courier New" SIZE 0, -12

DEFINE WINDOW oWnd TITLE "MemoEdit"

@ 2, 2 GET cText MULTILINE SIZE 200, 100 FONT oFont

@ 9, 5 BUTTON "Save" SIZE 80, 20 ACTION oWnd:End() // Save here...

@ 9, 21 BUTTON "Cancel" SIZE 80, 20 ACTION oWnd:End() // Cancel here

ACTIVATE WINDOW oWnd ;
ON CLICK MsgInfo( "Click!" )

return nil
