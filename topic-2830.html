This should be a working sample:

Code: Select all Collapse
#include "Fivewin.ch"


FUNCTION MAIN()

    ? GETTASKBARHEIGHT()

    RETURN NIL


#pragma BEGINDUMP

#include "windows.h"
#include "hbapi.h"

HB_FUNC( GETTASKBARHEIGHT )
{
    RECT rct;

    LONG nHDiff, nVDiff;

    SystemParametersInfo( SPI_GETWORKAREA, 0, &rct, 0 );

    nHDiff = GetSystemMetrics( SM_CXSCREEN ) - ( rct.right - rct.left );
    nVDiff = GetSystemMetrics( SM_CYSCREEN ) - ( rct.bottom - rct.top );

    if ( nHDiff > 0 )
        hb_retnl( nHDiff );
    else
        hb_retnl( nVDiff );
}

#pragma ENDDUMP
