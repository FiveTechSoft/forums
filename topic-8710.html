Francisco, asi lo uso y si problemas no hay salto de pagina

PRINT oPrn PREVIEW

 DEFINE FONT oFont   NAME "Times New Roman"  SIZE 0, -11 of oPrn
 DEFINE FONT oFont1  NAME "Times New Roman"  SIZE 0, -10 of oPrn

 oPrn:Setsize(190*10,186*10)

 SELECT A
 DO WHILE .NOT. EOF()
    INUM = A-&gt;NUM_DOC
    PAGE

    oPrn:CmSay( 2.2,13,+dtoc(A-&gt;FEC_DOC),oFont)
    oPrn:CmSay( 2.9,4,+ALLTRIM(A-&gt;Benefi),oFont)
    oPrn:CmSay( 2.9,15.5,+transform(A-&gt;VALOR,"99,999,999.99"),oFont)
    oPrn:CmSay( 3.8,4,+A-&gt;Detalle,oFont)
    oPrn:CmSay( 9.2,2,+A-&gt;DESC1,oFont1)
    oPrn:CmSay( 9.7,2,+A-&gt;DESC2,oFont1)
    SELECT b
    IF DbSeek(iNum)
       store 0 to gini,gdeb,ghab,gfin
       mLinea = 11.7
       DO WHILE b-&gt;Num_Doc = iNum
          oPrn:CmSay(mLinea,2,+OemToAnsi(alltrim(b-&gt;NOMBRE)),oFont1)
          oPrn:CmSay(mLinea,11,+B-&gt;Codigo,oFont1)
          oPrn:CmSay(mLinea,14,+Transform(b-&gt;DEBE,"@Z 999,999,999.99"),oFont1)
          oPrn:CmSay(mLinea,17,+Transform(b-&gt;HABER,"@Z 999,999,999.99"),oFont1)

          gdeb = gdeb + b-&gt;debe
          ghab = ghab + b-&gt;haber
          skip
          mLinea = mLinea + 0.5
       ENDDO
    ENDIF
    mLinea = mLinea + 1
    oPrn:CmSay(mLinea,2,"Totales  ",oFont)
    oPrn:CmSay(mLinea,14,+transform(gdeb,"999,999,999.99"),oFont1)
    oPrn:CmSay(mLinea,17,+transform(ghab,"999,999,999.99"),oFont1)
    SELECT A
    ENDPAGE
    SKIP
 ENDDO
 ENDPRINT

 oFont:End()
 oFont1:End()

 CLOSE MOVCK
 CLOSE DETCK

RETURN NIL

esta es una rutina desde donde imprimo cheques en forma continua y el tamaño es especial.
