Saludos esta forma generto mi QR para factura electronica
Code (fw): Select all Collapse
local cQrCode ,cFileQr := "QRCODE.bmp"

cQrCode := op:empresaruc+"|"+cQry:tipodoc+"|"+cQry:serie+"|"+Alltrim(cQry:numero)+"|"+Alltrim(Str(xIgv,12,4))+"|"+Alltrim(Str(xTotal,12,4))+"|"+;
                    Transform( Dtos( cQry:emision), "@R 9999-99-99" )+"|"+cQry:tiporuc+"|"+Alltrim(cQry:ruc)+"|"+Alltrim( cQry:hash_cpe )

QRCode( cQrCode, cFileQr )
