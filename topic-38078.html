function Borrando(aArray,oBrw)
local n
oBrw:nArrayAt := 2
For n := 2 to oBrw:nLen
   ADel( aArray, oBrw:nArrayAt, .f. )
   ASize(aArray,1)
   oBrw:Refresh()
Next
Return nil
