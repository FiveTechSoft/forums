Here is an updated BorlandConsole.Env for 5.82. Hope this Helps.

Rick Lipkin
Code (fw): Select all Collapse
[Environment]
Description 1=use the /n switch for new console apps .. for old
Description 2=dos apps do not use the /n switch
Description 3=
PRG Compiler ver=
C compiler ver=
Subsystem=
Author=
Last update=02/04/2015

[Advanced]
Command types=1,1,1,1,1,1,1,1,1

[Harbour]
HB_INSTALL=C:\XHARBOUR
HB_COMMAND=%HB_BIN_INSTALL%\Harbour.Exe %PRG% /m /gc0 /es2 /i%HB_INC_INSTALL%
Option prefix=/

[C]
C_INSTALL=C:\Borland\bcc582
C_COMP_COMMAND=%C_BIN_INSTALL%\Bcc32.Exe -c -O2 -OS -d -5 -6 -a8 -DFM_STATISTICS_OFF %CRLF%-I%HB_INC_INSTALL%;%C_INC_INSTALL%  -n%OBJDIR% %C%%CRLF%
C_LIB_COMMAND=%C_BIN_INSTALL%\TLib.Exe %LIB% /P512 @%RSP% , %LST%
C_LINK_COMMAND=%C_BIN_INSTALL%\iLink32.Exe -ap -Tpe /I%DBGDIR% /L%HB_LIB_INSTALL% %CRLF%@%LNK%
C_RC_COMMAND=%C_BIN_INSTALL%\Brc32.Exe -r  -fo%RES%  %RC%
C_DLL_COMMAND=
Option prefix=-

[User]
POSTEXE_COMMAND=
POSTLIB_COMMAND=
POSTDLL_COMMAND=

[Files]
%HB_LIB_INSTALL%\lang.lib=1
%HB_LIB_INSTALL%\vm.lib=2
%HB_LIB_INSTALL%\rtl.lib=3
%HB_LIB_INSTALL%\rdd.lib=4
%HB_LIB_INSTALL%\macro.lib=5
%HB_LIB_INSTALL%\pp.lib=6
%HB_LIB_INSTALL%\dbfntx.lib=7
%HB_LIB_INSTALL%\dbfcdx.lib=8
%HB_LIB_INSTALL%\dbffpt.lib=9
%HB_LIB_INSTALL%\common.lib=10
%HB_LIB_INSTALL%\gtwin.lib=11
%HB_LIB_INSTALL%\codepage.lib=12
%HB_LIB_INSTALL%\ct.lib=13
%HB_LIB_INSTALL%\tip.lib=14
%HB_LIB_INSTALL%\pcrepos.lib=15
%HB_LIB_INSTALL%\hsx.lib=16
%HB_LIB_INSTALL%\hbsix.lib=17
%HB_LIB_INSTALL%\zlib.lib=18
%C_LIB_INSTALL%\c0x32.obj=19
%C_LIB_INSTALL%\IMPORT32.LIB=20
%C_LIB_INSTALL%\CW32.LIB=21
