Function DRestore()
local cFile, lEsc, nHand, cComando, hCambios, cText, oBrush, oDls, oSay, cFileBackup, cNDBase, cUser, cPass

cUser       := "root"
cPass       := "qazwsxedc"

cFile := "restore.bat"
lEsc := .f.

IF !FFERASE( cFile )
  Return(.f.)
EndIf

cComando := ""

IF !FILE( cFile )

 nHand := FCreate( cFile )
 FWrite(nHand, cComando )
 FClose(nHand)

// ojo, ruta completa.....
 cComando :=  cNDBase + " < " + cFileBackUp + " -u " + cUser + " -p" + cPass
