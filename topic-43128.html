</i>#include <fivewin.ch>

#DEFINE xlPortrait 1
#DEFINE xlLandscape 2

STATIC oClip
STATIC cFileXls
STATIC cText
STATIC cHoja1
STATIC nSize
STATIC nPasteRow
STATIC nFormat

FUNCTION Main()
   WITH OBJECT XlsFileToPdf()
      :New()
   END OBJECT
RETURN (NIL)

CLASS XlsFileToPdf
   data oExcel, oSheet, oBook, nRow
   METHOD New()
   METHOD InitObjectXls()
   METHOD SendToExcel()
   METHOD InitObjectXls()
   METHOD BuildData()
   METHOD EndObjectXls()
   METHOD LoadLogo()
   METHOD Headers1()
   METHOD Headers2()
   METHOD SaveToPdf()
ENDCLASS

METHOD New() CLASS XlsFileToPdf
   if !File( ".\tmp" )
      lMkDir( ".\tmp" )
   endif
   MSGRUN( "send to xls and PDF... ", "User", {|| ::SendToExcel() } )
RETURN (NIL)

METHOD SendToExcel() CLASS XlsFileToPdf

   cFileXls := cTempFile(  "\"+CurDir()+'\tmp\'  , ".xlsx" )
   cText := ''
   nPasteRow := 0
   nFormat := 0

   ::InitObjectXls()

   nSize := 11

   ::LoadLogo()
   ::Headers1()
   ::Headers2()
   ::BuildData()

   ::EndObjectXls()

   ::SaveToPdf()


RETURN NIL

METHOD InitObjectXls() CLASS XlsFileToPdf
   LOCAL oRange

   cHoja1 := "DATA"

   cFileXls := cTempFile(  "\"+CurDir()+'\tmp\'  , ".xlsx" )

   ::nRow := 1
   ::oExcel := CREATEOBJECT( "Excel.Application" )

   ::oBook  := ::WorkBooks:Add()
   ::oSheet := ::Worksheets(1)
   nFormat     :=   ::Get("FileFormat")

   ::name := cHoja1

   ::Sheets( cHoja1  ):Select()

   ::oSheet := ::Get( "ActiveSheet" )

   ::PageSetup:PaperSize   := 1 //carta
   ::PageSetup:Orientation := xlPortrait
   ::PageSetup:Zoom = 100

   ::PageSetup:LeftMargin    := 0.5
   ::PageSetup:RightMargin   := 0.75
   ::PageSetup:TopMargin     := 1.5
   ::PageSetup:BottomMargin  := 2.5
   ::PageSetup:HeaderMargin  := 0.5
   ::PageSetup:FooterMargin  := 0.5
   ::PageSetup:FitToPagesWide := 1
   ::PageSetup:FitToPagesTall := .F.

   oRange := ::Range("A1")

RETURN (NIL)

METHOD LoadLogo() CLASS XlsFileToPdf
   LOCAL cRange,oRange
   local cFile :=  "\"+CurDir()+'\fivetech.jpg'
   IF file(cFile)
      cRange :=  "A2:C2"
      oRange := ::Range( cRange )
      ::AddPicture(cFile,.F., .T., oRange:Left, oRange:top, 150, 70 )
   ENDIF

RETURN (NIL)

METHOD EndObjectXls() CLASS XlsFileToPdf
   LOCAL nFor
   LOCAL cRange, oRange
   ::Sheets( cHoja1  ):Select()
   ::oSheet := ::Get( "ActiveSheet" )

   ::Columns("A:A"):ColumnWidth := 4
   ::Columns("B:B"):ColumnWidth := 10
   ::Columns("C:C"):ColumnWidth := 10
   ::Columns("D:D"):ColumnWidth := 10

RETURN NIL

METHOD SaveToPdf() CLASS XlsFileToPdf
   LOCAL cFilePdf := cTempFile(  "\"+CurDir()+'\tmp\'  , ".PDF" )
   ::oExcel := CREATEOBJECT( "Excel.Application" )
   ::WorkBooks:Open( cFileXls )
   ::Sheets( cHoja1  ):Select()
   ::oSheet := ::Get( "ActiveSheet" )
   ::PageSetup:PaperSize   := 1 //::PageSetup:Set( "PaperSize" , nPage )
   ::PageSetup:Orientation := 2 //::PageSetup:Set("Orientation",2 )
   ::ExportAsFixedFormat(0, cFilePdf, 0, .t., .f. )
   ::Quit()
   MSGRUN( "Exportando a PDF, por favor espere un momento: ", "Usuario", {|| Shellexecute( 0,"open", cFilePDF )  } )
RETURN (NIL)

METHOD Headers2() CLASS XlsFileToPdf
   LOCAL nRowIni := ::nRow
   LOCAL nCol
   LOCAL cRange, oRange

   FOR nCol := 2 TO 10
      ::Cells( ::nRow, nCol ):Value = "COL " + alltrim(cstr(nCol))
   NEXT

   cRange :=  "A" + alltrim(CSTR(nRowIni )) + ":" + "U"+alltrim(CSTR( ::nRow ))
   oRange := ::Range( cRange )
   oRange:Name := "Arial"
   oRange:Bold     := .t.
   oRange:Size     := nSize
   oRange:HorizontalAlignment := -4108
   ::nRow++

RETURN (nil)

METHOD BuildData() CLASS XlsFileToPdf
   LOCAL nRowIni := ::nRow
   LOCAL nCol,i
   LOCAL cRange, oRange

   nPasteRow := ::nRow
   oClip := TClipBoard():New()
   cText := ""


   FOR i := 1 TO 10
      FOR nCol := 1 TO 10
         IF nCol=1
            cText +=   cstr(i)
         else
            cText +=  CHR(9) + "DATA "+alltrim(cstr(nCol))
         ENDIF
      NEXT
      cText += CRLF
      ::nRow ++

   NEXT

  if ! Empty( cText )
      oClip:SetText( cText )
      ::Cells( nPasteRow, 1 ):Select()
      ::Paste()
      oClip:Clear()
      cText    := ""
   endif


RETURN (nil)

Code (fw): Select all Collapse

</i>// \SAMPLES\XLSATPDF.PRG

#include "FiveWin.ch"

STATIC INCLUDEDOCPROPERTIES := .F.
STATIC IGNOREPRINTAREAS     := .F.

FUNCTION Main()

   LOCAL vArchivoX, cArchivoPdf, oExcel, oSheet, Type, Quality, FileName

   // Donde tomar el .XLXS
   vArchivoX := "C:\FWH1905\SAMPLES\MATERIAL.xlsx" // nome completo do caminho com a unidade tambem

   IF .NOT. FILE( vArchivoX )

      MsgInfo( "Archivo de Origen no ejiste", "ERROR!" )

      RETURN NIL

   ENDIF

   // Donde Grabar: destino.
   cArchivoPdf := 'C:\TMP\ARCHIVO.pdf' // nome completo do caminho com a unidade tambem

   TRY
      oExcel = GetActiveObject( "Excel.Application" )
   CATCH
      TRY
         oExcel = CreateObject( "Excel.Application" )
      CATCH
         MsgInfo( "Excel is not installed on this PC. You need Excel to continue", "ERROR" )
      END
   END

   oExcel:WorkBooks:Open( vArchivoX )

   oExcel:Visible := .F.  // se .t. vai abrir o arquivo na tela

   oSheet = oExcel:ActiveSheet

   oExcel:WorkBooks( 1 ):Save()  // salva no excell
   oExcel:WorkBooks( 1 ):ExportAsFixedFormat( Type := 0, FileName := cArchivoPdf, Quality := 1, IncludeDocProperties := .T., IgnorePrintAreas := .T., 1, 1 ) // exporta em pdf

   oExcel:Quit()

   oExcel := NIL

   /*
   MsgRun( "AGUARDE... ABRINDO O PROGRAMA DO ADOBE...",                      ;
           "Por Favor, Espere Abertura. Pode Demorar.",                      ;
           { || WinExec( ShellExecute( 0, "Open", cArchivoPdf ) ), 5 } )
   */

   MsgRun( "Exportando a PDF, por favor espere un momento: ",                ;
           "Usuario", {|| Shellexecute( 0,"open", cArchivoPdf ), 3.5  } )

RETURN NIL
