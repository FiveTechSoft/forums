Librería QR: Asegúrate de tener QRCodelib.Dll o usar hbzebra.lib en versiones modernas de Harbour/FWH para la función QRBitmap.



Code: Select all Collapse
</s><i>
</i>#include "FiveWin.ch"

FUNCTION GenerarQR( cText, cFile )
   // cText: Datos de la factura (ej: "ID:123;Total:100;...")
   // cFile: Nombre de archivo de salida, ej: "factura.bmp"
   

   // Esta función genera la imagen del código QR
   QRBitmap( cText, cFile ) 
RETURN NIL
<e>


Code: Select all Collapse
</s><i>
</i>METHOD PrintFactura( oSender ) CLASS TFacturacion
   LOCAL oPrn, oFont, oPen
   LOCAL cQRFile := "tempqr.bmp"
   

   // 1. Generar la imagen QR
   GenerarQR( "Factura Nro 100 - Total: $500", cQRFile )

   // 2. Iniciar impresión
   PRINT oPrn PREVIEW
      DEFINE FONT oFont NAME "Arial" SIZE 0, -10
      PAGE
         // ... Otros datos de la factura ...
         

<i>    </i>     // 3. Imprimir el código QR en la posición deseada
<i>    </i>     // oPrn:SayBitmap( nRow, nCol, cFile, nWidth, nHeight )
<i>    </i>     oPrn:SayBitmap( 10, 10, cQRFile, 150, 150 ) // Ajustar tamaño según necesidad
<i>    </i>     
<i>    </i>  ENDPAGE
   ENDPRINT
   

   // 4. Borrar archivo temporal
   FErase( cQRFile )
RETURN NIL
<e>

</i> #include "fivewin.ch"

Function TesteQRCode()

<i>    </i>Local cStr := "Conteudo do seu QR Code Aqui (PIX, URL, etc)"
<i>    </i>Local cFile := "qrcode.bmp"
<i>    </i>Local oDlg, oBmp

<i>    </i>// Chama a função da DLL para gerar o arquivo BMP
<i>    </i>QRCode(cStr, cFile)

<i>    </i>
<i>    </i>// Exibe o BMP gerado
<i>    </i>@ 10,10 BITMAP oBmp FILE cFile OF oDlg Size 200,200 Pixel NOBORDER
<i>    </i>
<i>    </i>ACTIVATE DIALOG oDlg CENTERED
<i>    </i>
Return Nil

// Declaração da DLL

Ejemplo:



Code: Select all Collapse
</s><i>
</i>#include "FiveWin.ch"

function Main()
   local oDlg, oBmp, oFont
   local cCode := "https://www.fivetechsoft.com" // Texto do QR Code
   local cBmpFile := "qrcode.bmp" // Arquivo temporário

   // --- AQUII: Chame sua função geradora de QR Code para criar cBmpFile ---
   // Ex: HBQRCODE( cCode, cBmpFile )

   DEFINE Font oFont NAME "Verdana" SIZE 0, -14
   DEFINE DIALOG oDlg TITLE "Gerador de QR Code" SIZE 350, 400

   // Exibir o bitmap no diálogo
   @ 10, 10 IMAGE oBmp FILE cBmpFile OF oDlg SIZE 150, 150 PIXEL NOBORDER
   oBmp:lTransparent := .t.

   @ 170, 10 SAY "Código:" OF oDlg PIXEL FONT oFont
   @ 190, 10 GET cCode OF oDlg PIXEL FONT oFont

   ACTIVATE DIALOG oDlg CENTERED

return nil

Nuevas versiones de FWH, 25.05 por ejemplo:



Code: Select all Collapse
</s><i>
</i>#include "FiveWin.ch"

REQUEST FWZEBRA // Indispensável para desenhar QR Code

Function Main()

<i>    </i>local oDlg, hBmp, oImg
<i>    </i>local cTxt4QR := "https://www.fivetechsoft.com"

<i>    </i>// Gera o QR Code em um Handle de BitMap
<i>    </i>hBmp := FW_BarCodeBmp( cTxt4QR, "QRCODE", 250, 250 )                

<i>    </i>DEFINE DIALOG oDlg SIZE 300,300 PIXEL TRUEPIXEL TITLE "QR Code Example"
<i>    </i>
<i>    </i>@ 20,20 XIMAGE oImg SIZE -20,-20 OF oDlg PIXEL RESOURCE hBmp
<i>    </i>
<i>    </i>ACTIVATE DIALOG oDlg CENTERED
<i>    </i>
<i>    </i>DeleteObject( hBmp ) // Limpar memória
return nil
