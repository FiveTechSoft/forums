copiar "hbcurls.a" a fwh64_lib
copiar "libcurl.a" a fwh64_lib
copiar "hbcurl.ch" a fwh64_include
copiar "libcurl-x64.dll" en la carpeta del programa
copiar "cacert.pem" en la carpeta del programa (certificados)
