Joaquín,

Con este pequeño cambio en la Clase TFolderEx se soluciona:

Code (fw): Select all Collapse
METHOD Paint() CLASS TFolderEx

   LOCAL aInfo

   IF ::lWorking
      RETURN NIL
   ENDIF

   aInfo   := ::DispBegin()

   IF ::nLayOut == LAYOUT_TOP .OR. ::nLayOut == LAYOUT_BOTTOM
      ::PaintTB()
   ELSE
      ::PaintLR()
   ENDIF

   ::DispEnd( aInfo )

   ::aDialogs[ ::nOption ]:Refresh()   // Nuevo !!!

RETURN NIL
