For insert row

Code (fw): Select all Collapse
// select sheet
oSheet   := oExcel:Sheets( 1 )
oSheet:Rows(1).Insert()
oSheet:Columns("D:D"):Insert()

// or with objects range
objRange := oExcel.Range("A1").EntireRow()
objRange:Insert()

objRange := oExcel:Range("C1"):EntireColumn()
objRange:Insert()
