Yo lo hago de la siguiente forma, aqui veduelvo la fecha de creacion o odificacion del exe principal, espero te sirva..

FUNCTION VERSIONEXE()

LOCAL dVersion := DTOS(DIRECTORY(".\TUEXE.EXE")[1,3])
LOCAL cVer:= "Edición "+SUBSTR(ALLTRIM(dVersion),7,2)+"."+SUBSTR(ALLTRIM(dVersion),5,2)+"."+SUBSTR(ALLTRIM(dVersion),1,4)

RETURN cVer
