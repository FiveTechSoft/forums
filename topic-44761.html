</s>STATIC FUNCTION DOPPIOCLICK( oGet )

    LOCAL cVal := oGet:GetText()

    LOCAL nSta := 0

    LOCAL i
    cVal := ALLTRIM( cVal )

    FOR i = 1 TO LEN( cVal )
        IF !EMPTY( SUBSTR( cVal, i, 1 ) )
            nSta = i - 1
            EXIT
        ENDIF
    NEXT

    oGet:PostMsg( EM_SETSEL, nSta, nSta + LEN( ALLTRIM( cVal ) ) )
    RETURN NIL

      ACTION( MsgInfo( cFilter ), oDlg:End() ) OF oDlg

RETURN NIL
