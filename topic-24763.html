Estimado colega yo lo uso así y funciona correctamente.

Function Main()
              Local oDlg
              Local oFont1
              Local oSay
              Define FONT oFont1 NAME GetSysFont()  // Fuente del Sistema
              DEFINE DIALOG oDlg TITLE &quot;Prueba &quot;
                       @  nRow, nCol SAY oSay VAR &quot; cualquier cosa&quot; SIZE 100,11 PIXEL OF oDlg FONT oFont1 
               ACTIVATE DIALOG oDlg CENTER
               oFont1:End()
Return( Nil )
