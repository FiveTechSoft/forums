Con gusto Antonio este es el tramo del codigo, y de antemano muy agradecido.
//-----------------------------------------------------------//

DEFINE DIALOG oDlg RESOURCE "Browse" TITLE "Busqueda por Operaciones"
REDEFINE SAY oSay PROMPT Clientes->Placa ID 88 OF oDlg UPDATE

SEEK( Clientes->Placa)
REDEFINE LISTBOX oBrw FIELDS Clientes->Placa, operaciones->FechaSal,;
operaciones->descrip,operaciones->Cantidad,Operaciones->Carga;
HEADERS "Placa","Facha","Descripcion","Cantidad","Refrigerante";
ID 195 OF oDlg ALIAS "operaciones";
select operaciones->Placa for Clientes->placa UPDATE
// oBrw : autoedit = .T.
activate dialog oDlg CENTERED

//--------------------------------------------------------------------------//

RETURN found()

Note la forma que coloque tu sugerencia, aunque como puedes ver esta inhabilitado.
