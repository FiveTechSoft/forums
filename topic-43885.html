Function Anteprima(oBrowse,oRec)

local oWord := WinWordObj()
local oDoc := oWord:Documents:Add()
oDoc:PageSetup:Orientation = 0
       
             IF oRec:Rows = 0 .AND. oRec:Columns= 0
                    PrnList(oBrowse,oRec)

           else
                  PrnLabel(oBrowse,oRec)
           Endif

Return nil
