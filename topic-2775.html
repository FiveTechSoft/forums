Para los que le interese:

Code: Select all Collapse
rddRegister( "ADS", 1 )
RddSetDefault("ADS")
SET FILETYPE TO ADT
SET SERVER REMOTE
AdsUseDictionary( .T. )
AdsRightsCheck( .F. )

IF !AdsConnect60( "\\server\sistema\dicc.add", 7, "adssys", "pass" )
   MsgInfo( "Error.." )
   Quit
endif

AdsUseDictionary( .T. )
AdsRightsCheck( .F. )
