Paulo,

Añade el texto de esta forma:
Code: Select all Collapse
#include "FiveWin.ch"

function Main()

   local cText := MemoRead( "texto.txt" ), n

   DBCREATE( "NOTICW.DBF", {{ "LINHA","C",255,0 }} )

   USE NOTICW.DBF

   for n = 1 to MLCount( cText )
      APPEND BLANK
      field->Linha := MemoLine( cText,, n )
   next   

   GO TOP
   Browse()

return nil
