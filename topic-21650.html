Yo uso este código fuente, "wrapeando" la función GetLocalTime de la DLL "kernel32.dll".

Code (fw): Select all Collapse
#include "Fivewin.ch"
#include "struct.ch"
#include "exstruc.ch"

function main()
local cTemp:=""

STRUCT HoraSys
  MEMBER wAnno   AS _INT
  MEMBER wMes    AS _INT
  MEMBER wDiaSem AS _INT
  MEMBER wDia    AS _INT
  MEMBER wHora   AS _INT
  MEMBER wMinuto AS _INT
  MEMBER wSegun  AS _INT
  MEMBER wMilis  AS _INT
ENDSTRUCT

cTemp:=HoraSys:cBuffer

GetLocalTime(@cTemp) // Pide datos de fecha y hora.

HoraSys:cBuffer:=cTemp

msginfo("Año:"             +str(HoraSys:wAnno)  +chr(13)+;
        "Mes:"             +str(HoraSys:wMes)   +chr(13)+;
        "Dia de la semana:"+str(HoraSys:wDiaSem)+chr(13)+;
        "Dia:"             +str(HoraSys:wDia)   +chr(13)+;
        "Hora:"            +str(HoraSys:wHora)  +chr(13)+;
        "Minutos:"         +str(HoraSys:wMinuto)+chr(13)+;
        "Segundos:"        +str(HoraSys:wSegun) +chr(13)+;
        "Milisegundos:"    +str(HoraSys:wMilis))

return nil

DLL32 FUNCTION GetLocalTime(lpSystemTime AS LPSTR) AS LONG PASCAL FROM "GetLocalTime" LIB "kernel32.dll"
