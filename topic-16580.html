Code (fw): Select all Collapse
FUNCTION OPENCAJA( cPort , lSerial )
LOCAL o
Local cCajaCommand := chr(027) + chr( 112 ) + chr( 000) + chr(100) + chr( 250 )
Default lSERIAL := .F.
DEFAULT cPort   := "LPT1"

 o := TDosPrn():New( 3, cPort )
 o:SAY( 00, 00, IF( lSerial, Replicate("@", 32), cCajaComand ) )
 o:cFormFeed := ""
 o:End()

RETURN(.T.)
