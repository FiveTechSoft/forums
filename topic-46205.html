<i>    </i><script>
<i>    </i>    const chatMessages = document.getElementById('chatMessages');
<i>    </i>    const userInput = document.getElementById('userInput');
<i>    </i>    const sendBtn = document.getElementById('sendBtn');

<i>    </i>    function addMessage(text, sender) {
<i>    </i>        const div = document.createElement('div');
<i>    </i>        div.classList.add('message', sender);
<i>    </i>        
<i>    </i>        // Allow basic parsing or just text. For safety, typically textContent.
<i>    </i>        // But if the AI returns markdown/html, we might want to parse it.
<i>    </i>        // For now, let's stick to textContent to be safe from XSS unless we bring in a sanitizer/parser.
<i>    </i>        // Or simple interpretation if needed. The user example used `puter.print` which is simple.
<i>    </i>        
<i>    </i>        // Check if text is an object (error or complex response)
<i>    </i>        if (typeof text === 'object') {
<i>    </i>            text = JSON.stringify(text);
<i>    </i>        }
<i>    </i>        
<i>    </i>        div.textContent = text;
<i>    </i>        chatMessages.appendChild(div);
<i>    </i>        chatMessages.scrollTop = chatMessages.scrollHeight;
<i>    </i>    }

<i>    </i>    async function handleSend() {
<i>    </i>        const text = userInput.value.trim();
<i>    </i>        if (!text) return;

<i>    </i>        // Add user message via UI
<i>    </i>        addMessage(text, 'user');
<i>    </i>        userInput.value = '';
<i>    </i>        userInput.disabled = true;
<i>    </i>        sendBtn.disabled = true;

<i>    </i>        // Add loading indicator
<i>    </i>        const loadingDiv = document.createElement('div');
<i>    </i>        loadingDiv.classList.add('message', 'loading');
<i>    </i>        loadingDiv.textContent = 'Thinking...';
<i>    </i>        chatMessages.appendChild(loadingDiv);
<i>    </i>        chatMessages.scrollTop = chatMessages.scrollHeight;

<i>    </i>        try {
<i>    </i>            // Check for image generation command
<i>    </i>            if (text.toLowerCase().startsWith('/image ')) {
<i>    </i>                const prompt = text.substring(7).trim();
<i>    </i>                loadingDiv.textContent = 'Generating image...';
<i>    </i>                
<i>    </i>                // Call Puter.js for image generation
<i>    </i>                // Using testMode=true as per request/snippet to avoid using credits
<i>    </i>                const image = await puter.ai.txt2img(prompt, false);
<i>    </i>                
<i>    </i>                loadingDiv.remove();
<i>    </i>                
<i>    </i>                const imgDiv = document.createElement('div');
<i>    </i>                imgDiv.classList.add('message', 'ai');
<i>    </i>                imgDiv.style.maxWidth = '100%'; // Allow images to be larger
<i>    </i>                
<i>    </i>                // The snippet says it returns an image element.
<i>    </i>                // Let's make sure it fits our styling.
<i>    </i>                image.style.maxWidth = '100%';
<i>    </i>                image.style.borderRadius = '0.5rem';
<i>    </i>                
<i>    </i>                imgDiv.appendChild(image);
<i>    </i>                chatMessages.appendChild(imgDiv);
<i>    </i>                chatMessages.scrollTop = chatMessages.scrollHeight;

<i>    </i>            } else {
<i>    </i>                // Call Puter.js for chat
<i>    </i>                const response = await puter.ai.chat(text, { model: "gpt-5.2-pro", driver: "openrouter" });
<i>    </i>                
<i>    </i>                // Remove loading
<i>    </i>                loadingDiv.remove();

<i>    </i>                // Add AI response
<i>    </i>                let fpResponse = response;
<i>    </i>                if (response && typeof response === 'object') {
<i>    </i>                    if (response.message && response.message.content) {
<i>    </i>                        fpResponse = response.message.content;
<i>    </i>                    } else if (response.content) { 
<i>    </i>                      fpResponse = response.content; 
<i>    </i>                    } else {
<i>    </i>                        fpResponse = response.toString(); 
<i>    </i>                        if (fpResponse === '[object Object]') {
<i>    </i>                             fpResponse = JSON.stringify(response);
<i>    </i>                        }
<i>    </i>                    }
<i>    </i>                }
<i>    </i>                
<i>    </i>                addMessage(fpResponse, 'ai');
<i>    </i>            }

<i>    </i>        } catch (error) {
<i>    </i>            loadingDiv.remove();
<i>    </i>            addMessage("Error: " + error.message, 'ai');
<i>    </i>            console.error(error);
<i>    </i>        } finally {
<i>    </i>            userInput.disabled = false;
<i>    </i>            sendBtn.disabled = false;
<i>    </i>            userInput.focus();
<i>    </i>        }
<i>    </i>    }

<i>    </i>    sendBtn.addEventListener('click', handleSend);
<i>    </i>    userInput.addEventListener('keypress', (e) => {
<i>    </i>        if (e.key === 'Enter') handleSend();
<i>    </i>    });
<i>    </i>    
<i>    </i>    // Focus input on load
<i>    </i>    userInput.focus();
<i>    </i></script>
