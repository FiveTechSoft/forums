My Link-Files I use ( mabe someone else created a better one ) :
I don't know, if You are using BORLAND and xHarbour !
Using Borland, You have to check 2 Config-Files in Folder /Bin
BCC32.CFG and ILINK32.CFG

1. Batchfile GO.bat

if not exist obj md obj
c:\bcc582\bin\make -f File.rmk

2. File.rmk ( more than 1 PRG included )
( replace File-names and check Path for FWH, xHarbour and Borland )

#Borland make sample, (c) FiveTech Software 2005

HBDIR=c:\xharbour
BCDIR=c:\bcc582
FWDIR=c:\fwh

.path.OBJ = .\obj
.path.PRG = .\
.path.CH = $(FWDIR)\include;$(HBDIR)\include
.path.C = .\
.path.rc = .\

# List of Prg-Files ( PRG = uppercase )
# -------------------------------------------
PRG = \
Fontedit.PRG \
Project.PRG

# C = \
# two.C

# Exe-Filename
#-----------------
PROJECT : Fontedit.exe

# include Res-file
# ------------------
Fontedit.exe : $(PRG:.PRG=.OBJ) $(C:.C=.OBJ) Fontedit.res
echo off
echo $(BCDIR)\bin\c0w32.obj + > b32.bc

# OBJ-List
# -------------
echo obj\Fontedit.obj \
obj\Project.obj, + >> b32.bc

echo Fontedit.exe, + >> b32.bc
echo Fontedit.map, + >> b32.bc

# Fivewin Lib`s für xHARBOUR
# --------------------------------------
echo $(FWDIR)\lib\Fivehx.lib + >> b32.bc
echo $(FWDIR)\lib\FiveHC.lib + >> b32.bc

# xHARBOUR-Lib`s
# -----------------------
IF EXIST Fontedit.rc $(BCDIR)\bin\brc32 -r Fontedit

# XP-Manifest defined in RC-File
# -------------------------------------
# IF EXIST Fontedit.rc $(BCDIR)\bin\brc32 -r -d__FLAT__ Fontedit

echo $(HBDIR)\lib\rtl.lib + >> b32.bc
echo $(HBDIR)\lib\vm.lib + >> b32.bc
echo $(HBDIR)\lib\GTcgi.lib + >> b32.bc
echo $(HBDIR)\lib\GTgui.lib + >> b32.bc
echo $(HBDIR)\lib\GTpca.lib + >> b32.bc
echo $(HBDIR)\lib\GTstd.lib + >> b32.bc
echo $(HBDIR)\lib\GTwvg.lib + >> b32.bc
echo $(HBDIR)\lib\GTwvt.lib + >> b32.bc
echo $(HBDIR)\lib\GTwin.lib + >> b32.bc
echo $(HBDIR)\lib\gtgui.lib + >> b32.bc
echo $(HBDIR)\lib\lang.lib + >> b32.bc
echo $(HBDIR)\lib\macro.lib + >> b32.bc
echo $(HBDIR)\lib\rdd.lib + >> b32.bc
echo $(HBDIR)\lib\dbfntx.lib + >> b32.bc
echo $(HBDIR)\lib\dbfcdx.lib + >> b32.bc
echo $(HBDIR)\lib\dbffpt.lib + >> b32.bc
echo $(HBDIR)\lib\hbsix.lib + >> b32.bc
echo $(HBDIR)\lib\debug.lib + >> b32.bc
echo $(HBDIR)\lib\common.lib + >> b32.bc
echo $(HBDIR)\lib\pp.lib + >> b32.bc
echo $(HBDIR)\lib\codepage.lib + >> b32.bc
echo $(HBDIR)\lib\pcrepos.lib + >> b32.bc
echo $(HBDIR)\lib\ct.lib + >> b32.bc
echo $(HBDIR)\lib\hbzip.lib + >> b32.bc
echo $(HBDIR)\lib\zlib.lib + >> b32.bc

# Uncomment these two lines to use Advantage RDD
# echo $(HBDIR)\lib\rddads.lib + >> b32.bc
# echo $(HBDIR)\lib\Ace32.lib + >> b32.bc

echo $(BCDIR)\lib\cw32.lib + >> b32.bc
echo $(BCDIR)\lib\import32.lib + >> b32.bc
echo $(BCDIR)\lib\uuid.lib + >> b32.bc
echo $(BCDIR)\lib\psdk\odbc32.lib + >> b32.bc
echo $(BCDIR)\lib\psdk\rasapi32.lib + >> b32.bc
echo $(BCDIR)\lib\psdk\nddeapi.lib + >> b32.bc
echo $(BCDIR)\lib\psdk\msimg32.lib + >> b32.bc
echo $(BCDIR)\lib\psdk\iphlpapi.lib, >> b32.bc

IF EXIST Fontedit.res echo Fontedit.res >> b32.bc
$(BCDIR)\bin\ilink32 -Gn -aa -Tpe -s -v @b32.bc
# $(BCDIR)\bin\ilink32 -Gn -aa -Tpe -s @b32.bc
del b32.bc

.PRG.OBJ:
$(HBDIR)\bin\harbour $< /N /W /Oobj\ /I$(FWDIR)\include;$(HBDIR)\include > clip.log
$(BCDIR)\bin\bcc32 -c -tWM -I$(HBDIR)\include -oobj\$& obj\$&.c

.C.OBJ:
echo -c -tWM -D__HARBOUR__ -DHB_API_MACROS > tmp
echo -I$(HBDIR)\include;$(FWDIR)\include >> tmp
$(BCDIR)\bin\bcc32 -oobj\$& @tmp $&.c
del tmp

# Resource
# -----------
Fontedit.res : Fontedit.rc
$(BCDIR)\bin\brc32.exe -r Fontedit.rc
