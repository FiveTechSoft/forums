function fecha_cliente(oBrw,nGet,nLocal,cCli,cColuna)
  if numero_impar(cColuna)
     oBrw:aCols[cColuna+1]:nGrpBmpNo := 1
  else
     oBrw:aCols[cColuna]:nGrpBmpNo := 2
  endif
  oBrw:refresh()
return .t.
