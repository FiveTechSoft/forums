//-> Puedes continuar usando FOX BASE normalmente en WINDOWS XP, te basta
//-> Crear um ARCHIVO.BAT com el nombre FOX.BAT y copiar para C:\WINDOWS

REM FOX.BAT

@ECHO OFF
MODE 80,25
MODE CON LINES=25
SET ALLUSERSPROFILE=
SET APPDATA=
SET CLIENTNAME=
SET CommonProgramFiles=
SET COMPUTERNAME=
SET ComSpec=
SET FP_NO_HOST_CHECK=
SET HOMEDRIVE=
SET HOMEPATH=
SET LOGONSERVER=
SET NUMBER_OF_PROCESSORS=
SET OS=
SET Path=
SET PATHEXT=
SET PROCESSOR_ARCHITECTURE=
SET PROCESSOR_IDENTIFIER=
SET PROCESSOR_LEVEL=
SET PROCESSOR_REVISION=
SET ProgramFiles=
SET PROMPT=
SET SESSIONNAME=
SET SystemDrive=
SET SystemRoot=
SET TEMP=
SET TMP=
SET USERDOMAIN=
SET USERNAME=
SET USERPROFILE=
SET windir=
\FOXBASE\FOX.EXE
SET ALLUSERSPROFILE=C:\Documents and Settings\All Users
SET APPDATA=C:\Documents and Settings\micro\Dados de aplicativos
SET CLIENTNAME=Console
SET CommonProgramFiles=C:\Arquivos de programas\Arquivos comuns
SET COMPUTERNAME=ESCRITORIO
SET BComSpec=C:\WINDOWS\system32\cmd.exe
SET FP_NO_HOST_CHECK=NO
SET HOMEDRIVE=C:
SET HOMEPATH=\Documents and Settings\micro
SET LOGONSERVER=\ESCRITORIO
SET NUMBER_OF_PROCESSORS=1
SET OS=Windows_NT
SET Path=C:\WINDOWS\system32;C:\WINDOWS;C:\WINDOWS\System32\Wbem
SET PATHEXT=.COM;.EXE;.BAT;.CMD;.VBS;.VBE;.JS;.JSE;.WSF;.WSH
SET PROCESSOR_ARCHITECTURE=x86
SET PROCESSOR_IDENTIFIER=x86 Family 15 Model 2 Stepping 9, GenuineIntel
SET PROCESSOR_LEVEL=15
SET PROCESSOR_REVISION=0209
SET ProgramFiles=C:\Arquivos de programas
SET PROMPT=$P$G
SET SESSIONNAME=Console
SET SystemDrive=C:
SET SystemRoot=C:\WINDOWS
SET TEMP=C:\DOCUME~1\micro\CONFIG~1\Temp
SET TMP=C:\DOCUME~1\micro\CONFIG~1\Temp
SET USERDOMAIN=ESCRITORIO
SET USERNAME=micro
SET USERPROFILE=C:\Documents and Settings\micro
SET windir=C:\WINDOWS
