DEFINE TIMER otimer ;
INTERVAL 14000 ;
ACTION Msg() ;
OF oWnd

FUNCTION Msg(lok)
local odb1,odlg
private obrow
DEFAULT lok:=.f.

odb1:=oServer:Query(&quot;select receiver from message where receiver='&quot;+username1+&quot;' and lread&lt;&gt;'Y'&quot;;
                            +&quot; union all select receiver from message where receiver='' and lread&lt;&gt;'Y'&quot;;
                            +&quot;&quot;)
if odb1:LASTREC()&gt;0 .or. lok
    otimer:DeActivate()
    if odb1:LASTREC()&gt;0
        msginfo2(&quot;您有&quot;+cvaltochar(odb1:LASTREC())+&quot;条新的短消息！&quot;,&quot;提示&quot;)
    endif
    odb1:end()
    odb1:=oServer:Query(&quot;select * from message where receiver='&quot;+username1+&quot;' &quot;;
                             +&quot; union all select * from message where receiver=''&quot;;
                             +&quot; order by dateline DESC&quot;)

    DEFINE DIALOG odlg RESOURCE &quot;msg1&quot; TITLE &quot;即时通信系统&quot; FONT ofont14
    obrow := TXBrowse():New( odlg )
    obrow:SetMySQL(@odb1)
    obrow:nMarqueeStyle:=4
    obrow:bSkip:=obrow:bSkipper

    oCol := obrow:AddCol()
    oCol:bStrData := {||odb1:sender}
    oCol:cHeader := &quot;发送人&quot;

    oCol := obrow:AddCol()
    oCol:bStrData := {||odb1:dateline}
    oCol:cHeader := &quot;发送时间&quot;

    oCol := obrow:AddCol()
    oCol:bStrData := {||odb1:lread}
    oCol:cHeader := &quot;已读&quot;

    obrow:CreateFromResource(201)

    REDEFINE SBUTTON ID 101 OF odlg PROMPT &quot;创建&quot;  xp ACTION NewMsg()
    REDEFINE SBUTTON ID 102 OF odlg PROMPT &quot;查看&quot;  xp ACTION ViewMsg(@odb1)
    REDEFINE SBUTTON ID 103 OF odlg PROMPT &quot;删除&quot;  xp ACTION (oserver:query(&quot;delete from message where messageid=&quot;+cvaltochar(odb1:messageid)+if(username1==&quot;SUPER   &quot;,&quot;&quot;,&quot; and receiver='&quot;+username1+&quot;'&quot;)),odb1:refresh(),obrow:refresh(.t.))
    REDEFINE SBUTTON ID 104 OF odlg PROMPT &quot;刷新&quot;  xp ACTION (odb1:refresh(),obrow:refresh())
    REDEFINE SBUTTON ID 105 OF odlg PROMPT &quot;退出&quot;  xp ACTION odlg:end()

    ACTIVATE DIALOG odlg CENTER on  INIT odlg:autoresize()

    otimer:ACTIVATE()
endif
odb1:end()

return

FUNCTION NewMsg(receiver)
local odlg,content:=SPACE(200)
DEFAULT receiver:=SPACE(8)

DEFINE DIALOG odlg RESOURCE &quot;msg3&quot; TITLE &quot;新建短消息&quot; FONT ofont14

REDEFINE GET receiver ID 101 OF odlg PICTURE &quot;@!&quot;
REDEFINE GET content ID 201 OF odlg
REDEFINE SBUTTON ID 102 OF odlg PROMPT &quot;发送&quot; xp ACTION (oServer:Query(&quot;insert into message (receiver,sender,dateline,content) values ('&quot;+cvaltochar(receiver)+&quot;','&quot;+username1+&quot;',now(),'&quot;+TranMsg(content)+&quot;')&quot;),odlg:end())
REDEFINE SBUTTON ID 103 OF odlg PROMPT &quot;退出&quot; xp ACTION odlg:end()

    REDEFINE SAY PROMPT &quot;收件人&quot; ID 901 OF odlg

ACTIVATE DIALOG odlg CENTER on  INIT odlg:autoresize()

return

FUNCTION ViewMsg(odb1)
local odlg

oServer:Query(&quot;update message set lread='Y' where messageid=&quot;+cvaltochar(odb1:messageid))


DEFINE DIALOG odlg RESOURCE &quot;msg2&quot; TITLE &quot;查看短消息&quot; FONT ofont14
REDEFINE GET odb1:sender ID 101 OF odlg WHEN .f.
REDEFINE GET odb1:dateline ID 102 OF odlg WHEN .f.
REDEFINE GET odb1:content ID 201 OF odlg READONLY MEMO

REDEFINE SAY PROMPT &quot;发件人:&quot; ID 901 OF odlg
REDEFINE SAY PROMPT &quot;时间&quot; ID 902 OF odlg

REDEFINE SBUTTON ID 103 OF odlg PROMPT &quot;回复&quot; xp ACTION (RepMsg(odb1:sender),odlg:end())
REDEFINE SBUTTON ID 104 OF odlg PROMPT &quot;删除&quot; xp ACTION (oserver:query(&quot;delete from message where messageid=&quot;+cvaltochar(odb1:messageid)+if(username1==&quot;SUPER   &quot;,&quot;&quot;,&quot; and receiver='&quot;+username1+&quot;'&quot;)),odb1:refresh(),obrow:refresh(.t.),odlg:end())
REDEFINE SBUTTON ID 105 OF odlg PROMPT &quot;退出&quot; xp ACTION odlg:end()

ACTIVATE DIALOG odlg CENTER on INIT odlg:autoresize()

return

FUNCTION RepMsg(receiver)
local odlg,content:=space(200)

DEFINE DIALOG odlg RESOURCE &quot;msg3&quot; TITLE &quot;回复短消息&quot; FONT ofont14
REDEFINE GET receiver ID 101 OF odlg WHEN .f.
REDEFINE GET content ID 201 OF odlg

REDEFINE SAY PROMPT &quot;收件人&quot; ID 901 OF odlg

REDEFINE SBUTTON ID 102 OF odlg PROMPT &quot;发出&quot; xp ACTION (oServer:Query(&quot;insert into message (receiver,sender,dateline,content) values ('&quot;+cvaltochar(receiver)+&quot;','&quot;+username1+&quot;',now(),'&quot;+TranMsg(content)+&quot;')&quot;),odlg:end())
REDEFINE SBUTTON ID 103 OF odlg PROMPT &quot;退出&quot; xp ACTION odlg:end()

ACTIVATE DIALOG odlg CENTER on  INIT odlg:autoresize()

return
