Hola la solucion

Code (fw): Select all Collapse
function DecToHex( nValue )

   local nMod, cDec := ""

   while nValue >= 16
      nMod   = nValue % 16
      nValue = Int( nValue / 16)
      cDec   = HexVal( nMod ) + cDec
   end

   cDec = HexVal( nValue ) + cDec

return cDec


esta es la funcion corregida, puedes utilizar esta en lugar de la que viene en FW
