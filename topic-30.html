// Test of ASC function

FUNCTION MAIN()

LOCAL n, cStr := "", lf := CHR(13)+CHR(10)

FOR n := 0 TO 255
cStr += LSTR( ASC(CHR(n)) )+IF(n%10=0,lf," - ")
NEXT

MSGINFO( cStr )

RETURN NIL

FUNCTION LSTR( val )
RETURN LTRIM(STR( val ))
