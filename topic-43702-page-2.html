You can use LISDIR():
Code (fw): Select all Collapse
#include "Fivewin.ch"


FUNCTION MAIN()

    ? 2, LISDIR( "MYTEST" )

    RETURN NIL


INIT FUNCTION TEST()

    MAKEDIR( "MYTEST" )

    ? 1, LISDIR( "MYTEST" )

    RETURN NIL
