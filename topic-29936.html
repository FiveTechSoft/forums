Proc ferrorsys(_swE)

static func DefError(e,l)
Loca i, cMenssageInf, cDefVariable

cMessageInf := ErrorMessage(e)
cMessage := cMessageInf


if 'Error BASE/1003'$cMessageInf

cDefVariable:=substr(cMessageInf,at(':',cMessageInf)+2)

static func ErrorMessage(e)
loca cMessage

// start error message
cMessage := if( e:severity &gt; ES_WARNING, &quot;Error &quot;, &quot;Warning &quot; )

// add subsystem name if available
if ValType(e:subsystem) == &quot;C&quot;
    cMessage += e:subsystem()
else; cMessage += &quot;???&quot;; end

    // add subsystem's error code if available
    if ValType(e:subCode) == &quot;N&quot;
        cMessage += (&quot;/&quot; + NTRIM(e:subCode))
    else; cMessage += &quot;/???&quot;; end

        // add error description if available
        if ValType(e:description) == &quot;C&quot;
            cMessage += (&quot;  &quot; + e:description)
        end

        // add either filename or operation
        if !Empty(e:filename)
            cMessage += (&quot;: &quot; + e:filename)
        elseif !Empty(e:operation)
            cMessage += (&quot;: &quot; + e:operation)
