Function SaveImg( oImg )

   local oBmp  := GdiBmp():New()
   oBmp:hBmp   := GDIP_FROMHBITMAP( oImg:GetHBitmap(), oImg:aPalBmp[ 1 ], .T. )
   ? oBmp:Save( "ejemplo.png" )
   oBmp:End()

Return nil
