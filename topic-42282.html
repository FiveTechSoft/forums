Look

Code (fw): Select all Collapse
  TEXT INTO cHtml
     <div class="easyui-panel" title="Equipamento" style="width:100%;max-width:400px;padding:30px 60px;">
        <div style="margin-bottom:20px">
           <label for="nCodEquip" class="label-top">&cPromptSay:</label>
         
            <input id="nCodEquip" class="easyui-validatebox" data-options=",required:#nObri#">

        </div>
     </div>
  ENDTEXT

        cHtml := StrTran( cHtml, "#nObri#", if( ::nObrigatório == 1, "true", "false" ) )
