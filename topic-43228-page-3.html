STATIC FUNCTION DoSwitch( ddatacambio, oBtnGetExchange  )
LOCAL dDate := ddatacambio:GetDate() 
LOCAL cDate := DTOC(dDate)

   IF EMPTY(cDate)
      oBtnGetExchange:Disable()
   ELSE
      oBtnGetExchange:enable()
   ENDIF
   oBtnGetExchange:Refresh()
RETURN .T.<e>
