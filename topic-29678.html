#include "FiveWin.ch"

function main()
local oWnd

DEFINE WINDOW oWnd TITLE "Keys"

                            IIF( nKey = 34, MsgInfo( "Page DOWN : " + Str( nKey ) ), ) }

ACTIVATE WINDOW oWnd

RETURN ( NIL )
