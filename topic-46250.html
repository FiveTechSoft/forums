function Main()

local n, lAll := .F., lCode := .F., lNonCode := .F. 
local cHtml := ""
local hPost := UPost()
local cFilter := "ALL"

if hb_HHasKey( hPost, "filter" )
   cFilter := hPost[ "filter" ]
endif

do case
   case cFilter == "CODE"
        lCode := .T.
   case cFilter == "NONCODE"
        lNonCode := .T.
   otherwise
        lAll := .T.
endcase

for n = __DynsCount() to 1 step -1
    if lAll .or. lCode 
        if __DynsIsFun( n )
           cHtml += '<p style="color:green; margin: 2px 0;">' + AllTrim( Str( __DynsCount() - n + 1 ) ) + ". " + __DynsGetName( n ) + "</p>" + hb_EoL()
        endif
    endif
    if lAll .or. lNonCode
        if ! __DynsIsFun( n )
           cHtml += '<p style="color:blue; margin: 2px 0;">' + AllTrim( Str( __DynsCount() - n + 1 ) ) + ". " + __DynsGetName( n ) + "</p>" + hb_EoL()
        endif
    endif
next

return View( "symbols", "cContent", cHtml )

//----------------------------------------------------------------------------//

function View( cName, cParamName, uParamValue )

   local cData
   local cViewPath := "./views/" + cName + ".view"

   if File( cViewPath )
      cData = MemoRead( cViewPath )
      while ReplaceBlocks( @cData, "{{", "}}", cParamName, uParamValue )
      end
   else
      cData = "<h2>" + cName + " not found!</h2>" 
   endif    

return cData

//----------------------------------------------------------------------------//

function ReplaceBlocks( cCode, cStartBlock, cEndBlock, cParamName, uParamValue )

   local nStart, nEnd, cBlock, nQ1, nQ2
   local lReplaced := .F.
   

   while ( nStart := At( cStartBlock, cCode ) ) != 0 .and. ;
         ( nEnd := At( cEndBlock, cCode ) ) != 0
      cBlock = SubStr( cCode, nStart + Len( cStartBlock ), nEnd - nStart - Len( cEndBlock ) )
      

  // Keep it simple for this example, only support variable replacement or View call
  if "View" $ cBlock
     // Manual parsing to avoid macro compiler issues with function visibility: View("name")
     nQ1 := At( '"', cBlock )
     nQ2 := RAt( '"', cBlock )
     
     if nQ1 > 0 .and. nQ2 > nQ1
        cCode = SubStr( cCode, 1, nStart - 1 ) + ;
                View( SubStr( cBlock, nQ1 + 1, nQ2 - nQ1 - 1 ) ) + ;
                SubStr( cCode, nEnd + Len( cEndBlock ) )
     else
        cCode = SubStr( cCode, 1, nStart - 1 ) + ;
                "" + ;
                SubStr( cCode, nEnd + Len( cEndBlock ) )
     endif
  else
     // Simple variable replacement
     if !Empty( cParamName ) .and. AllTrim( cBlock ) == cParamName
         cCode = SubStr( cCode, 1, nStart - 1 ) + ;
                 ValToChar( uParamValue ) + ;
                 SubStr( cCode, nEnd + Len( cEndBlock ) )
     else
         // Unknown block, just remove brackets
         cCode = SubStr( cCode, 1, nStart - 1 ) + ;
                 "" + ;
                 SubStr( cCode, nEnd + Len( cEndBlock ) )
     endif
  endif

  lReplaced = .T.
   end
   

return lReplaced

//----------------------------------------------------------------------------//

function ValToChar( u )
   local cType := ValType( u )
   local cResult

   do case
      case cType == "C"
           cResult = u
      case cType == "N"
           cResult = AllTrim( Str( u ) )
      otherwise
           cResult = ""
   endcase
return cResult

Hello, Yes, this is exactly the topic I’ve been working on over the last days.

For me this is mainly a design question, not a missing function problem.

In a web context, long-running jobs simply don’t belong inside the web server process. Whether it’s Hix, PHP, or any other web runtime: blocking calls, process execution with I/O, or long database work will always clash with request/response handling.

My conclusion so far:

short, reactive logic → web layer (Hix / PHP / JS)

long-running or stateful jobs → external service process

Node.js, Python or even a standalone Harbour EXE

communication via HTTP or WebSocket

This keeps the web layer responsive and avoids timeouts, while still allowing full control over longer jobs.

So in my view, Hix behaving this way actually confirms a clean separation of concerns, rather than being a limitation.

//backend_persistent.prg
// HIX Backend - Persistent Service
// Uses a simple JSON file to simulate persistent state across requests,
// bypassing thread/process isolation issues.

#pragma -w0
#pragma -es0

FUNCTION Main()
    LOCAL hState := {=>}
    LOCAL cStateFile := "server_state.json"
    LOCAL cJson_In, cJson_Out
    LOCAL hResponse := {=>}

// 1. LOAD STATE (Persistence Layer)
IF File( cStateFile )
cJson_In := MemoRead( cStateFile )
hState   := hb_jsonDecode( cJson_In )
ELSE
hState[ "hits" ]       := 0
hState[ "start_time" ] := Time()
ENDIF

// 2. MODIFY STATE
hState[ "hits" ] := hState[ "hits" ] + 1

// 3. SAVE STATE
hb_MemoWrit( cStateFile, hb_jsonEncode( hState ) )

// 4. PREPARE RESPONSE
hResponse[ "type" ]         := "PERSISTENT"
hResponse[ "status" ]       := "ok"
hResponse[ "hits" ]         := hState[ "hits" ]
hResponse[ "server_start" ] := hState[ "start_time" ]
hResponse[ "message" ]      := "State Updated"

?? hb_jsonEncode( hResponse )

RETURN ""

Thanks Bayron for sharing the example and for your continued support of Hix and FiveWin. I think we actually agree on most fundamentals. The difference is mainly how we look at the workflow and responsibilities, not at Harbour’s capabilities.

To make this clearer, I’d like to put the two approaches side by side.

Hix-internal workflow (as in your example) Browser → HTTP request → Hix executes PRG → optional file-based state (JSON) → immediate response to UI

In this model:

logic runs inside the web request persistence is simulated via files there is no job lifecycle beyond the request the UI always drives execution

This works well as your code shows and clearly shows what Hix can do.

Our workflow (Hix + Harbour desktop program) Browser / UI → Request to Hix → Hix triggers Harbour desktop EXE (ShellExecute + unique job id) → Harbour program runs independently → writes result/status (JSON or file) → UI fetches or polls the result when it is ready

Here:

Hix stays lightweight and reactive long or blocking work runs outside the web process state belongs to the job, not to the request lifecycle, logging, recovery and cleanup are explicit

This is not about adding layers, but about separating roles early.

About Node.js The only reason I mentioned Node.js earlier is that it is optional and portable, not because it is required. A setup like: Harbour (worker) ↔ WebSocket ↔ Node.js (SQL adapter) ↔ SQL database

is simply one interesting option, especially for SQL access: no SQL libraries inside Harbour no additional DLL complexity clean isolation and well-maintained drivers

But to be very clear: Node.js is not mandatory at all.

Hix can directly trigger and communicate with an existing Harbour desktop program. No PHP, no additional framework, no extra runtime is required.

The core concern

What worries me from a product and long-term maintenance perspective is that once we start adding:

SQL libraries threading more dependencies directly into the Hix process, we slowly turn a clean web layer into a hard-to-maintain monolith.

My intention is not to limit Hix, but to protect it from becoming overloaded with responsibilities.

Best regards, Otto

CLASS MiClase
DATA aDatos INIT {}
METHOD Init()
METHOD GuardarLocal()
METHOD CargarLocal()
ENDCLASS

METHOD Init() CLASS MiClase
::aDatos := {}
RETURN Self

METHOD GuardarLocal() CLASS MiClase
LOCAL cJson := hb_jsonEncode( ::aDatos )

RETURN NIL

METHOD CargarLocal() CLASS MiClase
LOCAL cJson := MemoRead( "localstore.json" )
IF !Empty( cJson )
::aDatos := hb_jsonDecode( cJson )
ENDIF
RETURN NIL

ENDIF
