Function conectar()
Local Retorno:=.F.
LOCAL oErr,oCs
Public oCn

TRY
oCn:= CreateObject( "ADODB.Connection" )
oCs:= "DRIVER={MySQL ODBC 3.51 Driver}; SERVER=localhost; DATABASE=ecst06; UID=root; PWD=10157763"

CATCH oErr

END
