Code (fw): Select all Collapse
#include "fivewin.ch"

procedure main()
    ?fw_fscale(), fw_fscale()*100
return

#pragma BEGINDUMP

#include "hbapi.h"
#include <windows.h>

/*-----------------------------------------------------------------------------*/

HB_FUNC( FW_FSCALE )
{
   HWND hWnd = GetActiveWindow();
   HDC hdc = GetDC( hWnd );
   float fScale = GetDeviceCaps( hdc, LOGPIXELSX ) / 96.0f;
   ReleaseDC( hWnd, hdc );

   hb_retnd( (double) fScale );
}

/*-----------------------------------------------------------------------------*/

#pragma ENDDUMP
