La llamo de esta manera

Seguro( ( EMPLEADO ), "A" )
..............

Function Seguro( cAlias, cTipo) // B - Rlock A - Append Blank
local nSegundos := 1

Do Case
Case cTipo == "A"

   ( cAlias )-&gt;( dbappend() )
   If !NetErr()
      Return TRUE
   Endif

   Do While TRUE
      nSegundos++
      Inkey(1)

      ( cAlias )-&gt;( dbappend() )
      If !Neterr()
         Return TRUE
      Endif

      If nSegundos &gt;= 5
         MsgAlert("No se puede actualizar el registro"+Chr(13)+;
                     "Reintentar nuevamente")
         nSegundos := 1
      Endif

   EndDo

EndCase

Return FALSE
