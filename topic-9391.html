Prueba haciendolo asi

Code: Select all Collapse
(xArea)->( OrdScope(0, " 13P") )
(xArea)->( OrdScope(1, " 13P") )
(xArea)->(DbGoTop())

cTipDoc :=Str(field->tipodoc,3)+field->Serie+Field->Nombre 
(xArea)->( OrdSetFocus( 2 ) )
(xArea)->( OrdScope(0, cTipDoc ) )
(xArea)->( OrdScope(1, cTipDoc ) )

(xArea)->( OrdSetFocus( 1 ) )
