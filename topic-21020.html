REDEFINE GET oCuenta VAR CTABCO ;
OF oFld:aDialogs[1] ;
ID 336 PICTURE "@!" ;
VALID (IIF(EMPTY(CTABCO), ;
( CatchCta(oCuenta, CTABCO), oFld:Update(),.T.), .F.) )

oCuenta:lUpdate=.t.
