Here it is:

Code (fw): Select all Collapse
#include "Fivewin.ch"


FUNCTION MAIN()

    LOCAL oExcel, oSheet

    oExcel = CREATEOBJECT( "Excel.Application" )

    oExcel:WorkBooks:Add()

    oSheet = oExcel:ActiveSheet

    oSheet:Cells( 1, 1 ):Value = "This is a test"

    oSheet:SaveAs( CURDRIVE() + ":\" + CURDIR() + "\ThisIsATest.xlsx" )

    oExcel:Quit()

    RETURN NIL
