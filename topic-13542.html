local Result := nil, i := 0

   if PCount() == 2
   
      //-- Set -----------------------------------------------------------//

      for i := 1 to Len(::aFields)

         if Upper(cFieldName) == ::aFields[i,1]

      for i := 1 to Len(::aFields)

return Result
