my resize function
oApp():oBar is a ribbonbar

Function ResizeWndMain()
local aClient
if oApp():oDlg != NIL
aClient := GetClientRect (oApp():oWndMain:hWnd )
oApp():oDlg:SetSize( aClient[4], aClient[3] - oApp():oBar:nHeight - oApp():oWndMain:oMsgBar:nHeight )
oApp():oDlg:Refresh()
endif
return NIL
