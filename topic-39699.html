Function cGetFile32( cFileMask, cTitle, nDefaultMask, cInitDir, lSave, nFlags, cIniFile )



local cSection, cFile

local nRet



DEFAULT lSave := .f.



CursorWait()

// On cGetFile() that parameter is a boolean for long filenames

nFlags := Nil

endif

else

endif
