Function  RefreshCont(oCont,oDbf)
    if oCont != NIL
    oCont:cTitle := tran(oDbf:OrdKeyNo(),'@E 999,999')+" / "+tran( oDbf:OrdKeycount(),'@E 999,999')
    oCont:refresh()
   endif
   RETURN NIL
//------------------------------------------------------------//
 function ChangeButtons( oBar )
       AEval( oBar:aControls, { | oCtrl | oCtrl:nTop += 4, oCtrl:nHeight -= 4 } )
       return .T.
