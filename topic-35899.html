#include "FiveWin.ch"

function Main()
local cPrg, cError, cResult

   cPrg := 'function Test(x,y) ' + CHR(13)+CHR(10)
   cPrg += 'local z '            + CHR(13)+CHR(10)
   cPrg += '   z := x * y - 2 '  + CHR(13)+CHR(10)
