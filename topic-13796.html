Function dia2dig(campo)
local retval:=''
local dctos:=Str(Day(campo),2,0)

If Len(dctos) = 1
Retval:='0'+dctos +' de '+cMonth(campo)
Else
Retval:=dctos +' de '+cMonth(campo)
EndIf
Return Retval
