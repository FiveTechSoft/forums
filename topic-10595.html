How to make this with FWH ?

Posted: Sat Apr 05, 2008 02:35 PM

This is the vtitle CH file.

xcommand @ <nTop>, <nLeft> VTITLE [ <oTitle> ] ;

           [ SIZE &lt;nWidth&gt;, &lt;nHeight&gt; ] ;
           [ &lt;dlg:OF,DIALOG&gt; &lt;oDlg&gt; ] ;
           [ CAPTION &lt;cText&gt; ] ;
           [ ACTION &lt;uAction,...&gt; ] ;
           [ FONT &lt;oFont&gt; ] ;
           [ IMAGE &lt;cImage&gt; ] ;
           [ MARGIN &lt;nMargin&gt; ] ;
           [ &lt;lLeftImg: LEFTIMG&gt; ] ;
           [ &lt;lIcon: ICON&gt; ] ;
           [ &lt;lBorder: BORDER&gt; ] ;
           [ &lt;lOnOff: ONOFF&gt; ] ;
           [ &lt;color: COLOR, COLORS&gt; &lt;nClrText&gt; [,&lt;nClrPane&gt; ] ] ;
           [ COLORBAR &lt;nClrBar&gt;[, &lt;nClrBar2&gt; ]  ] ;
           [ COLORBORDER &lt;nClrBorde&gt; ] ;
           [ SPEED &lt;nSpeed&gt; ] ;
           [ ANIMATION &lt;aBmps,...&gt; ] ;
           [ &lt;lDnArrow: NODOWNARROW&gt; ] ;
           [ &lt;lFlash: FLASH&gt; ] ;
  =&gt; ;
      [ &lt;oTitle&gt; := ] TTitle():New( &lt;nTop&gt;, &lt;nLeft&gt;, &lt;nWidth&gt;, &lt;nHeight&gt;, &lt;oDlg&gt;,;
                      &lt;cText&gt;, &lt;cImage&gt;, &lt;.lLeftImg.&gt;, &lt;.lIcon.&gt;, &lt;.lBorder.&gt;, &lt;nClrText&gt;, &lt;nClrPane&gt;,;
                      &lt;nClrBar&gt;, &lt;nClrBar2&gt;, [{|this|&lt;uAction&gt;}], &lt;oFont&gt;, &lt;nMargin&gt;, [\{&lt;aBmps&gt;\}], &lt;nSpeed&gt;,;
                       &lt;.lOnOff.&gt;, &lt;.lDnArrow.&gt;,;
                      &lt;nClrBorde&gt;, &lt;.lFlash.&gt; )
