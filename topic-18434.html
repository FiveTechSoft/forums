static function fsedron(oBrw,oHasedro,nclia)
local aProd := bprodu()
local oDatasedro:= oclone(oHasedro)



if empty(aProd)
return .f.
endif


if aProd[5] = "S"
oDatasedro:blank()

oDatasedro:codigo := aProd[1]
oDatasedro:producto := aProd[2]
oDatasedro:append()
if reviso(oDatasedro)
oDatasedro:save()
oDatasedro:commit()
oDatasedro:unlock()
endif
oBrw:refresh()
else
msgstop("No pertenece al listado de Sedronar")
endif

sysrefresh()
return nil
