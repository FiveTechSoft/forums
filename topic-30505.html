Function PintaCelula( nRow, nCol, status, qCol )
Local nCor := nRGB(255,255,255)
if nRow > 0
do case
case status = "V"
if nCol = qCol
nCor = nRGB( 236,253,255 ) // CLR_AZULADITO
endif
case status = "A"
if nCol = qCol
nCor = nRGB( 234,255,229 ) // CLR_VERDEAGUA
endif
case status = "R"
if nCol = qCol
nCor = nRGB( 255,238,238 ) // CLR_VENCIDOS
endif
endcase
endif
return nCor
