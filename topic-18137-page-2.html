Marcelo,

Modifica fwdbg.prg y añade...
Code (fw): Select all Collapse
*------------------------------------------------------------------------
FUNCTION __dbgEntry( nMode, uParam1, uParam2, uParam3, uParam4, uParam5 )
*------------------------------------------------------------------------
   LOCAL hRes := GetResources()     // <-- Esta linea

   DO CASE

       CASE nMode == HB_DBG_GETENTRY

            __dbgSetEntry()

       CASE nMode == HB_DBG_ACTIVATE

            IF oDbg == NIL

               oDbg := TDbg():New()
               oDbg:Activate()

            ENDIF

            oDbg:pInfo      := uParam1
            oDbg:nProcLevel := uParam2
            oDbg:aStack     := uParam3
            oDbg:aModules   := uParam4
            oDbg:Exec()

   ENDCASE

   SetResources( hRes )    // <-- Esta linea


RETU NIL
