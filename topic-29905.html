Jose Luis,

Prueba este ejemplo:

Code (fw): Select all Collapse
#include "FiveWin.ch"

static oWnd

function Main()

   local oBar

   DEFINE WINDOW oWnd MDI

   DEFINE BUTTONBAR oBar OF oWnd 2007 SIZE 50, 50 

   DEFINE BUTTON OF oBar ACTION VentanaHija()

   DEFINE MSGBAR OF oWnd PROMPT "Ejemplo MDI" 2007

   ACTIVATE WINDOW oWnd

return nil

function VentanaHija()

   local oWndChild, oBar, oFolder

   DEFINE WINDOW oWndChild MDICHILD OF oWnd

   DEFINE BUTTONBAR oBar OF oWndChild 2007 SIZE 50, 50 

   DEFINE BUTTON OF oBar ACTION MsgInfo( "acción" )

   @ 0, 0 FOLDER oFolder OF oWndChild

   oWndChild:oClient = oFolder

   DEFINE MSGBAR OF oWndChild PROMPT "Mensaje ventana hija" 2007

   ACTIVATE WINDOW oWndChild

return nil
