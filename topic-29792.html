aquí está el código:
Code (fw): Select all Collapse
#include "Fivewin.ch"
#include "report.ch"

FUNCTION Main()
    LOCAL aReturn:={}, i:=0, j:=0
    //listado desde un array
    FOR i=1 TO 25
        for j=1 to 5
            aadd( aReturn, { Str(i,3), "texto", 25 } )
        NEXT j
    NEXT i
    
    REPORT oReport TITLE "Listado" CAPTION "Listado" PREVIEW SUMMARY
    
        COLUMN TITLE "# " ;
            DATA aReturn[i,1] ;
            PICTURE "999" ;
            SIZE 4
        
        COLUMN TITLE "# " ;
            DATA aReturn[i,2] ;
            PICTURE "!!!!!" ;
            SIZE 7
        
        COLUMN TITLE "Cantidad:" ;
            DATA aReturn[i,3] ;
            PICTURE "9,999,999" ;
            SIZE 10;
            TOTAL
            
        GROUP ON aReturn[i,1];
            HEADER "Producto: " + aReturn[i,1]
            
    END REPORT
    
    oReport:bSkip := { || i++  }
    oReport:bInit := {|| i := 1 }
    
    ACTIVATE REPORT oReport WHILE i <= LEN( aReturn )

RETURN nil

Bueno, lo probé con tablas dbf y es igual, vease este ejemplo de código:
Code (fw): Select all Collapse
#include "Fivewin.ch" 
#include "Ord.Ch"
#include "report.ch"

FUNCTION Main()
    LOCAL oReport
    FIELD age
    
    REQUEST DBFCDX
    RddSetDefault( 'DBFCDX' )
     SET AUTOPEN ON
    
    //listado desde una tabla
    USE CUSTOMER NEW SHARED
    INDEX ON AGE TAG AGE
    GO TOP
    
    REPORT oReport TITLE "Listado" CAPTION "Listado" PREVIEW SUMMARY
    
        COLUMN TITLE "Nombre" ;
            DATA CUSTOMER->first ;
            PICTURE "!!!!!!!!!!!!!!!!!!!!" ;
            SIZE 10
        
        COLUMN TITLE "Monto Salarios:" ;
            DATA CUSTOMER->salary ;
            PICTURE "9,999,999.99" ;
            SIZE 15;
            TOTAL
            
        GROUP ON CUSTOMER->age ;
            HEADER "Edad: " + Str( CUSTOMER->age, 3 )
            
    END REPORT
    
    oReport:bInit := {|| DbGoTop() }
    
    ACTIVATE REPORT oReport 

    CLOSE DATABASES
    
RETURN nil
