Añadido un primer soporte para arrays en el browser .

Code (fw): Select all Collapse
function Browse()

   local oDlg, oBrw
   local aDatos:={}

   if ! File( "clients.dbf" )
      DbCreate( "clients.dbf", { { "title",   "C", 10, 0 },;
                                 { "family",  "C", 80, 0 },;
                                 { "vorname", "C", 80, 0 } } )
   endif
   
   USE clients SHARED

   aadd( aDatos, {"title","Family","Vorname"} )
   
   go top
   do while !eof()
      aadd(aDatos,{clients->title, clients->family,clients->vorname })
      skip
   enddo

   
   DEFINE DIALOG oDlg TITLE "Clients browse" SIZE 800, 600
   
   @ 10, 10 BROWSE oBrw SIZE 500, 400 OF oDlg ARRAY aDatos
   
   ACTIVATE DIALOG oDlg NOWAIT
   
   USE

return nil

function Browse()

   local oDlg, oBrw
   local aDatos:={}

   if ! File( "clients.dbf" )
      DbCreate( "clients.dbf", { { "title",   "C", 10, 0 },;
                                 { "family",  "C", 80, 0 },;
                                 { "vorname", "C", 80, 0 } } )
   endif
   
   USE clients SHARED

   aadd( aDatos, {"title","Family","Vorname"} )
   
   go top
   do while !eof()
      aadd(aDatos,{clients->title, clients->family,clients->vorname })
      skip
   enddo

   DefineClassLine()   
   DefineClassHead()  
   
   DEFINE DIALOG oDlg TITLE "Clients browse" SIZE 800, 600
   
   @ 10, 10 BROWSE oBrw SIZE 500, 400 OF oDlg ARRAY aDatos
   
    oBrw:cClassTable:= "" //"browse"
    oBrw:cClassLine:="linea"
    obrw:cClassHead:= "boxtitulo"
    oBrw:lZebra:= .t.

   oBrw:CreateFromCode()
   
   ACTIVATE DIALOG oDlg NOWAIT
   
   USE

return nil
