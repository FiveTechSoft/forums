Silvio,

@ 10, 10 XBROWSE oBrw OF oDlg ;
COLUMNS 1, 2 ;
HEADERS "Cars","Values" ;
COLSIZES 110, 110 ;
ARRAY aData LINES FASTEDIT CELL ;
SIZE 140, 200 PIXEL FONT oTxtFont ;
ON CHANGE { aData1 := aData[oBrw:nArrayAt, 1 ], ; // Value of 1. Column
aData2 := aData[oBrw:nArrayAt, 2 ], ; // 2. Column
oGet1:Refresh(), oGet2:Refresh() } // 2. get displays Value of Column 2

@ 10, 160 GET oGet1 VAR aData1 OF oDlg SIZE 48, 15 PIXEL ;
FONT oTxtFont COLOR 0, 16443068 UPDATE
@ 10, 212 GET oGet2 VAR aData2 OF oDlg SIZE 48, 15 PIXEL ;
FONT oTxtFont COLOR 0, 16443068 UPDATE
