FUNCTION QueoCbx( nOpt, cText, oGet )

   LOCAL cTxtSil := "Silvio"
   LOCAL nYear := YEAR( DATE() )

   IF nOpt = 1

      lWhenoCbx := .F.

      oGet:Hide()
      oGet:Refresh()

   ELSEIF nOpt = 2

      lWhenoCbx := .F.

      oGet:Show()
      oGet:Disable()
      oGet:VARPUT( nYear )
      oGet:Refresh()

   ELSEIF nOpt = 3  // Silvio Falconi

      lWhenoCbx := .T.

      oGet:Show()
      oGet:Enable()
      oGet:VARPUT( cTxtSil )
      oGet:Refresh()

      xFocus( oGet )

   ENDIF

RETURN NIL

FUNCTION xFocus( oObj )

   xSetFocus( oObj )
   xSetFocus( oObj )

RETURN( .T. )

FUNCTION xSetFocus( oObj )    

   LOCAL _oWnd := oObj:oWnd, _oTempo := ""

   DEFINE TIMER _oTempo INTERVAL 10 OF _oWnd ;
      ACTION ( oObj:SetFocus(), _oTempo:End() )

   ACTIVATE TIMER _oTempo

RETURN( .T. )
