<i>    </i>aDir = DIRECTORY( "src\codepage\*.*" )

<i>    </i>FOR i = 1 TO LEN( aDir )
<i>    </i>    cTxt = MEMOREAD( "src\codepage\" + aDir[ i, F_NAME ] )

<i>    </i>    cTxt = TOOCTAL( cTxt )

<i>    </i>    IF cTxt != NIL
<i>    </i>        MEMOWRIT( "src\codepage\" + aDir[ i, F_NAME ], cTxt, .F. )
<i>    </i>    ENDIF
<i>    </i>NEXT

<i>    </i>aDir = DIRECTORY( "src\lang\*.*" )

<i>    </i>FOR i = 1 TO LEN( aDir )
<i>    </i>    cTxt = MEMOREAD( "src\lang\" + aDir[ i, F_NAME ] )

<i>    </i>    cTxt = TOOCTAL( cTxt )

<i>    </i>    IF cTxt != NIL
<i>    </i>        MEMOWRIT( "src\lang\" + aDir[ i, F_NAME ], cTxt, .F. )
<i>    </i>    ENDIF
<i>    </i>NEXT
<i>    </i>
 STATIC FUNCTION TOOCTAL( cTxt )

<i>    </i>LOCAL cRes := ""

<i>    </i>LOCAL nVal, cVal

<i>    </i>LOCAL lUtf := .F.

<i>    </i>LOCAL i

<i>    </i>FOR i = 1 TO LEN( cTxt )
<i>    </i>    IF ASC( SUBSTR( cTxt, i, 1 ) ) <= 127
<i>    </i>        cRes += SUBSTR( cTxt, i, 1 )
<i>    </i>    ELSE
<i>    </i>        nVal = ASC( SUBSTR( cTxt, i, 1 ) )

<i>    </i>        cVal = ""

<i>    </i>        WHILE nVal / 8 > 0
<i>    </i>            cVal = NTRIM( INT( nVal % 8 ) ) + cVal
<i>    </i>            nVal = INT( nVal / 8 )
<i>    </i>        ENDDO

<i>    </i>        cRes += "\" + cVal

<i>    </i>        lUtf = .T.
<i>    </i>    ENDIF
<i>    </i>NEXT

<i>    </i>RETURN IF( lUtf, cRes, NIL )
