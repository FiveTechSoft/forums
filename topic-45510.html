* ========================================================================
CLASS TVWeb FROM TVisualForm
* ========================================================================
  DATA cPage   
  DATA oWebView
    
  METHOD New() CONSTRUCTOR
  METHOD AddControl()
  METHOD BuildMenu()
    
END CLASS

* ========================================================================
METHOD New( xcPage, xcNome ) CLASS TVWeb
* ========================================================================

#ifdef __XHARBOUR__
  
  TVWebMsg() ; return Self
  
#endif
    
  DEFAULT xcPage := ""
  DEFAULT xcNome := Empresas(1)  
  
  ::New("rel")
  
  ::cNome       := xcNome
  ::cPage       := "http://localhost:3000/" + xcPage  // nodejs
  ::lToolBar    := FALSE  
  ::lTask       := FALSE
  ::lRastrear   := FALSE
   
  ::AddControl()
  
  ::bValid := {|| ::Terminate(),;
                                  WndMain():Show()    ,;
                                  WndMain():SetFocus(), TRUE }
                                
return Self

* ========================================================================
METHOD AddControl() CLASS TVWeb
* ========================================================================

#ifndef __XHARBOUR__

  ::= TWebView2():New()
  
  ::SetTitle( WebTitle("Sisrev-Web") ) 
  ::SetUserAgent( WebAgent() )
  ::bResized := {|| ::Maximize() }
  ::SetMenu( ::BuildMenu() )  
  ::SetColor( CLR_BLACK ,nRGB( 231, 242, 255 )  )
  ::l2007    := .T.
  ::SetSize( WndWidth( ::hWnd ) - 15, WndHeight( ::hWnd ) - 58 )  
 
  ::Navigate( ::cPage )  
  
  ::OpenDevToolsWindow(.F.)
  ::ShowDownloads(.F.)    
  ::bInit := {|| ::Maximize() }
                                  
  ::Run()  
  ::Destroy()
  
  WndMain():Hide()
 
#endif

return nil

* =========================================================================
METHOD BuildMenu() CLASS TVWeb
* =========================================================================
  local oMenu

  MENU oMenu 
     MENUITEM "Fechar" ACTION ::End()
  ENDMENU
   
return oMenu

* ========================================================================
Function ExecWeb( cData )
* ========================================================================
   
  if ValType( cData ) == "A"
     cData := HB_VALTOEXP(cData) /* converts any data type to STR Value */
     cData := strtran( cData, "{", "")
     cData := strtran( cData, "}", "")
     cData := strtran( cData, "'", "")
  Endif  

  OnGet( "&classe=" + cData )
   
return nil
