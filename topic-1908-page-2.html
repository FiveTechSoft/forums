Try this:

Code: Select all Collapse
#include "Fivewin.ch" 


FUNCTION MAIN() 

    LOCAL oDlg, oBrw 

    use test 
    index on field -> last to temp for recno()<1 
    go top 

    DEFINE DIALOG oDlg; 
           SIZE 400, 400 

    @ 1, 1 LISTBOX oBrw FIELDS alias "test" 

    oBrw:bLogicLen = { || ( oBrw:cAlias ) -> ( OrdKeyCount() ) }

    ACTIVATE DIALOG oDlg; 
             ON INIT oDlg:SetControl( oBrw ); 
             CENTER 

    RETURN NIL
