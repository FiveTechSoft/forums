Hola lucasdebeltran, si se puede.

ACTIVATE DIALOG oDlg.......ON INIT MenuDialogo(oDlg)

FUNCTION MenuDialogo(oDlg)
LOCAL oMenu
MENU oMenu 2007
MENUITEM "Mantenimientos "
MENU
MENUITEM "&Nuevo " ACTION ...lo que quieras
MENUITEM "&Modificar " ACTION ....
MENUITEM "&Borrar " ACTION ...
MENUITEM "&Imprimir " ACTION ...
ENDMENU
MENUITEM "S&alir " ACTION(oDlg:End())

ENDMENU
oDlg:SetMenu(oMenu)

Return nil
