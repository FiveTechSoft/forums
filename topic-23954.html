Code (fw): Select all Collapse
   INDEX ON FIELD->FSERIE TAG WFSERIE FOR !Deleted()

   USE BANCOSERIE....

   SET ORDER...

   cIniSerie := "B2"
   cFinSerie := "D1"

   OrdScope(0, cIniSerie )
   OrdScope(1, cFinSerie )
   GO TOP

   BROWSE()
