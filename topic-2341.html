Por este otro:

ACTIVATE DIALOG oDlg CENTERED ON INIT pru(oDlg) ;

Function Pru(oDlg)
Local oGet , cPru := "Un get cualquiera"

oDlg:Move( 0,0, 600,600, .T.)
oDlg:Center()

@500,100 GET oGet VAR cPru PIXEL SIZE 200,24 OF oDlg

Return Nil
