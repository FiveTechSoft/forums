Yo utilizo esta función. Es 'cutre' está orientada para depurar lo que codifico pero me da muy buen resultado:
Eemplos de uso:
Traza( 0, "C:\TEMP\", "LOG.TXT" )
Traza( 1, "lo que me de la gana", "valor de la variable nCaso=", nCaso )
/* ***************************************** */
Code (fw): Select all Collapse
#include "FiveWin.ch"
#include "fileio.ch"
#include "Set.ch"
#include "common.ch"

STATIC lGraba  := .T.
STATIC cPath   := ""
STATIC cFitxer := ""

FUNCTION Traza( nAccion, A, B, C, D, E, F, G, H, I, J, K, L, M, N, O )

Local cPathFitx := ""
Local hfichero  := 0
Local cLiteral  := ""
Local cFormatdate := ""

DEFAULT A TO ""
DEFAULT B TO ""

/* Inicializa valores para la grabación de las trazas.
   -------------------------------------------------*/
If nAccion = 0
    cPath = A
    cFitxer = B
    IF cFitxer <> "SINTRAZA.TXT"
        lGraba  := .T.
        Traza(1, "Started.")
    EndIf

/* Graba las trazas.
   ---------------*/
ElseIf nAccion = 1 .and. lGraba

    DEFAULT C TO ""
    DEFAULT D TO ""
    DEFAULT E TO ""
    DEFAULT F TO ""              //37
    DEFAULT G TO ""
    DEFAULT H TO ""
    DEFAULT I TO ""
    DEFAULT J TO ""
    DEFAULT K TO ""
    DEFAULT L TO ""
    DEFAULT M TO ""
    DEFAULT N TO ""
    DEFAULT O TO ""

    cFormatdate := set( _SET_DATEFORMAT , "dd-mm-yy" )
    cPathFitx = cPath + cFitxer

    /*cLiteral = dtoc(Date()) + "/" + Time() + ;
               If( At(":", ProcName(5) ) > 0, "/" + ProcName(6),"") + ;
               If( At(":", ProcName(4) ) > 0, "/" + ProcName(5),"") + ;
               If( At(":", ProcName(3) ) > 0, "/" + ProcName(4),"") + ;
               If( At(":", ProcName(2) ) > 0, "/" + ProcName(3),"") + ;
               If( At(":", ProcName(1) ) > 0, "/" + ProcName(2) + Str(ProcLine(2),4,0),"") + ;
               "/" + ProcName(1) + "<>" + Str(ProcLine(1),4,0) + "->" + ;
               cValToChar(A) + cValToChar(B) + cValToChar(C) + cValToChar(D) + cValToChar(E) + ;
               cValToChar(F) + cValToChar(G) + cValToChar(H) + cValToChar(I) + cValToChar(J) + ;
               cValToChar(K) + cValToChar(L) + cValToChar(M) + cValToChar(N) + cValToChar(O)+ CRLF*/

    cLiteral = dtoc(Date()) + "/" + Time() + ProcName(1) + "<>" + Str(ProcLine(1),4,0) + "->" + ;
               cValToChar(A) + cValToChar(B) + cValToChar(C) + cValToChar(D) + cValToChar(E) + ;
               cValToChar(F) + cValToChar(G) + cValToChar(H) + cValToChar(I) + cValToChar(J) + ;
               cValToChar(K) + cValToChar(L) + cValToChar(M) + cValToChar(N) + cValToChar(O)+ CRLF


    hfichero =fopen(cPathFitx, FO_READWRITE + FO_SHARED)

    If Ferror() != 0
        hfichero =fcreate(cPathFitx,FC_NORMAL)
    Endif

    If Ferror() != 0               //74
        traza(0, cfilepath( Arg( 0 ) ), "AMPA32.TXT" )
        traza(1, ProcName(2), ;
                  "--Error al grabar el fichero: " + cPathFitx + ;
                  CRLF + "Gravado en:" + cfilepath( Arg( 0 ) ) + "AMPA32.TXT" )
        Traza(1, ProcName(1) + "<>" + A, B, C, D, E, F, G, H, I, J, K, L, M, N, O )
        ? "Error al grabar el fichero: " + cPathFitx + CRLF + "Gravado en:" + ;
          cfilepath( Arg( 0 ) ) + "AMPA32.TXT"
      else
        fseek(hfichero, 0, FS_END)
        fwrite(hfichero, cLiteral )
        fclose(hfichero)
    Endif

    set( _SET_DATEFORMAT , cFormatDate )

/* Detiene la grabación de trazas.
   -----------------------------*/
ElseIf nAccion = 2

    Traza(1, "Stop.")
    lGraba  := .F.

/* Reinicia la grabación de trazas.
   ------------------------------*/
ElseIf nAccion = 3

    lGraba  := .T.

/* Devuelve una array de dos elementos con el cPath y el cFitxer.
   ------------------------------------------------------------*/
ElseIf nAccion = 4

    Return { cPath, cFitxer }

/* Borra el fichero de trazas.
   -------------------------*/
ElseIf nAccion = 5

    cPathFitx = cPath + cFitxer

    If File(cPathFitx)
        ferase( cPathFitx )
    EndIf

EndIf

return .T.

Adolfo,

Puedes usar la funcion LogFile() de FW. Es muy simple y sencilla de usar:

LogFile( cFileName, aValues )

aValues es un array con los elementos que quieras y del tipo que quieras. No hace falta que los pases a cadena.

Por defecto te incluye la fecha y la hora en el fichero log. Puedes llamar a LogFile() tantas veces como quiera que va añadiendo lineas al final del fichero log.
