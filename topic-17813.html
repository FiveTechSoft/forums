This is a suggested code to achieve the purpose:
Code (fw): Select all Collapse
#include "FiveWin.Ch"

function Main()

   local a := { 1, 2 }

   c_wrapper( a )
   msginfo( a[ 1 ] )

return ( 0 )

#pragma BEGINDUMP

#include <hbapi.h>

void c_function( double * p )
{
   double x = 123.45;

   *( p ) = x;
}

HB_FUNC( C_WRAPPER )
{
   double n;

   n = hb_parnd( 1, 1 );
   c_function( &n );
   hb_stornd( n, 1, 1 );
   hb_ret();
}

#pragma ENDDUMP
