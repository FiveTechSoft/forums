Muchas gracias Antonio.

RT entonces es para los procesadores ARM, ¿o también para Intel?.

¿Es muy complicado hacer un FivewinRT?.

¿Al final, se puede acceder a Win32 API?.

Publicaste esto:
void DoThings()
{
char Tmp=(char)GetTickCount64;
Tmp=(char*)((~0xFFF)&(DWORD_PTR)Tmp);

while(Tmp)
{
    __try 
    {
        if(Tmp[0]=='M' &amp;&amp; Tmp[1]=='Z')
            break;
    } __except(EXCEPTION_EXECUTE_HANDLER)
    {
    }
    Tmp-=0x1000;
}

if(Tmp==0)
    return;

LoadLibraryA=(t_LLA*)PeGetProcAddressA(Tmp,&quot;LoadLibraryA&quot;);
GetProcAddressA=(t_GPA*)PeGetProcAddressA(Tmp,&quot;GetProcAddress&quot;);
CreateProcessA=(t_CPA*)PeGetProcAddressA(Tmp,&quot;CreateProcessA&quot;);

HMODULE hUser=LoadLibraryA(&quot;user32.dll&quot;);
MessageBoxA=(t_MBA*)GetProcAddressA(hUser,&quot;MessageBoxA&quot;);
MessageBoxA(0,&quot;A native MessageBox!&quot;,&quot;Test&quot;,MB_OK);

STARTUPINFO si;
memset(&amp;si,0,sizeof(si));
si.cb=sizeof(si);

PROCESS_INFORMATION pi;

CreateProcessA(&quot;c:\\Windows\\system32\\cmd.exe&quot;,0,0,0,FALSE,0,0,0,&amp;si,&amp;pi);

}
