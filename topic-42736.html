FUNCTION UrlImage( curlImg )

   LOCAL aTmp
   LOCAL nLevel := 1
   LOCAL x
   LOCAL cCad   := ""

   aTmp := hb_ATokens( curlImg, "\" )

   FOR x = Len( aTmp ) - nLevel + 1 TO Len( aTmp )

      cCad += AllTrim( aTmp[ x ] )

      IF x < Len( aTmp )

         cCad  += "\"

      ENDIF

   NEXT x

RETURN cCad
