#include "FiveWin.ch"

STATIC oWnd

function Main()

   local n,hwnd

   n:=Seconds()

   do while .t.

      sysrefresh()

      if seconds()>n+20

         //hWnd:=GetActiveWindow()
         
         hWnd := ShellExecute( GetActiveWindow() )

         if msgyesNo("Active Window handle is "+str(hWnd,10)+" Finish test ?")

            exit

         endif

         n:=Seconds()

      endif

   enddo

Return nil
