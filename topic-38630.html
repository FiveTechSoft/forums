function WebPageSource( cURL )


   local oURL
   local oClient

   local cRetBuf

   oURL := tURL():New( cUrl )

   oClient := TIpClientHttp():new( oUrl )


   if oClient:Open()

      cRetBuf := oClient:readAll()

   endif

   oClient:Close()

   msgalert( "Close", oClient:lastErrorMessage() )
