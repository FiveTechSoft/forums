Hua,

Please modify Class TTabs Method MouseLeave() this way:
Code: Select all Collapse
METHOD MouseLeave( nRow, nCol, nFlags ) CLASS TTabs

   if ::nOver != nil .and. ::nOver > 0
      ::PaintOver( ::nOver, .F. )
   endif
 
   ::nOver = 0

return nil
