STATIC FUNCTION TROCA_INDICE( nOpcaoRadio, oDbfTabela, oLbx  )

    IF nOpcaoRadio == 2
        SELECT ( oDbfTabela:cAlias )
        oDbfTabela:SetOrder( 2 )  //-> Por Nome do Curso
    ELSEIF nOpcaoRadio == 1
        SELECT ( oDbfTabela:cAlias )
        oDbfTabela:SetOrder( 1 )  //-> Por C¢digo do Curso
    ENDIF

RETURN NIL
