Bueno, la verdad que no es mérito mio, si no de un compañero de curro, que además usando Java daba el mismo problema si nos empeñamos en usar un cliente SOAP.

Para ello simplemente hay que usar HTTP, si, HTTP sin mayores problemas,
eso si, el contenido tiene que estar formador por los tags de SOAP, en el ejemplo lo veréis mejor;

FUNCTION WithPost()
Local oHttp
Local cLocation := "http://ipservices/mensaje_Soap"
Local cUser := "usario:password"
Local cAuthorization := hb_base64( cUser, Len( cUser ) )
Local cXml := Mirar_Ejemplo_XML_Mas_Abajo()

try
  oHttp := CreateObject( &quot;MSXML2.XMLHTTP&quot; )
catch e
     ? &quot;Error de creacion&quot;
end

oHttp:Open( &quot;POST&quot;, cLocation, .F. )
oHttp:SetRequestHeader( &quot;Content-Type&quot;, &quot;application/soap+xml&quot; )

// --&gt; Basic es un sistema, hay otros, mirar que tipo se usa.
oHttp:SetRequestHeader( &quot;Authorization&quot;,&quot;Basic &quot; + cAuthorization )
oHttp:SetRequestHeader( &quot;SOAPAction&quot;,&quot;Operacion_del_Soap&quot; )

// Ahora solamente el XML con tags SOAP.
oHttp:Send( cXml )

? oHttp:responseText

return nil

// --- XML DE EJEMPLO
<soapenv:Envelope xmlns:soapenv="http://schemas.xmlsoap.org/soap/envelope/" xmlns:sol="http://ejemplo">
<soapenv:Header/>
<soapenv:Body>
<sol:respuesta_ws>
<EJEMPLO>
<item>
<NOMBRE>Nuevo_Cliente</NOMBRE>
</item>
</EJEMPLO>
</sol:respuesta_ws>
</soapenv:Body>
</soapenv:Envelope>

include <string.h>

include <limits.h>

include <stdlib.h>

include "hbapi.h"

/
* base64enc()
*
* Encode the data in 's' (length of the data is 'len') in BASE64. The returned
* string should be freed when not used anymore.
/

static char * base64enc( char *s, size_t s_len )
{
char b64chars[] =
"ABCDEFGHIJKLMNOPQRSTUVWXYZabcdefghijklmnopqrstuvwxyz0123456789+/";
char * t;
char * p;
int x, y;
int len;

if( s_len > ( size_t ) INT_MAX )
{
return NULL ; / die("data too long in base64enc()"); /
}
len = ( int ) s_len;
t = ( char * ) hb_xgrab( ( 4 * ( ( len + 2 ) / 3 ) + 1 ) * sizeof( char ) );
p = t;

while( len-- > 0 )
{
x = s++;
p++ = b64chars[(x >> 2) & 63];
if( len-- <= 0 )
{
p++ = b64chars[(x << 4) & 63];
p++ = '=';
p++ = '=';
break;
}
y = s++;
p++ = b64chars[((x << 4) | ((y >> 4) & 15)) & 63];
if( len-- <= 0 )
{
p++ = b64chars[(y << 2) & 63];
p++ = '=';
break;
}
x = s++;
p++ = b64chars[((y << 2) | ((x >> 6) & 3)) & 63];
p++ = b64chars[x & 63];
}
*p = '\0';

return t;
}

HB_FUNC( BUILDUSERPASSSTRING )
{
char * s;
char * szUser = hb_parcx( 1 );
char * szPass = hb_parcx( 2 );
size_t p_len = strlen( szPass );
size_t u_len = strlen( szUser );

s = ( char * ) hb_xgrab( ( u_len + p_len + 3 ) * sizeof( char ) );
s[0] = '\0';
strcpy( s + 1, szUser );
strcpy( s + u_len + 2, szPass );

hb_retc_buffer( s );
}

HB_FUNC( HB_BASE64 )
{
char * szItem = hb_parc( 1 );
int nLen = hb_parni( 2 );
char * szRet = szItem ? base64enc( szItem, nLen ) : NULL;

if( szRet )
hb_retc_buffer( szRet );
else
hb_retc( NULL );
}
