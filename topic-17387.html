Code (fw): Select all Collapse
static procedure BarraBot( oWChld, oBrw, oDbAct, nEsArea, bBrowse2 )
local oImgLst , ;
      oReBar  , ;
      oToolBar, ;
      oCursor , ;
      nFor

   // Cursor mouse
   DEFINE CURSOR oCursor HAND

   // Imagenes
   oImgLst:= TImageList():New( 32, 32 )
   oImgLst:AddMasked( TBitmap():Define( "bmp_Agrega",, oWChld ), "bmp_Trans32" )
   if nEsArea == DB_FAMILIAS .or. nEsArea == DB_RUBROS
      oImgLst:AddMasked( TBitmap():Define( "bmp_BuscaGL",, oWChld ), "bmp_Trans32" )
   end
   oImgLst:AddMasked( TBitmap():Define( "bmp_Fin",, oWChld ), "bmp_Trans32" )

   // Botones
   oReBar:= TReBar():New( oWChld )
   oToolBar:= TToolBar():New( oReBar, 50, 50, oImgLst, TRUE )
   oToolBar:nWidth:= 400

   oToolBar:AddButton( { || ap_Abms( oBrw, oDbAct, oDbAct, TRUE, nEsArea) }, ;
                       "Agrega "+if( nEsArea == DB_FAMILIAS, "Familia", ;
                                 if( nEsArea == DB_RUBROS,   "Rubros" , ;
                                 if( nEsArea == DB_OBRAS,    "Obras", "Proveedores" ))), ;
                       "Agrega" )
   if nEsArea == DB_FAMILIAS .or. nEsArea == DB_RUBROS
      oToolBar:AddButton( { || BusquedaGlb( oWchld, oDbAct, oBrw, bBrowse2, if( nEsArea==DB_FAMILIAS, DB_INSUMOS, DB_ITEMS ) ) }, ;
                          "Busqueda Avanzada de "+if( nEsArea == DB_FAMILIAS, "Insumo", "Item" ), ;
                          "Busca" )
   end
   oToolBar:AddButton( { || oWChld:End() }, "Cierra Tabla", "Cierra" )

   oReBar:InsertBand( oToolBar )
   
   for nFor:= 1 TO Len(oToolBar:aControls)
       oToolBar:aControls[nFor]:oCursor:= oCursor
   next

// Fin
