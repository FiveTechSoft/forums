Function FindRm(nKey,oSay)

?nKey

if nKey == VK_BACK
   if len(MEMVAR->FindKey) >= 1
      MEMVAR->FindKey := left(MEMVAR->FindKey,len(MEMVAR->FindKey)-1)
   end
else
   MEMVAR->FindKey += upper(chr(nKey))
end
oSay:SetText(MEMVAR->FindKey)

oSay:Refresh()
return .T.
