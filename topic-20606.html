Ahí va
Code (fw): Select all Collapse
   @ 0,0 XBROWSE oBrw ;
      HEADERS ahead;
      SIZE 260,320 PIXEL ;
      COLSIZES ,205,15,150,20,150,20,150,20,150,20,150,20,150,20,150,20 ;      
      JUSTIFY AL_CENTER;
      OF oDlg1 ARRAY aData FASTEDIT FOOTERS AUTOCOLS LINES CELL
      oBrw:bClrSelFocus    := { || { CLR_BLUE, nRGB( 230, 255, 230 ) } }
      oBrw:lColDividerComplete := .f.
      oBrw:lKineticBrw         := .f.
      oBrw:nTop    := oDlg1:nTop()
      oBrw:nBottom := oDlg1:nheight()
      oBrw:nRight  := oDlg1:nWidth()
      oBrw:Domingo:bClrStd          := { || { CLR_BLACK, nRgb(255,128,128) } }
      oBrw:Sabado:bClrStd          := { || { CLR_BLACK, nRgb(255,236,129) } }
      oBrw:nStretchCol := STRETCHCOL_LAST
      oBrw:Setfocus()
      AEval( oBrw:aCols, { |o| o:nEditType := 1 } )
      oBrw:Codigo:nEditType:=0
      oBrw:Nombre:nEditType:=0
      oBrw:T:nEditType:=0
      oBrw:nFreeze:=3
      for n:=4 to 17
      oBrw:aCols[n]:cEditPicture:="@!"
      next
      oBrw:Lunes:bClrStd      := { || Colores(oBrw,'Lunes')}
      oBrw:Martes:bClrStd     := { || Colores(oBrw,'Martes')}
      oBrw:Miercoles:bClrStd  := { || Colores(oBrw,'Miercoles')}
      oBrw:Jueves:bClrStd     := { || Colores(oBrw,'Jueves')}
      oBrw:Viernes:bClrStd    := { || Colores(oBrw,'Viernes')}
     WITH OBJECT oBrw
      // Make Groups
      :SetGroupHeader(s1, 4,  5, oBold )
      :SetGroupHeader(s2, 6,  7, oBold )
      :SetGroupHeader(s3, 8,  9, oBold )
      :SetGroupHeader(s4,10, 11, oBold )
      :SetGroupHeader(s5,12, 13, oBold )
      :SetGroupHeader(s6,14, 15, oBold )
      :SetGroupHeader(s7,16, 17, oBold )
      :lAllowColReGroup := .t.  // columns can be moved accross groups
      END
      oBrw:Codigo:lMergeVert   := .t.
      oBrw:Nombre:lMergeVert   := .t.
      oBrw:bKeyDown :={ | nKey | Teclas(nKey, oBrw,nret),oBrw:Refresh()}  
      oBrw:CreateFromCode()
      oBrw:Setfocus()
      titulo:=Titulo+s1+" AL "+s7
      barra1(oDlg1,oBrw,titulo)
