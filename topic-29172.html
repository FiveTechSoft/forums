y esta es mi configuracion para FWH utilizando el xmate.

Code (fw): Select all Collapse
[Environment]
Description 1=Entorno para xHarbour
Description 2=
Description 3=
PRG Compiler ver=
C compiler ver=
Subsystem=
Author=GoosFancito
Last update=05/02/2012

[Advanced]
Command types=1,1,1,1,1,1,1,1,1

[Harbour]
HB_INSTALL=z:\xh\xharbour
HB_COMMAND=%HB_BIN_INSTALL%\Harbour.Exe %PRG% /m /n0 /gc1 /w1 /es2 /i%HB_INC_INSTALL% /iz:\xh\fw\INCLUDE
Option prefix=/

[C]
C_INSTALL=z:\xh\bcc582
C_COMP_COMMAND=%C_BIN_INSTALL%\Bcc32.Exe -M -c -O2  -tW -v- -X %CRLF%-DHB_FM_STATISTICS_OFF %CRLF%-D__HARBOUR__%CRLF%-DHB_NO_DEFAULT_API_MACROS %CRLF%-DHB_NO_DEFAULT_STACK_MACROS %CRLF%-DHB_OS_WIN_32_USED %CRLF%-Iz:\xh\fw\INCLUDE %CRLF%-I%C_INC_INSTALL% -I%HB_INC_INSTALL% -n%HOME%\Obj %C%%CRLF%
C_LIB_COMMAND=%C_BIN_INSTALL%\TLib.Exe %LIB% /P512 @%RSP% , %LST%
C_LINK_COMMAND=%C_BIN_INSTALL%\iLink32.Exe -Gn -aa -Tpe -s @%LNK%
C_RC_COMMAND=z:\xh\PellesC\Bin\PORC.EXE  %RC%  /I z:\xh\PellesC\include /I z:\xh\PellesC\include\win
C_DLL_COMMAND=%C_BIN_INSTALL%\iLink32.Exe -Gn -aa -Tpd -M -L%C_LIB_INSTALL% @%LNK%
Option prefix=-
