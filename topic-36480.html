/*

FUNCTION cShortName( cLong )

   LOCAL   cShort  := Space( 164 ) + Chr( 0 )
   LOCAL   nBuffer := 164
   LOCAL   nLen

   nLen    := GetSPName( cLong, @cShort, nBuffer )

   cShort  := Left( cShort, nLen )

   RETURN( cShort )

// Windows Api
