BtnBmp: caption position

@ 1,2 BTNBMP oBtn ;
PROMPT "Text" TOP ADJUST OF oDlg ;
SIZE 100 , 100 ;
FILE "BTN.BMP" NOBORDER ;
FONT oFont1 UPDATE

           oBtn:SetColor(  CLR_HBLUE ,CLR_WHITE)
           oBtn:nClrText(  CLR_HRED )
           oBtn:nClrText(  CLR_HRED )
