FUNCTION ADJ_ZOOM1(oImage, hDC, cAction, oZSelect1)
DEFINE IMAGE oImg0 FILENAME c_path1 + cZImage
oImg0:lScroll := .F. // scrollbar only disabled !!
// Size-calculations
...
oImage := FILoadImg( c_path1 + cZImage )

hBmp = ResizeImg( oImage, nSizeW, nSizeH )

ELSE

ENDIF
DeleteObject( oImg0 )
