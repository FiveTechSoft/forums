For Harbour and FWH:
Code: Select all Collapse
function Main()

   MsgInfo( Country_Code( "yahoo.com" ) )

return nil

#pragma BEGINDUMP

#include <hbapi.h>
#include <GeoIP.h>

HB_FUNC( COUNTRY_CODE )
{ 
  GeoIP * gi = GeoIP_new( GEOIP_STANDARD ); 

  hb_retc( GeoIP_country_code_by_name( gi, hb_parc( 1 ) ) ); 
}

#pragma ENDDUMP
