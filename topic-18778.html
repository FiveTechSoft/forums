Code (fw): Select all Collapse
bFile := {||'c:\awt\kyocera.xls'}

oExcel := TOleAuto():New("Excel.Application")
oExcel:WorkBooks:Open(eval(bFile))
oSheet := oExcel:Get("ActiveSheet")
oExcel:Visible := FALSE
nRows := oSheet:UsedRange:Rows:Count()
do while nRowInc <= nRows
   CursorWait()
   cCode   := oSheet:Cells(nRowInc,"A"):Value
   cDesc   := oSheet:Cells(nRowInc,"B"):Value
   cGroup  := 15
   nCost   := oSheet:Cells(nRowInc,"F"):Value
   nSell   := oSheet:Cells(nRowInc,"G"):Value
   cSpplr  := 'KYO01  '
   if (valtype(cGroup) = 'C')
       cGroup := str(cGroup,3)
       oKyocera:group := LEFT(ALLTRIM(cGroup) + space(3),3)
   endif
   if (valtype(cGroup) = 'N')
       cGroup := str(cGroup,3)
       oKyocera:group := LEFT(ALLTRIM(cGroup) + space(3),3)
   endif
   if (valtype(cCode) = 'C')
       oKyocera:code := cCode
   endif
   if (valtype(cCode) = 'N')
      oKyocera:code := LTRIM(str(cCode,15))
   endif
   if ! empty(cDesc)
      oKyocera:desc := cDesc
   endif
   if ! empty(nCost)
      oKyocera:cost := nCost
   endif
   if ! empty(nSell)
      oKyocera:sell := nSell
   endif
   oKyocera:append()
   nRowInc++
enddo
oExcel = nil
CursorArrow()
return(nil)
//---------------------------------------------------------------------------//

Hope this helps
