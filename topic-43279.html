   local cClrBack

   oBrw:lFooter           := lFoot

   //oBrw:lColDividerComplete := .F.
   //oBrw:lRowDividerComplete := .F.

   oBrw:lFullGrid         := .F.

   oBrw:nColDividerStyle  := LINESTYLE_NOLINES //LIGHTGRAY // LINESTYLE_NOLINES

   cClrBack = Eval( oBrw:bClrSelFocus )[ 2 ]
