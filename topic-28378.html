STATIC FUNCTION Reg_Cap()
 Local nNumMov,cCtaSN

 IF lAdd  // SI AGREGA NUEVO REGISTRO
     (S13)->(DbGoBottom())

     <strong>nNumMov  += nNumMov     //(S13)->MV_NUMV + 1 // Incrementa nuevos registros de 1 a 99999</strong>

     (S13)->(DbSkip())
     (S13)->(DbAppend())

     (S13)->FECALTA := DATE()
     (S13)->REGHORA := AMPM(TIME())
     (S13)->NICKUSUA:= cNombUsua
     (S13)->NOMBUSUA:= cNombre
 ELSE
     <strong>nNumMov := (S13)->MV_NUMV</strong>
 ENDIF

 IF !OCUPADO(S13) // Bloquea S13:=MOVTxMxA.DBF
   <strong>(S13)->MV_NUMV := nNumMov // Incrementa registros</strong>
   (S13)->MV_TIPO := cTipo
   (S13)->MV_NUMP := nNumPol
   (S13)->MV_FHAM := dFecha


   (S13)->(DbCommit())
   (S13)->(DbUnlock()) // Desbloquea
 ENDIF
