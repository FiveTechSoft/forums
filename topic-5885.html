This is a working sample:

Code: Select all Collapse
#include "Fivewin.ch"


FUNCTION MAIN()

    LOCAL oDlg

    LOCAL oTabs

    DEFINE DIALOG oDlg

    @ 0, 0 BUTTON "Change";
           ACTION ( oTabs:aPrompts[ 2 ] := "Changed",;
                    oTabs:Refresh() )

    @ 2, 0 TABS oTabs;
           PROMPTS "First", "Second", "Third"

    ACTIVATE DIALOG oDlg;
             CENTER

    RETURN NIL
