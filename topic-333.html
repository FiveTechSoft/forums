sockaddr:s_family:=AF_INET

cTemp:=sockaddr:cBuffer
rc:=bind(SocketNum,@cTemp,sockaddr_size)
sockaddr:=cTemp
if rc<>0
MsgAlert("Fallo en BIND")
SocketNum:=-1
return nil
endif
stMreq2:i_interf:=inet_addr(m_HostSet)

stMreq2:=cTemp
if rc<>0
MsgAlert("Fallo en SETSOCKOPT-IF")

return nil
endif
