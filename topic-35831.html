I did not find any problem when I tested this sample:
You may try too.
Code (fw): Select all Collapse
#include "fivewin.ch"

function Main()

   local oCn, oRs, cSql

   oCn   := FW_DemoDB( "ADO" )

   if oCn == nil
      ? "fail"
      return nil
   endif

   ? "connected"

TEXT INTO cSql
 CREATE TABLE IF NOT EXISTS cuentas (
 CUE_CUE DECIMAL(10,0) NOT NULL DEFAULT 0,
 CUE_DES CHAR(35) NOT NULL DEFAULT '',
 PRIMARY KEY(CUE_CUE)
 )
 ENGINE = InnoDB
ENDTEXT

   oCn:Execute( cSql )

   do while MsgYesNo( "Open and edit?" )
      oRs   := FW_OpenRecordSet( oCn, "cuentas" )
      XBROWSER oRs FASTEDIT
      oRs:Close()
   enddo

   oCn:Execute( "DROP TABLE `cuentas`" )

   oCn:Close()

return nil
