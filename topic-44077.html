function Main()

    local oDlg 

    DEFINE DIALOG oDlg RESOURCE "test"

    ACTIVATE DIALOG oDlg CENTERED ;
       ON INIT EnumChildWindows( oDlg:hWnd,;
          { | hChild | MsgInfo( GetClassName( hChild ), GetDlgCtrlID( hChild ) ) } )
