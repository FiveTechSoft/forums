Mira se és esto que necesitas: Cambia para español e prueba.

Code (fw): Select all Collapse
#Include "FiveWin.Ch"

REQUEST HB_LANG_PT, HB_CODEPAGE_PTISO, HB_CODEPAGE_PT850 //-> Portuguese

FUNCTION TESTUTF8()

   LOCAL cUTF8, cStrUtf8, cAsc, cUTF8Fin

   cStrUtf8 := "Texto em caracteres UTF-8: (dígrafo BR)'ção',(dígrafo espanhol-enhe)'ñ',(trema)'Ü',(ordinais)'ªº',(ligamento s+z alemão)'ß'."

   cUTF8 := HB_STRTOUTF8( cStrUtf8 ) // functon in XHARBOUR

   cAsc  := HB_UTF8TOSTR( cUtf8 )

   cUTF8Fin := cAsc

   ? cUTF8Fin

RETURN NIL
