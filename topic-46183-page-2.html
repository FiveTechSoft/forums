    if lines->invoice_id == cInvoiceId
    select products

function GenerateID()
    local cChars := "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ"
    local cNewId := ""
    local i
    for i := 1 to 8
    cNewId += SubStr( cChars, Int( hb_Random() * Len(cChars) ) + 1, 1 )
    next
return cNewId
