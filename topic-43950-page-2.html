    MEMOWRIT( "utils\hbmk2\hbmk2.prg", cTxt, .F. )

STATIC FUNCTION TOOCTAL( cTxt )

    LOCAL cRes := ""

    LOCAL nVal, cVal

    LOCAL lUtf := .F.

    LOCAL i

    FOR i = 1 TO LEN( cTxt )
        IF ASC( SUBSTR( cTxt, i, 1 ) ) <= 127
            cRes += SUBSTR( cTxt, i, 1 )
        ELSE
            nVal = ASC( SUBSTR( cTxt, i, 1 ) )

            cVal = ""

            WHILE nVal / 8 > 0
                cVal = NTRIM( INT( nVal % 8 ) ) + cVal
                nVal = INT( nVal / 8 )
            ENDDO

            cRes += "\" + cVal

            lUtf = .T.
        ENDIF
    NEXT

    RETURN IF( lUtf, cRes, NIL )
