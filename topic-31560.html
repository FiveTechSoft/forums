Este ejemplo está en samples en un dialogo y se ve bastante mal, pero no así en una ventana. Bastante parecido a lo que muestra Marco. ( Solo el tree )

Code (fw): Select all Collapse
#include "FiveWin.ch"

//----------------------------------------------------------------------------//
Function Main()

   Local oWnd
    Local oTree 

   Define Window oWnd

    @ 0, 0 Treeview oTree Of oWnd SIZE oWnd : nWidth, oWnd : nHeight CHECKBOXES

   Activate Window oWnd Centered On Init BuildTree( oTree ) 
   MsgInfo( oTree : aItems[ 1 ] : GetCheck() ) 
    Return Nil 

//----------------------------------------------------------------------------//
Function BuildTree( oTree ) 

    Local oMenu       := Array( 2 )
    Local oSubMenu := Array( 10 ) 

   oMenu[ 1 ] := oTree : Add( "Principal" ) 
      oSubMenu[ 1 ] := oMenu[ 1 ] : Add( "Imprimir..." )
      oSubMenu[ 1 ] : SetCheck( .t. ) 

   oMenu[ 2 ] := oTree : Add( "Proyectos" ) 
      oSubMenu[ 2 ] := oMenu[ 2 ] : Add( "Definir Proyectos" ) 
      oSubmenu[ 3 ] := oMenu[ 2 ] : Add( "Actualización datos" ) 
      oSubmenu[ 4 ] := oMenu[ 2 ] : Add( "Actualización datos" ) 
      oSubmenu[ 5 ] := oMenu[ 2 ] : Add( "Actualización datos" ) 
      oSubmenu[ 6 ] := oMenu[ 2 ] : Add( "Actualización datos" ) 
      oSubmenu[ 7 ] := oMenu[ 2 ] : Add( "Actualización datos" ) 

   oTree : Expand() 

return nil
