El codigo si esta dentro de los fuentes de harbour, al parecer es muy facil usarlo
busca en harbour\contrib\hbzebra\

hay que enlazar las librerias
hbzebra.lib

dejo el link del exe

http://www.sitasoft.net/fivewin/samples/barcode.zip

Code (fw): Select all Collapse
#include "fivewin.ch"

function main()

   local oWnd

   define window oWnd
   
   activate window oWnd;
            on paint( BarCode( hDC ) )
   
return nil

function BarCode( hDC )
   
   local hZebra
   local cCode := "testing from fivewin"
   local hBrush := CreateSolidBrush( 0 )

   hZebra := hb_zebra_create_pdf417( cCode, NIL )

   hb_zebra_draw( hZebra, {| x, y, w, h | FillRect( hDC, { y, x, y +  h, x + w }, hBrush ) }, 5, 5, 1, 4 )
   
   deleteobject( hBrush )
   hb_zebra_destroy( hZebra )
   
return nil
