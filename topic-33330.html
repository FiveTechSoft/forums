Code (fw): Select all Collapse
   DEFINE DIALOG oDlg

   @ 2, 2 GET oGet VAR cTest ACTION MsgInfo( "ok" ) COLOR 0, CLR_WHITE ;
      BITMAP "bdate"

   oGet:lBtnTransparent := .t.

   ACTIVATE DIALOG oDlg CENTERED ;
      ON INIT ( oGet:oBtn:lBoxSelect := .f., ;
                oGet:oBtn:lFlatStyle := .t. )
