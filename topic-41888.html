function Main()

   local oPrn

   PRINT oPrn PREVIEW

   PAGE

   @ 3,3 PRINT TO oPrn IMAGE "..\bitmaps\olga1.jpg" ;
      SIZE 3.5,4.5 CM

   @ 3,9 PRINT TO oPrn ;
      TEXT "This text is displayed as QR code" ;
      AS BARCODE TYPE "QR" ;
      SIZE 3,3 CM

   ENDPAGE

   ENDPRINT

return nil
