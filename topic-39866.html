yes I use it with both, I have not found any problem. I use this config

REQUEST DBFCDX
rddsetdefault( "dbfcdx" )
REQUEST DBFFPT
SET DBFLOCKSCHEME TO 2 // clp53
REQUEST HB_LANG_IT
HB_LANGSELECT( 'IT' )
#ifndef XHARBOUR
request HB_CODEPAGE_ITWIN
set( _SET_CODEPAGE, "ITWIN" )
#endif
