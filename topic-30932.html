function Rinfresca_oFrm()

   oImage:hBitmap = ResizeBmp( oImage:cargo, oPanel:nWidth, oPanel:nHeight, .T. )    
   oImage:SetSize( oPanel:nWidth, oPanel:nHeight )  
   oImage:Center()
   oImage:Refresh()
