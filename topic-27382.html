function Main()

        nSize := ROUND( FSIZE( "C:\prova.bmp" ) / 1024, 0 )  
        
        DEFINE WINDOW oWnd FROM 1, 1 TO 20, 50 TITLE "Testing"
        
        @ 10, 10 SAY ALLTRIM( STR( nSize ) ) + " KB" OF oWnd SIZE 60, 25 PIXEL COLOR "N/W*"
               
        ACTIVATE WINDOW oWnd                
return nil
