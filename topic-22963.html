Tip: MakeFiles, double substitution within macros
Posted: Sun Nov 27, 2011 03:56 PM
This is a recent tip I discovered using Borland and Microsoft make utilities. Lets consider the files list:

Code (fw): Select all Collapse
PRG =          \
.\APPLICAT.PRG \
.\CODE.PRG     \
.\DATABASE.PRG \
.\DEBUG.PRG    \
.\DESIGN.PRG   \
.\INDEX.PRG    \
.\INSPECT.PRG  \
.\MENU.PRG     \
.\OPTIONS.PRG  \
.\PROGRAM.PRG  \
.\PROJECT.PRG  \
.\RECORD.PRG   \
.\REP.PRG      \
.\REPORT.PRG   \
.\RESOURCE.PRG \
.\TOOLBAR.PRG  \
.\TOOLS.PRG    \
.\VISUAL.PRG


The macro:
OBJ = $(PRG:.PRG=.OBJ)
creates a new macro OBJ where .PRG ocurrences are replaced with .OBJ. The trick now is to create a new macro from this one and perform another substitution:

OBJS = $(OBJ:.\=.\obj\)
replacing ".\" with ".\obj\" in the list.

Now, we can use $(OBJS) to set the EXE dependencies and also to link the EXE:
VISUALFW.EXE : $(OBJS) VISUALFW.RES ..\lib\FiveH.lib ..\lib\FiveHC.lib
...
echo $(OBJS), + >> b32.bc

Here you have the complete makefile visualfw.mak
Code (fw): Select all Collapse
#VisualFW Borland make, (c) FiveTech Software 2005-2011

HBDIR=c:\harbour
BCDIR=c:\bcc582
FWDIR=c:\fwh

#change these paths as needed
.path.OBJ = .\obj
.path.PRG = .\
.path.CH  = $(FWDIR)\include;$(HBDIR)\include
.path.C   = .\
.path.rc  = .\

#important: Use Uppercase for filenames extensions!

PRG =          \
.\APPLICAT.PRG \
.\CODE.PRG     \
.\DATABASE.PRG \
.\DEBUG.PRG    \
.\DESIGN.PRG   \
.\INDEX.PRG    \
.\INSPECT.PRG  \
.\MENU.PRG     \
.\OPTIONS.PRG  \
.\PROGRAM.PRG  \
.\PROJECT.PRG  \
.\RECORD.PRG   \
.\REP.PRG      \
.\REPORT.PRG   \
.\RESOURCE.PRG \
.\TOOLBAR.PRG  \
.\TOOLS.PRG    \
.\VISUAL.PRG               

OBJ = $(PRG:.PRG=.OBJ)
OBJS = $(OBJ:.\=.\obj\)

PROJECT    : VISUALFW.EXE 

VISUALFW.EXE : $(OBJS) VISUALFW.RES ..\lib\FiveH.lib ..\lib\FiveHC.lib
   echo off
   echo $(BCDIR)\lib\c0w32.obj + > b32.bc
   echo $(OBJS), + >> b32.bc
   echo visualfw.exe, + >> b32.bc
   echo visualfw.map, + >> b32.bc
   echo $(FWDIR)\lib\FiveH.lib $(FWDIR)\lib\FiveHC.lib + >> b32.bc

.PRG.OBJ:
  $(HBDIR)\bin\harbour $< /N /W /Oobj\ /I$(FWDIR)\include;$(HBDIR)\include
  $(BCDIR)\bin\bcc32 -c -tWM -I$(HBDIR)\include -oobj\$& obj\$&.c

.C.OBJ:
  echo -c -tWM -D__HARBOUR__ > tmp
  echo -I$(HBDIR)\include;$(FWDIR)\include >> tmp
  $(BCDIR)\bin\bcc32 -oobj\$& @tmp $&.c
  del tmp

VISUALFW.RES : VISUALFW.RC
  $(BCDIR)\bin\brc32.exe -r -I$(BCDIR)\include visualfw.rc
