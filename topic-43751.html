LOCAL cValue, nPos
cValue = cToSearch

nPos = AT(cBeginDelim, cValue)
IF nPos = 0
    RETURN ""
ENDIF

cValue = SUBS(cValue, nPos + LEN(cBeginDelim))

IF nPos = 0
    RETURN ""
ENDIF

cValue = LEFT(cValue,nPos - 1)
