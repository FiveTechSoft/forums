no problem

set all together

Code (fw): Select all Collapse
METHOD Refresh() CLASS TMsgItem

   local cMsg

   if ::bMsg != nil
      cMsg = cValToChar( Eval( ::bMsg ) )
      if cMsg != ::cMsg .or. ::lActive != ::lWasActive
         ::cMsg = cMsg
         if IsWindowVisible( ::oMsgBar:hWnd )
            ::Paint()
         endif   
         ::lWasActive = ::lActive
      endif
   endif

   DeleteObject( ::hBack)
   ::hBack := nil
   ::oMsgBar:Refresh()

return nil
