include "fivewin.ch"

define olFolderContacts 10

ANNOUNCE RDDSYS

FUNCTION MAIN( cNomeFile )

LOCAL oOutlook := CREATEOBJECT( &quot;Outlook.Application&quot; )
LOCAL oNameSpace := oOutlook:GetNameSpace(&quot;MAPI&quot;)
LOCAL oContacts := oNameSpace:GetDefaultFolder( olFolderContacts )

LOCAL i
LOCAL aStruct := {}
AADD( aStruct , {  &quot;NOME&quot;      ,     &quot;C&quot;   ,     50   ,  0  } )
AADD( aStruct , {  &quot;INDI&quot;      ,     &quot;C&quot;   ,     50   ,  0  } )
AADD( aStruct , {  &quot;CAP&quot;       ,     &quot;C&quot;   ,      5   ,  0  } )
AADD( aStruct , {  &quot;CITTA&quot;     ,     &quot;C&quot;   ,     50   ,  0  } )
AADD( aStruct , {  &quot;STATO&quot;     ,     &quot;C&quot;   ,     50   ,  0  } )
AADD( aStruct , {  &quot;STATO2&quot;    ,     &quot;C&quot;   ,     50   ,  0  } )
AADD( aStruct , {  &quot;TELEFONO&quot;  ,     &quot;C&quot;   ,     50   ,  0  } )
AADD( aStruct , {  &quot;EMAIL&quot;     ,     &quot;C&quot;   ,     50   ,  0  } )
AADD( aStruct , {  &quot;NOTE&quot;      ,     &quot;M&quot;   ,     10   ,  0  } )

DbCreate( cNomeFile , aStruct, &quot;DBFCDX&quot; )


SET DATE BRITISH
USE &amp;cNomeFIle

FOR i = 1 TO oContacts:Items:Count
    APPEND BLANK
    REPLACE field-&gt;nome      WITH  oContacts:Items[ i ]:FullName
    REPLACE field-&gt;indi      WITH  oContacts:Items[ i ]:BusinessAddressStreet
    REPLACE field-&gt;cap       WITH  oContacts:Items[ i ]:BusinessAddressPostalCode
    REPLACE field-&gt;citta     WITH  oContacts:Items[ i ]:BusinessAddressCity
    REPLACE field-&gt;stato     WITH  oContacts:Items[ i ]:BusinessAddressState
    REPLACE field-&gt;stato2    WITH  oContacts:Items[ i ]:BusinessAddressCountry
    REPLACE field-&gt;telefono  WITH  oContacts:Items[ i ]:BusinessTelephoneNumber
    REPLACE field-&gt;email     WITH  oContacts:Items[ i ]:Email1Address
    REPLACE field-&gt;note      WITH  oContacts:Items[ i ]:Body

NEXT

oOutlook:Quit()

RETURN NIL

INIT PROCEDURE RddInit
