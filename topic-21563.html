Maybe you can create a block using macro compiler.
cSFD := "document:forms:item(0):search"
bSFD := &( "{ | cValue | IE:"+cSFD+":Value = cValue }" )
eval( bSFD, cPARTS )

Or maybe
cSFD := "document:forms:item(0):search"
bSFD := &( "{ || IE:"+cSFD+":Value = cPARTS }" )
eval( bSFD )
