DEFINE DIALOG oDlg RESOURCE "CEM44_LIQ_FEC" OF oWnd ICON "ICONO_PPAL";

   REDEFINE BTNBMP Btn2 ID 201 OF oDlg RESOURCE "CANCELAR" LEFT;
