function CalcTextHeight( hDC, cText, oFont, nWidth )

   local oWnd, nHt

   DEFAULT hDC := GetDC( 0 )
   if HB_ISOBJECT( hDC )
      oWnd     := hDC
      hDC      := oWnd:GetDC()
      DEFAULT oFont := oWnd:oFont, nWidth := oWnd:nWidth
   else
      DEFAULT nWidth := 800
   endif

   nHt   := DrawTextEx( hDC, cText, { 0, 0, 1000, nWidth }, 1040, ;
               If( oFont == nil, nil, oFont:hFont ),, @nWidth )

   if oWnd == nil
      if PValue( 1 ) == nil
         ReleaseDC( 0, hDC )
      endif
   else
      oWnd:ReleaseDC()
   endif

return nHt
