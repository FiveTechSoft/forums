Code (fw): Select all Collapse
#include "fivewin.ch"

function Main()

   // Keep this in the Main() function
   WITH OBJECT TGet()
      :lDisColors    := .f.
      :lClrFocus     := .t.
      :bColorBlock   := <|oGet|
         WITH OBJECT oGet
            if IsWindowEnabled( :hWnd ) .and. !:lReadOnly
               if :HasFocus
                  return { CLR_BLACK, :nClrFocus }
               else
                  return { CLR_HRED, CLR_WHITE }
               endif
            endif
          END
          return { CLR_BLACK, CLR_HGRAY }
          >
   END
   //  end


   USE CUSTOMER
   XEDIT()

return nil
