This is a sample with VB:

Set session = CreateObject("Notes.NotesSession")
Set db = session.GetDatabase("", "")
Set doc = db.CreateDocument()

doc.Form = "Main Topic"
doc.Subject = Form1.Text3.Text
doc.Body = Form1.Text2.Text
Call doc.Save(True, False)
