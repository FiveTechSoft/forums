STATIC FUNCTION Cli_Bar( o, v )
------------------------------
LOCAL oDummy

         ACTION Cli_Del( o, v )

              oDummy:Disable()

ShowKeyboard()

------------------------------
STATIC FUNCTION Cli_New( o, v )
------------------------------

IF v[ O_RECNO ] &gt; 0 .AND. ! MsgYesNo( 'Nuevo registro?', 'Sistema' )
   o[ O_FIRST    ]:SetFocus()
   RETU NIL
ENDIF

Cli_IniVar    ( o, v )
Cli_RefreshVar( o, v )

RETU NIL

-------------------------------
STATIC FUNCTION Cli_Find( o, v )
-------------------------------
LOCAL cAlias := aDbf[ DBF_CUSTOMER ]
LOCAL nRecno := Sel_Client()

IF nRecno &gt; 0

   ( cAlias )-&gt;( DbGoto( nRecno ) )

   v[ O_FIRST    ] := ( cAlias )-&gt;first
   v[ O_LAST     ] := ( cAlias )-&gt;last
   v[ O_STREET   ] := ( cAlias )-&gt;street
   v[ O_CITY     ] := ( cAlias )-&gt;city
   v[ O_STATE    ] := ( cAlias )-&gt;state
   v[ O_ZIP      ] := ( cAlias )-&gt;zip
   v[ O_HIREDATE ] := ( cAlias )-&gt;hiredate
   v[ O_MARRIED  ] := ( cAlias )-&gt;married
   v[ O_AGE      ] := ( cAlias )-&gt;age
   v[ O_SALARY   ] := ( cAlias )-&gt;salary

   v[ O_RECNO    ] := nRecno

   Cli_RefreshVar( o, v )

  ELSE

   o[ O_FIRST    ]:SetFocus()

ENDIF

RETU NIL

-------------------------------
STATIC FUNCTION Cli_Save( o, v )
-------------------------------
LOCAL lOk := .f.
LOCAL cAlias := aDbf[ DBF_CUSTOMER ]

IF ! MsgYesNo( 'Actualizar datos ?', 'Sistema' )
   o[ O_FIRST    ]:SetFocus()
   RETU NIL
ENDIF

IF v[ O_RECNO ] == 0

      ( cAlias )-&gt;( DbAppend() )
   IF ( cAlias )-&gt;( NetErr()  )
      MsgAlert( 'Error a¤adiendo registro', 'Sistema' )
      RETU NIL
   ENDIF

ENDIF

( cAlias )-&gt;first     := v[ O_FIRST    ]
( cAlias )-&gt;last      := v[ O_LAST     ]
( cAlias )-&gt;street    := v[ O_STREET   ]
( cAlias )-&gt;city      := v[ O_CITY     ]
( cAlias )-&gt;state     := v[ O_STATE    ]
( cAlias )-&gt;zip       := v[ O_ZIP      ]
( cAlias )-&gt;hiredate  := v[ O_HIREDATE ]
( cAlias )-&gt;married   := v[ O_MARRIED  ]
( cAlias )-&gt;age       := v[ O_AGE      ]
( cAlias )-&gt;salary    := v[ O_SALARY   ]

Cli_IniVar    ( o, v )
Cli_RefreshVar( o, v )

RETU NIL

------------------------------
STATIC FUNCTION Cli_Del( o, v )
------------------------------
LOCAL lOk := .f.
LOCAL cAlias := aDbf[ DBF_CUSTOMER ]

IF v[ O_RECNO ] == 0
   Cli_IniVar    ( o, v )
   Cli_RefreshVar( o, v )
   RETU NIL
ENDIF

IF ! MsgYesNo( 'Eliminar registro ?', 'Sistema' )
   RETU NIL
ENDIF

( cAlias )-&gt;( DbDelete() )

Cli_IniVar    ( o, v )
Cli_RefreshVar( o, v )

RETU NIL

---------------------------
STATIC FUNCTION Sel_Client()
---------------------------
LOCAL oDlg, oFont, oLbx
LOCAL nRecno := 0
LOCAL cAlias := aDbf[ DBF_CUSTOMER ]

HideKeyboard()

DEFINE FONT oFont NAME "Arial" SIZE 0, -12

DEFINE DIALOG oDlg RESOURCE "Sel_Client"

    REDEFINE LISTBOX oLbx ;
       FIELDS (cAlias)-&gt;first, (cAlias)-&gt;last ;
       HEADERS "First", "Last"  ;
       ID 500 OF oDlg ALIAS cAlias

       oLbx:aColSizes  := { 100, 100 }
       oLbx:blDblClick := {|| ( nRecno := (cAlias)-&gt;( Recno() ),;
                                oDlg:End() ) }

   REDEFINE BUTTON  ID 100 OF oDlg ACTION oDlg:End()
   REDEFINE BUTTON  ID 110 OF oDlg ;
            ACTION ( nRecno := (cAlias)-&gt;( Recno() ) , oDlg:End() )

ACTIVATE DIALOG oDlg CENTERED

ShowKeyboard()

RETU nRecno
