#include "FiveWin.ch"

FUNCTION MAIN()

   // Create a Word application object
   LOCAL oWord := WinWordObj():New()

   // Convert 2 centimeters to points
   LOCAL nPts := oWord:CentimetersToPoints(2)

   // Quit Word
   oWord:Quit()

RETURN
