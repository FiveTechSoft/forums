Another TMenu bug. In the following sample, please open Test menu and you will see that the mdichild window title is missing from the last menu item.

Code (fw): Select all Collapse
#include "Fivewin.ch"


FUNCTION MAIN()

    LOCAL oWnd, oMenu

    MENU oMenu 2007
        MENUITEM "Test"

        MENU
            MENUITEM "Test"
            MENUITEM "Test"
            MENUITEM "Test"
        ENDMENU
    ENDMENU

    DEFINE WINDOW oWnd MDI;
           TITLE "MDI Test";
           MENU oMenu

    ACTIVATE WINDOW oWnd;
             ON INIT CREATECHILD( oWnd )

    RETURN NIL


STATIC FUNCTION CREATECHILD( oMdi )

    LOCAL oWnd, oMenu

    MENU oMenu 2007
        MENUITEM "Test"

        MENU
            MENUITEM "Test"
            MENUITEM "Test"
            MENUITEM "Test"
        ENDMENU
    ENDMENU

    DEFINE WINDOW oWnd MDICHILD OF oMdi;
           TITLE "MDI Child Test";
           MENU oMenu

    ACTIVATE WINDOW oWnd;
             VALID !GETKEYSTATE( VK_ESCAPE )

    RETURN NIL
