STATIC FUNCTION REEMPLACE (oDoc, cViejo, cNuevo)
LOCAL oSelection, oTexto, oFind
oSelection := oDoc
oTexto := oSelection:Range
oFind  := oTexto:Find
oFind:= cViejo
oFind:=.T.
oFind:= INT(1)
oFind:=.f.
oFind:MatchCase:=.f.
oFind:Execute()
while oFind:Found
  oTexto:=cNuevo
  oFind:Execute()
enddo
RETURN nil
