I suggest this code:
Code (fw): Select all Collapse
   local aData, cData

   USE CUSTOMER
   aData := FW_DbfToArray()
   CLOSE CUSTOMER

   cData := FW_ValToExp( aData )
   cData := StrTran( cData, "},{", "} ;" + CRLF + ",  {" )
   cData := StrTran( cData, "{{", "function customer_array()" + CRLF + "return ;" + CRLF + "{  {" )
   cData := Left( cData, Len( cData ) - 1 ) + " ;" + CRLF + "}" + CRLF

   FW_MEMOEDIT( cData )
