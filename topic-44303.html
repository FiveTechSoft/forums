Function Blinker(oCtrl, oTmr)

Static lOn := .f.

oTmr:DeActivate()

if lOn

oCtrl:Enable() //Hide()

else

oCtrl:Disable() //Show()

endif

lOn := !lOn

oTmr:Activate()

oCtrl:Refresh()

Return(Nil)

Static Function Fnct_Blinker(oSay1)

nCountBlink++



if (nCountBlink/2) &lt;&gt; int(nCountBlink/2)

    oSay1:Hide()

 else

    oSay1:Show()

endif



oSay1:Refresh()
