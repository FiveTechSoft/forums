   local nRec, aRows

   WITH OBJECT oBrw:oDbf
      if nKey == VK_UP
        :Skip( -1 )
        if :Bof()
           :GoTop()
        else
           nRec     := :RecNo()
           aRows    := :DbfToArray( cList, nil, nil, 2 )
           :GoTo( nRec )
