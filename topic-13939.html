Code (fw): Select all Collapse
#define KEYEVENTF_KEYUP 2
//----------------------------------------------------------------------------//
Function ShowMenu( oM, nNiv )
                        
    Local cMv
    Local nChr
    Local nItem
    Local oLItM
    Local cMov := ""
    
    Default oM   := WndMain():oMenu
    Default nNiv :=0
    oLItM := oM : LastItem()
    For nItem := 1 to Len( oM : aItems )
        If( ValType( oM : aItems[ nItem ] : bAction ) == 'O' )
            cMv := ShowMenu( oM : aItems[ nItem ] : bAction, nNiv + 1 )
            If( Right( cMv, 1 ) == "x" )
                cMov += Chr( If( nNiv =0, VK_DOWN, VK_RIGHT ) ) + cMv
                If( nNiv == 0 )
                    cMov := Left( cMov, Len( cMov ) - 1 )
                    keybd_event( VK_MENU, 0, 0, 0 )
                    keybd_event( VK_MENU, 0, KEYEVENTF_KEYUP, 0 )
                    For nChr := 1 to Len( cMov )
                        keybd_event( Asc( SubStr( cMov, nChr, 1 ) ), 0, 0, 0 )
                        keybd_event( Asc( SubStr( cMov, nChr, 1 ) ), 0, KEYEVENTF_KEYUP, 0 )
                    Next i
                    Return( .t. )
                End
                Return cMov
            End
        Else
            If( oM : aItems[ nItem ] : nId == oLItM : nId )
                Return cMov + "x"
            End
        End
        If( !Empt( oM : aItems[ nItem ] : cPrompt ) ) // se não for um separador
            cMov +=Chr( If( nNiv == 0, VK_RIGHT, VK_DOWN ) )
        End
    Next nItem
    Return if( nNiv ==0, .f., "" )
            
    
DLL32 FUNCTION keybd_event ( bVk as _INT,bScan as _INT, dwFlags as LONG, dwExtraInfo as LONG ) AS LONG ;
    PASCAL LIB "user32.dll"

FUNCTION BldMenu()
----------------------------------------------------------------
LOCAL oMainMenu, oMnuItem, lRight, cValue, cMnuAlias, i:= 1, nRec, fAction, cFileBmp, cResName
LOCAL cQry, oQry, nL, nEnd

IF ! oApp:lMySql        &amp;&amp; Si es DBF

    IF !AbreDbf( @cMnuAlias, oSystem:csysmenu, 1 )
            oWnd:PostMsg(WM_CLOSE)
    ENDIF
    nRec := (cMnuAlias)-&gt;( AdsKeyCount( ,,ADS_RESPECTSCOPES ) )

     MENU oApp:oMainMenu 2010

    FOR i := 1 To nRec

         IF (cMnuAlias)-&gt;FLDSEPARAT
            SEPARATOR
        ENDIF

         IF (cMnuAlias)-&gt;FLDVIRTKEY == 0
            MENUITEM oMnuItem PROMPT OemToAnsi(ALLTRIM((cMnuAlias)-&gt;FLDTXTOPT))  ;
            MESSAGE OemToAnsi(ALLTRIM((cMnuAlias)-&gt;FLDMESSAGE))
        ELSE
            MENUITEM oMnuItem PROMPT OemToAnsi(ALLTRIM((cMnuAlias)-&gt;FLDTXTOPT))  ;
             MESSAGE OemToAnsi(ALLTRIM((cMnuAlias)-&gt;FLDMESSAGE))                             ;
            ACCELERATOR (cMnuAlias)-&gt;FLDNSTATE, (cMnuAlias)-&gt;FLDVIRTKEY
        ENDIF

        oMnuItem:lChecked := (cMnuAlias)-&gt;FLDCHECK

        oMnuItem:lActive := ( lRight:= Eval( {||&amp;(ALLTRIM((cMnuAlias)-&gt;FLDCONDI))} ) )

        IF ! EMPTY( fAction := ALLTRIM((cMnuAlias)-&gt;FLDACTION) )
            oMnuItem:bAction := &amp;( fAction )
        ENDIF

        IF FILE( cFileBmp := oGraph:cmnuimgpath + ALLTRIM((cMnuAlias)-&gt;FLDFILEBMP) )
            oMnuItem:hBitmap:= ReadBitmap( 0, cFileBmp )
        ENDIF
        IF ! EMPTY( cResName := ALLTRIM((cMnuAlias)-&gt;FLDRESBMP) )
            oMnuItem:hBitmap:= LoadBitmap( GetResources(), cResName )
        ENDIF

        IF ! EMPTY( ALLTRIM((cMnuAlias)-&gt;FLDSYS) )
            AAdd( oApp:aObjSys, oMnuItem )
        ENDIF

         cValue := ALLTRIM((cMnuAlias)-&gt;CNEXT)

        nEnd := (cMnuAlias)-&gt;NLOOP
        FOR nL := 1 TO nEnd

            DO CASE
                CASE Left(cValue,1) = 'M'
                    MenuBegin( ,,, TRUE )
                CASE Left(cValue,1) = 'E'
                    MenuEnd()
            END CASE

         NEXT nL

        (cMnuAlias)-&gt;(dbSkip())
    NEXT i

     MENUITEM 'A&amp;yuda' HELP
        MENU
            MENUITEM '&amp;Help' ACTION HelpIndex() RESOURCE 'AYUDA1'
             MENUITEM '&amp;Acerca de...' ACTION fAbout()
        ENDMENU
    ENDMENU

     (cMnuAlias)-&gt;(dbCloseArea())

ELSE        &amp;&amp; MySql

    cQry := &quot;SELECT * FROM &quot; + oSystem:csysmenu + &quot; WHERE lActive ORDER BY fldnivel ASC&quot;

     IF !AbreDbf( @oQry, oSystem:csysmenu, 1,, oRDD:rdbms, cQry )
        MsgInfo('Error de apertura')
        oWnd:PostMsg(WM_CLOSE)
    ENDIF

    nRec := oQry:LastRec()

    oApp:oMainMenu := MenuBegin( ,,, TRUE )

            FOR i := 1 To nRec

           IF ( oQry:FLDSEPARAT )
              oMnuItem:= MenuAddItem()
            ENDIF

        IF FILE( cFileBmp := oGraph:cmnuimgpath + ALLTRIM(oQry:FLDFILEBMP) )
        ENDIF
        IF ! EMPTY( cResName := ALLTRIM(oQry:FLDRESBMP) )
        ENDIF
        IF ! EMPTY( ALLTRIM(oQry:FLDACTION) )
            fAction := &amp;( ALLTRIM(oQry:FLDACTION) )
        ENDIF

        &amp;&amp; Crea nuevo Item del Menú
        WITH OBJECT ( oMnuItem := MenuAddItem( ,,,,, cFileBmp, cResName, oApp:oMainMenu ) )

            :cPrompt        := ALLTRIM( oQry:FLDTXTOPT )
            :cMsg       := ALLTRIM( oQry:FLDMESSAGE )
            :bAction        := fAction
            :lActive        := ( lRight:= Eval( {||&amp;( ALLTRIM( oQry:FLDCONDI ) )} ) )
            :lBreak     := FALSE
            :lChecked       := oQry:FLDCHECK
            :lHelp      := FALSE
            :oMenu      := oApp:oMainMenu
            :nVKState       := oQry:FLDNSTATE
            :nVirtKey       := oQry:FLDVIRTKEY

        END

        IF ! EMPTY( ALLTRIM(oQry:FLDSYS) )
            AAdd( oApp:aObjSys, oMnuItem )
        ENDIF

        cValue := ALLTRIM(oQry:CNEXT)       &amp;&amp; Próximo nivel a crear o eliminar

        nEnd := oQry:NLOOP
        FOR nL := 1 TO nEnd

            DO CASE
                CASE Left(cValue,1) = 'M'
                    MenuBegin( ,,, TRUE )
                CASE Left(cValue,1) = 'E'
                    MenuEnd()
            END CASE

        NEXT nL

        oQry:Skip()
    NEXT i

    MENUITEM 'A&amp;yuda' HELP
        MENU
            MENUITEM '&amp;Help' ACTION HelpIndex() RESOURCE 'AYUDA1'   
            MENUITEM '&amp;Acerca de...' ACTION fAbout()
        ENDMENU
    ENDMENU

    oQry:End()

ENDIF

RELEASE lRight, cValue, cMnuAlias, i, nRec, fAction, cFileBmp, cResName, cQry

RETURN oApp:oMainMenu
