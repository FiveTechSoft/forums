static function EditGetkeyDown(Self, nKey)

local lExit

lExit := .f.

do case
case nKey == VK_ESCAPE
lExit := .T.

If !GetKeyState( VK_CONTROL )
lExit := .T.

Endif
Else
lExit := .T.

Endif

Endcase
