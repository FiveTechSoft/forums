Aquí te va un ejemplo de un fichero .rc con un control TGraph que uso sin problemas.

// RESOURCE SCRIPT generated by "Pelles C for Windows, version 9.00".

#include <windows.h>
#include <commctrl.h>
#include <richedit.h>

LANGUAGE LANG_ENGLISH,SUBLANG_ENGLISH_US

ESTADISTICA DIALOG DISCARDABLE 83, 81, 523, 297
STYLE WS_CHILD|DS_3DLOOK|WS_VISIBLE
FONT 8, "Segoe UI"
{
  CONTROL "TGraph", 111, "TGraph", WS_TABSTOP|0x000000a1, 0, 0, 261, 144
}
