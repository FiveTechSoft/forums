Modesto,

$(HDIR)\bin\harbour $< /L /N /W /Oobj64\ /I.\include;$(HDIR)\include
$(BCCDIR)\bin\bcc64 -c -I$(HDIR)\include -I$(BCCDIR)\include\windows\sdk -I$(BCCDIR)\include\windows\crtl -Iinclude -Imysql\include64 -oobj64\$&.obj obj64\$&.c
$(BCCDIR)\bin\tlib64 .\lib\five64.a -+$@ /0 /P64,,
