Buenas devtuxtla, el siguiente codigo me funciona correctamente:



Code: Select all Collapse
static function GeneraCorreo(Serie,Numero,Correo,PersonaContacto,NombreProveedor,textolibre,cFicheros)
    local oOutMail, cIP,cMensaje:=""
    cMensaje:="A/A " + NombreProveedor + chr(10) + chr(13)
    cMensaje+="   Adjuntamos a este mensaje " + alltrim(str(len(cFicheros))) + " pagina(s) correspondientes al pedido nº "+ serie + "/" + numero + ". " + + chr(10) + chr(13)
    cMensaje+=" Ruego cualquier anomalía, se pongan en contacto con nosotros." + chr(10) + chr(13) + chr(10) + chr(13)
    if len(alltrim(textolibre))<>0
        cMensaje+=textolibre +  chr(10) + chr(13) + chr(10) + chr(13)    
    end if
    cMensaje+=" Gracias" + chr(10) + chr(13)
    cdi:oWnd:SetMsg( "Enviando correo..." )
    WSAStartup()
    oOutMail := TSmtp():New( cIP := GetHostByName( alltrim(cdi:vSmtp) ) ) <-- Tu Servidor SMTP
    oOutMail:bConnecting = { || cdi:oWnd:SetMsg( "Conectando con el servidor de correo..." ) }
    oOutMail:bConnected  = { || cdi:oWnd:SetMsg( "Conectado" ) }
    oOutMail:bDone       = { || cdi:oWnd:SetMsg( "Mensaje enviado satisfactoriamente" ) }
    oOutMail:SendMail( "tudireccion@loquesea",;     // From
                      { "laotradireccion@loquesea"},; // To
                      cMensaje,;              // Msg Text
                      "Subject",; // Subject
                      cFicheros )  // attached files   <--> array de ficheros adjuntos
return
