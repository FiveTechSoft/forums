// Function to load group IDs and names for the DBCombo
function LoadGroups(oCn)
   local oRs, aGroups

   oRs :=oCn:rowset("SELECT groups_id, groups_name FROM groups")

   aGroups := {}
   oRs:MoveFirst()
   do while !oRs:EOF
      AAdd(aGroups, { oRs:Fields("groups_id"):Value, oRs:Fields("groups_name"):Value })
      oRs:MoveNext()
   enddo

   oRs:Close()
   return aGroups
