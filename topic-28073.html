   If GetKeyState(VK_SHIFT) // Shift
      ? "Shift+"  // funciona

      ? "Ctrl+"  // funciona
   end

   If GetKeyState(VK_MENU) // Alt
      ? "Alt+"  // No funciona
   end

   DO CASE
   CASE nKey = VK_F9

      ? [nkey 1: ], nKey  // funciona

   CASE ( nKey == ASC( "A" ) .OR. nKey == ASC( "a" ) ) .AND. ;
          GetKeyState( VK_CONTROL )  // No funciona
