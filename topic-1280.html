Eccoti un esempio funzionante:

Code: Select all Collapse
#define wdPrintView  3


FUNCTION MAIN()

    LOCAL oWord

    oWord = CreateObject( "Word.Application" )

    oWord:Documents:Add()

    oWord:ActiveDocument:ActiveWindow:View:Type = wdPrintView

    oWord:Visible = .T.

    RETURN NIL

oWord:Visible:=.t.

Prova così:

Code: Select all Collapse
#define wdPrintView  3


FUNCTION MAIN()

    LOCAL oWord

    TRY
        oWord = GETACTIVEOBJECT( "Word.Application" )
    CATCH
        oWord = CREATEOBJECT( "Word.Application" )
    END

    oWord:Documents:Add()

    oWord:ActiveDocument:ActiveWindow:View:Type = wdPrintView

    oWord:Visible = .T.

    RETURN NIL
