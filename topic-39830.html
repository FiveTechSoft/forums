This works for me.

waituntil.prg

Code (fw): Select all Collapse
#include "fivewin.ch"

function Main()

   local cDay  := HB_ARGV( 1 )
   local nSec  := TIMETOSEC( HB_ARGV( 2 ) )
   local n

   do while !( CDOW( Date() ) = cDay )
      Sleep( ( 86401 - Seconds() ) * 1000 )
   enddo

   do while ( n := Seconds() ) < nSec
      Sleep( ( nSec - n ) * 1000 )
   enddo

   ? "Start work"

return nil
