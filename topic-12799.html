Function AjustaZoom(oBmp,cAccion)
local nZoom:=oBmp:Zoom()

do case
   case cAccion == "S"
        oBmp:lStretch := !oBmp:lStretch
        oBmp:ScrollAdjust()
        oBmp:Refresh( .T. )
   case cAccion == "R"
        if nZoom*10>=1
           oBmp:lStretch := .F.
           nZoom:=nZoom-(0.10)
           oBmp:Zoom(nZoom)
           oBmp:Refresh()
           oBmp:ScrollAdjust()
        endif
   case cAccion == "A"
        oBmp:lStretch := .F.
        nZoom:=nZoom+0.10
        oBmp:Zoom(nZoom)
        oBmp:Refresh()
        oBmp:ScrollAdjust()
endcase

return NIL
