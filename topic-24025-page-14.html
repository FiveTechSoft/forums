Code (fw): Select all Collapse
METHOD SetZoom( nZ ) CLASS TScintilla

    Local  nZoomFactor := ::SendEdit( SCI_GETZOOM, 0, 0 )
    DEFAULT nZ  := 0
    if nZ > -11 .and. nZ < 21
       ::SendEdit( SCI_SETZOOM, nZ, 0 )
    endif
    nZoomFactor := ::SendEdit( SCI_GETZOOM, 0, 0 )

Return nZoomFactor
