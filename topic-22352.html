Code (fw): Select all Collapse
Function BlurEffect( oDialog )
  LOCAL hBmp 
  LOCAL hBlur
  
  oDialog:GetDC()
  
  hBmp := MakeBkBmp( oDialog:hWnd , oDialog:hDC )
  hBlur := Blur( hBmp , 12 )
  
  PalBmpDraw( oDialog:hDC , 0 , 0 , hBlur )

  DeleteObject( hBmp )
  DeleteObject( hBlur )

  oDialog:ReleaseDC()

Function BlurEffect( oDialog )

Function BlurEffect( oDialog , ForWard)
  LOCAL hBmp, i
  LOCAL n := 25
  LOCAL hBlur

  oDialog:GetDC()

  hBmp := MakeBkBmp( oDialog:hWnd , oDialog:hDC )

  FOR i := 1 TO 25
  
  If( !ForWard, n := n - 1 , )
  
  hBlur := Blur( hBmp , If( ForWard, i , n ) )

  PalBmpDraw( oDialog:hDC , 0 , 0 , hBlur )

  NEXT
  //MsgInfo( "Probando Empañado de Pantalla..." )
  //oDialog:refresh()

  DeleteObject( hBmp )
